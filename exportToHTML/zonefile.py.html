41   class CNAMEAndOtherData(dns.exception.DNSException):
42       """A node has a CNAME and other data"""
43   
44   
45   def _check_cname_and_other_data(txn, name, rdataset):
46       rdataset_kind = dns.node.NodeKind.classify_rdataset(rdataset)
47       node = txn.get_node(name)
48       if node is None:
49           # empty nodes are neutral.
50           return
51       node_kind = node.classify()
52       if node_kind == dns.node.NodeKind.CNAME and \
53          rdataset_kind == dns.node.NodeKind.REGULAR:
54           raise CNAMEAndOtherData('rdataset type is not compatible with a '
55                                   'CNAME node')
56       elif node_kind == dns.node.NodeKind.REGULAR and \
57          rdataset_kind == dns.node.NodeKind.CNAME:
58           raise CNAMEAndOtherData('CNAME rdataset is not compatible with a '
59                                   'regular data node')
60       # Otherwise at least one of the node and the rdataset is neutral, so
61       # adding the rdataset is ok

68       def __init__(self, tok, rdclass, txn, allow_include=False,
69                    allow_directives=True, force_name=None,
70                    force_ttl=None, force_rdclass=None, force_rdtype=None,
71                    default_ttl=None):
72           self.tok = tok
73           (self.zone_origin, self.relativize, _) = \
74               txn.manager.origin_information()
75           self.current_origin = self.zone_origin
76           self.last_ttl = 0
77           self.last_ttl_known = False
78           if force_ttl is not None:
79               default_ttl = force_ttl
80           if default_ttl is None:
81               self.default_ttl = 0
82               self.default_ttl_known = False
83           else:
84               self.default_ttl = default_ttl
85               self.default_ttl_known = True
86           self.last_name = self.current_origin
87           self.zone_rdclass = rdclass
88           self.txn = txn
89           self.saved_state = []
90           self.current_file = None
91           self.allow_include = allow_include
92           self.allow_directives = allow_directives
93           self.force_name = force_name
94           self.force_ttl = force_ttl
95           self.force_rdclass = force_rdclass
96           self.force_rdtype = force_rdtype
97           self.txn.check_put_rdataset(_check_cname_and_other_data)

99       def _eat_line(self):
100          while 1:
101              token = self.tok.get()
102              if token.is_eol_or_eof():
103                  break

105      def _get_identifier(self):
106          token = self.tok.get()
107          if not token.is_identifier():
108              raise dns.exception.SyntaxError
109          return token
110  
111      def _rr_line(self):
112          """Process one line from a DNS zone file."""
113          token = None
114          # Name
115          if self.force_name is not None:
116              name = self.force_name
117          else:
118              if self.current_origin is None:
119                  raise UnknownOrigin
120              token = self.tok.get(want_leading=True)
121              if not token.is_whitespace():
122                  self.last_name = self.tok.as_name(token, self.current_origin)
123              else:
124                  token = self.tok.get()
125                  if token.is_eol_or_eof():
126                      # treat leading WS followed by EOL/EOF as if they were EOL/EOF.
127                      return
128                  self.tok.unget(token)
129              name = self.last_name
130              if not name.is_subdomain(self.zone_origin):
131                  self._eat_line()
132                  return
133              if self.relativize:
134                  name = name.relativize(self.zone_origin)
135  
136          # TTL
137          if self.force_ttl is not None:
138              ttl = self.force_ttl
139              self.last_ttl = ttl
140              self.last_ttl_known = True
141          else:
142              token = self._get_identifier()
143              ttl = None
144              try:
145                  ttl = dns.ttl.from_text(token.value)
146                  self.last_ttl = ttl
147                  self.last_ttl_known = True
148                  token = None
149              except dns.ttl.BadTTL:
150                  if self.default_ttl_known:
151                      ttl = self.default_ttl
152                  elif self.last_ttl_known:
153                      ttl = self.last_ttl
154                  self.tok.unget(token)
155  
156          # Class
157          if self.force_rdclass is not None:
158              rdclass = self.force_rdclass
159          else:
160              token = self._get_identifier()
161              try:
162                  rdclass = dns.rdataclass.from_text(token.value)
163              except dns.exception.SyntaxError:
164                  raise
165              except Exception:
166                  rdclass = self.zone_rdclass
167                  self.tok.unget(token)
168              if rdclass != self.zone_rdclass:
169                  raise dns.exception.SyntaxError("RR class is not zone's class")
170  
171          # Type
172          if self.force_rdtype is not None:
173              rdtype = self.force_rdtype
174          else:
175              token = self._get_identifier()
176              try:
177                  rdtype = dns.rdatatype.from_text(token.value)
178              except Exception:
179                  raise dns.exception.SyntaxError(
180                      "unknown rdatatype '%s'" % token.value)
181  
182          try:
183              rd = dns.rdata.from_text(rdclass, rdtype, self.tok,
184                                       self.current_origin, self.relativize,
185                                       self.zone_origin)
186          except dns.exception.SyntaxError:
187              # Catch and reraise.
188              raise
189          except Exception:
190              # All exceptions that occur in the processing of rdata
191              # are treated as syntax errors.  This is not strictly
192              # correct, but it is correct almost all of the time.
193              # We convert them to syntax errors so that we can emit
194              # helpful filename:line info.
195              (ty, va) = sys.exc_info()[:2]
196              raise dns.exception.SyntaxError(
197                  "caught exception {}: {}".format(str(ty), str(va)))
198  
199          if not self.default_ttl_known and rdtype == dns.rdatatype.SOA:
200              # The pre-RFC2308 and pre-BIND9 behavior inherits the zone default
201              # TTL from the SOA minttl if no $TTL statement is present before the
202              # SOA is parsed.
203              self.default_ttl = rd.minimum
204              self.default_ttl_known = True
205              if ttl is None:
206                  # if we didn't have a TTL on the SOA, set it!
207                  ttl = rd.minimum
208  
209          # TTL check.  We had to wait until now to do this as the SOA RR's
210          # own TTL can be inferred from its minimum.
211          if ttl is None:
212              raise dns.exception.SyntaxError("Missing default TTL value")
213  
214          self.txn.add(name, ttl, rd)

216      def _parse_modify(self, side):
217          # Here we catch everything in '{' '}' in a group so we can replace it
218          # with ''.
219          is_generate1 = re.compile(r"^.*\$({(\+|-?)(\d+),(\d+),(.)}).*$")
220          is_generate2 = re.compile(r"^.*\$({(\+|-?)(\d+)}).*$")
221          is_generate3 = re.compile(r"^.*\$({(\+|-?)(\d+),(\d+)}).*$")
222          # Sometimes there are modifiers in the hostname. These come after
223          # the dollar sign. They are in the form: ${offset[,width[,base]]}.
224          # Make names
225          g1 = is_generate1.match(side)
226          if g1:
227              mod, sign, offset, width, base = g1.groups()
228              if sign == '':
229                  sign = '+'
230          g2 = is_generate2.match(side)
231          if g2:
232              mod, sign, offset = g2.groups()
233              if sign == '':
234                  sign = '+'
235              width = 0
236              base = 'd'
237          g3 = is_generate3.match(side)
238          if g3:
239              mod, sign, offset, width = g3.groups()
240              if sign == '':
241                  sign = '+'
242              base = 'd'
243  
244          if not (g1 or g2 or g3):
245              mod = ''
246              sign = '+'
247              offset = 0
248              width = 0
249              base = 'd'
250  
251          if base != 'd':
252              raise NotImplementedError()
253  
254          return mod, sign, offset, width, base

256      def _generate_line(self):
257          # range lhs [ttl] [class] type rhs [ comment ]
258          """Process one line containing the GENERATE statement from a DNS 
259          zone file."""
260          if self.current_origin is None:
261              raise UnknownOrigin
262  
263          token = self.tok.get()
264          # Range (required)
265          try:
266              start, stop, step = dns.grange.from_text(token.value)
267              token = self.tok.get()
268              if not token.is_identifier():
269                  raise dns.exception.SyntaxError
270          except Exception:
271              raise dns.exception.SyntaxError
272  
273          # lhs (required)
274          try:
275              lhs = token.value
276              token = self.tok.get()
277              if not token.is_identifier():
278                  raise dns.exception.SyntaxError
279          except Exception:
280              raise dns.exception.SyntaxError
281  
282          # TTL
283          try:
284              ttl = dns.ttl.from_text(token.value)
285              self.last_ttl = ttl
286              self.last_ttl_known = True
287              token = self.tok.get()
288              if not token.is_identifier():
289                  raise dns.exception.SyntaxError
290          except dns.ttl.BadTTL:
291              if not (self.last_ttl_known or self.default_ttl_known):
292                  raise dns.exception.SyntaxError("Missing default TTL value")
293              if self.default_ttl_known:
294                  ttl = self.default_ttl
295              elif self.last_ttl_known:
296                  ttl = self.last_ttl
297          # Class
298          try:
299              rdclass = dns.rdataclass.from_text(token.value)
300              token = self.tok.get()
301              if not token.is_identifier():
302                  raise dns.exception.SyntaxError
303          except dns.exception.SyntaxError:
304              raise dns.exception.SyntaxError
305          except Exception:
306              rdclass = self.zone_rdclass
307          if rdclass != self.zone_rdclass:
308              raise dns.exception.SyntaxError("RR class is not zone's class")
309          # Type
310          try:
311              rdtype = dns.rdatatype.from_text(token.value)
312              token = self.tok.get()
313              if not token.is_identifier():
314                  raise dns.exception.SyntaxError
315          except Exception:
316              raise dns.exception.SyntaxError("unknown rdatatype '%s'" %
317                                              token.value)
318  
319          # rhs (required)
320          rhs = token.value
321  
322          # The code currently only supports base 'd', so the last value
323          # in the tuple _parse_modify returns is ignored
324          lmod, lsign, loffset, lwidth, _ = self._parse_modify(lhs)
325          rmod, rsign, roffset, rwidth, _ = self._parse_modify(rhs)
326          for i in range(start, stop + 1, step):
327              # +1 because bind is inclusive and python is exclusive
328  
329              if lsign == '+':
330                  lindex = i + int(loffset)
331              elif lsign == '-':
332                  lindex = i - int(loffset)
333  
334              if rsign == '-':
335                  rindex = i - int(roffset)
336              elif rsign == '+':
337                  rindex = i + int(roffset)
338  
339              lzfindex = str(lindex).zfill(int(lwidth))
340              rzfindex = str(rindex).zfill(int(rwidth))
341  
342              name = lhs.replace('$%s' % (lmod), lzfindex)
343              rdata = rhs.replace('$%s' % (rmod), rzfindex)
344  
345              self.last_name = dns.name.from_text(name, self.current_origin,
346                                                  self.tok.idna_codec)
347              name = self.last_name
348              if not name.is_subdomain(self.zone_origin):
349                  self._eat_line()
350                  return
351              if self.relativize:
352                  name = name.relativize(self.zone_origin)
353  
354              try:
355                  rd = dns.rdata.from_text(rdclass, rdtype, rdata,
356                                           self.current_origin, self.relativize,
357                                           self.zone_origin)
358              except dns.exception.SyntaxError:
359                  # Catch and reraise.
360                  raise
361              except Exception:
362                  # All exceptions that occur in the processing of rdata
363                  # are treated as syntax errors.  This is not strictly
364                  # correct, but it is correct almost all of the time.
365                  # We convert them to syntax errors so that we can emit
366                  # helpful filename:line info.
367                  (ty, va) = sys.exc_info()[:2]
368                  raise dns.exception.SyntaxError("caught exception %s: %s" %
369                                                  (str(ty), str(va)))
370  
371              self.txn.add(name, ttl, rd)
372  
373      def read(self):
374          """Read a DNS zone file and build a zone object. 
375   
376          @raises dns.zone.NoSOA: No SOA RR was found at the zone origin 
377          @raises dns.zone.NoNS: No NS RRset was found at the zone origin 
378          """
379  
380          try:
381              while 1:
382                  token = self.tok.get(True, True)
383                  if token.is_eof():
384                      if self.current_file is not None:
385                          self.current_file.close()
386                      if len(self.saved_state) > 0:
387                          (self.tok,
388                           self.current_origin,
389                           self.last_name,
390                           self.current_file,
391                           self.last_ttl,
392                           self.last_ttl_known,
393                           self.default_ttl,
394                           self.default_ttl_known) = self.saved_state.pop(-1)
395                          continue
396                      break
397                  elif token.is_eol():
398                      continue
399                  elif token.is_comment():
400                      self.tok.get_eol()
401                      continue
402                  elif token.value[0] == '$' and self.allow_directives:
403                      c = token.value.upper()
404                      if c == '$TTL':
405                          token = self.tok.get()
406                          if not token.is_identifier():
407                              raise dns.exception.SyntaxError("bad $TTL")
408                          self.default_ttl = dns.ttl.from_text(token.value)
409                          self.default_ttl_known = True
410                          self.tok.get_eol()
411                      elif c == '$ORIGIN':
412                          self.current_origin = self.tok.get_name()
413                          self.tok.get_eol()
414                          if self.zone_origin is None:
415                              self.zone_origin = self.current_origin
416                          self.txn._set_origin(self.current_origin)
417                      elif c == '$INCLUDE' and self.allow_include:
418                          token = self.tok.get()
419                          filename = token.value
420                          token = self.tok.get()
421                          if token.is_identifier():
422                              new_origin =\
423                                  dns.name.from_text(token.value,
424                                                     self.current_origin,
425                                                     self.tok.idna_codec)
426                              self.tok.get_eol()
427                          elif not token.is_eol_or_eof():
428                              raise dns.exception.SyntaxError(
429                                  "bad origin in $INCLUDE")
430                          else:
431                              new_origin = self.current_origin
432                          self.saved_state.append((self.tok,
433                                                   self.current_origin,
434                                                   self.last_name,
435                                                   self.current_file,
436                                                   self.last_ttl,
437                                                   self.last_ttl_known,
438                                                   self.default_ttl,
439                                                   self.default_ttl_known))
440                          self.current_file = open(filename, 'r')
441                          self.tok = dns.tokenizer.Tokenizer(self.current_file,
442                                                             filename)
443                          self.current_origin = new_origin
444                      elif c == '$GENERATE':
445                          self._generate_line()
446                      else:
447                          raise dns.exception.SyntaxError(
448                              "Unknown zone file directive '" + c + "'")
449                      continue
450                  self.tok.unget(token)
451                  self._rr_line()
452          except dns.exception.SyntaxError as detail:
453              (filename, line_number) = self.tok.where()
454              if detail is None:
455                  detail = "syntax error"
456              ex = dns.exception.SyntaxError(
457                  "%s:%d: %s" % (filename, line_number, detail))
458              tb = sys.exc_info()[2]
459              raise ex.with_traceback(tb) from None

462  class RRsetsReaderTransaction(dns.transaction.Transaction):
463  
464      def __init__(self, manager, replacement, read_only):
465          assert not read_only
466          super().__init__(manager, replacement, read_only)
467          self.rdatasets = {}
468  
469      def _get_rdataset(self, name, rdtype, covers):
470          return self.rdatasets.get((name, rdtype, covers))
471  
472      def _get_node(self, name):
473          rdatasets = []
474          for (rdataset_name, _, _), rdataset in self.rdatasets.items():
475              if name == rdataset_name:
476                  rdatasets.append(rdataset)
477          if len(rdatasets) == 0:
478              return None
479          node = dns.node.Node()
480          node.rdatasets = rdatasets
481          return node
482  
483      def _put_rdataset(self, name, rdataset):
484          self.rdatasets[(name, rdataset.rdtype, rdataset.covers)] = rdataset
485  
486      def _delete_name(self, name):
487          # First remove any changes involving the name
488          remove = []
489          for key in self.rdatasets:
490              if key[0] == name:
491                  remove.append(key)
492          if len(remove) > 0:
493              for key in remove:
494                  del self.rdatasets[key]
495  
496      def _delete_rdataset(self, name, rdtype, covers):
497          try:
498              del self.rdatasets[(name, rdtype, covers)]
499          except KeyError:
500              pass
501  
502      def _name_exists(self, name):
503          for (n, _, _) in self.rdatasets:
504              if n == name:
505                  return True
506          return False
507  
508      def _changed(self):
509          return len(self.rdatasets) > 0
510  
511      def _end_transaction(self, commit):
512          if commit and self._changed():
513              rrsets = []
514              for (name, _, _), rdataset in self.rdatasets.items():
515                  rrset = dns.rrset.RRset(name, rdataset.rdclass, rdataset.rdtype,
516                                          rdataset.covers)
517                  rrset.update(rdataset)
518                  rrsets.append(rrset)
519              self.manager.set_rrsets(rrsets)
520  
521      def _set_origin(self, origin):
522          pass

525  class RRSetsReaderManager(dns.transaction.TransactionManager):
526      def __init__(self, origin=dns.name.root, relativize=False,
527                   rdclass=dns.rdataclass.IN):
528          self.origin = origin
529          self.relativize = relativize
530          self.rdclass = rdclass
531          self.rrsets = []
532  
533      def writer(self, replacement=False):
534          assert replacement is True
535          return RRsetsReaderTransaction(self, True, False)
536  
537      def get_class(self):
538          return self.rdclass
539  
540      def origin_information(self):
541          if self.relativize:
542              effective = dns.name.empty
543          else:
544              effective = self.origin
545          return (self.origin, self.relativize, effective)
546  
547      def set_rrsets(self, rrsets):
548          self.rrsets = rrsets

551  def read_rrsets(text, name=None, ttl=None, rdclass=dns.rdataclass.IN,
552                  default_rdclass=dns.rdataclass.IN,
553                  rdtype=None, default_ttl=None, idna_codec=None,
554                  origin=dns.name.root, relativize=False):
555      """Read one or more rrsets from the specified text, possibly subject 
556      to restrictions. 
557   
558      *text*, a file object or a string, is the input to process. 
559   
560      *name*, a string, ``dns.name.Name``, or ``None``, is the owner name of 
561      the rrset.  If not ``None``, then the owner name is "forced", and the 
562      input must not specify an owner name.  If ``None``, then any owner names 
563      are allowed and must be present in the input. 
564   
565      *ttl*, an ``int``, string, or None.  If not ``None``, the the TTL is 
566      forced to be the specified value and the input must not specify a TTL. 
567      If ``None``, then a TTL may be specified in the input.  If it is not 
568      specified, then the *default_ttl* will be used. 
569   
570      *rdclass*, a ``dns.rdataclass.RdataClass``, string, or ``None``.  If 
571      not ``None``, then the class is forced to the specified value, and the 
572      input must not specify a class.  If ``None``, then the input may specify 
573      a class that matches *default_rdclass*.  Note that it is not possible to 
574      return rrsets with differing classes; specifying ``None`` for the class 
575      simply allows the user to optionally type a class as that may be convenient 
576      when cutting and pasting. 
577   
578      *default_rdclass*, a ``dns.rdataclass.RdataClass`` or string.  The class 
579      of the returned rrsets. 
580   
581      *rdtype*, a ``dns.rdatatype.RdataType``, string, or ``None``.  If not 
582      ``None``, then the type is forced to the specified value, and the 
583      input must not specify a type.  If ``None``, then a type must be present 
584      for each RR. 
585   
586      *default_ttl*, an ``int``, string, or ``None``.  If not ``None``, then if 
587      the TTL is not forced and is not specified, then this value will be used. 
588      if ``None``, then if the TTL is not forced an error will occur if the TTL 
589      is not specified. 
590   
591      *idna_codec*, a ``dns.name.IDNACodec``, specifies the IDNA 
592      encoder/decoder.  If ``None``, the default IDNA 2003 encoder/decoder 
593      is used.  Note that codecs only apply to the owner name; dnspython does 
594      not do IDNA for names in rdata, as there is no IDNA zonefile format. 
595   
596      *origin*, a string, ``dns.name.Name``, or ``None``, is the origin for any 
597      relative names in the input, and also the origin to relativize to if 
598      *relativize* is ``True``. 
599   
600      *relativize*, a bool.  If ``True``, names are relativized to the *origin*; 
601      if ``False`` then any relative names in the input are made absolute by 
602      appending the *origin*. 
603      """
604      if isinstance(origin, str):
605          origin = dns.name.from_text(origin, dns.name.root, idna_codec)
606      if isinstance(name, str):
607          name = dns.name.from_text(name, origin, idna_codec)
608      if isinstance(ttl, str):
609          ttl = dns.ttl.from_text(ttl)
610      if isinstance(default_ttl, str):
611          default_ttl = dns.ttl.from_text(default_ttl)
612      if rdclass is not None:
613          rdclass = dns.rdataclass.RdataClass.make(rdclass)
614      default_rdclass = dns.rdataclass.RdataClass.make(default_rdclass)
615      if rdtype is not None:
616          rdtype = dns.rdatatype.RdataType.make(rdtype)
617      manager = RRSetsReaderManager(origin, relativize, default_rdclass)
618      with manager.writer(True) as txn:
619          tok = dns.tokenizer.Tokenizer(text, '<input>', idna_codec=idna_codec)
620          reader = Reader(tok, default_rdclass, txn, allow_directives=False,
621                          force_name=name, force_ttl=ttl, force_rdclass=rdclass,
622                          force_rdtype=rdtype, default_ttl=default_ttl)
623          reader.read()
