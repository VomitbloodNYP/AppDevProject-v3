18   import base64
19   import struct
20   
21   import dns.exception
22   import dns.immutable
23   import dns.rcode
24   import dns.rdata
25   
26   
27   @dns.immutable.immutable
28   class TSIG(dns.rdata.Rdata):
29   
30       """TSIG record"""
31   
32       __slots__ = ['algorithm', 'time_signed', 'fudge', 'mac',
33                    'original_id', 'error', 'other']
34   
35       def __init__(self, rdclass, rdtype, algorithm, time_signed, fudge, mac,
36                    original_id, error, other):
37           """Initialize a TSIG rdata. 
38    
39           *rdclass*, an ``int`` is the rdataclass of the Rdata. 
40    
41           *rdtype*, an ``int`` is the rdatatype of the Rdata. 
42    
43           *algorithm*, a ``dns.name.Name``. 
44    
45           *time_signed*, an ``int``. 
46    
47           *fudge*, an ``int`. 
48    
49           *mac*, a ``bytes`` 
50    
51           *original_id*, an ``int`` 
52    
53           *error*, an ``int`` 
54    
55           *other*, a ``bytes`` 
56           """
57   
58           super().__init__(rdclass, rdtype)
59           self.algorithm = self._as_name(algorithm)
60           self.time_signed = self._as_uint48(time_signed)
61           self.fudge = self._as_uint16(fudge)
62           self.mac = self._as_bytes(mac)
63           self.original_id = self._as_uint16(original_id)
64           self.error = dns.rcode.Rcode.make(error)
65           self.other = self._as_bytes(other)
66   
67       def to_text(self, origin=None, relativize=True, **kw):
68           algorithm = self.algorithm.choose_relativity(origin, relativize)
69           error = dns.rcode.to_text(self.error, True)
70           text = f"{algorithm} {self.time_signed} {self.fudge} " + \
71                  f"{len(self.mac)} {dns.rdata._base64ify(self.mac, 0)} " + \
72                  f"{self.original_id} {error} {len(self.other)}"
73           if self.other:
74               text += f" {dns.rdata._base64ify(self.other, 0)}"
75           return text
76   
77       @classmethod
78       def from_text(cls, rdclass, rdtype, tok, origin=None, relativize=True,
79                     relativize_to=None):
80           algorithm = tok.get_name(relativize=False)
81           time_signed = tok.get_uint48()
82           fudge = tok.get_uint16()
83           mac_len = tok.get_uint16()
84           mac = base64.b64decode(tok.get_string())
85           if len(mac) != mac_len:
86               raise SyntaxError('invalid MAC')
87           original_id = tok.get_uint16()
88           error = dns.rcode.from_text(tok.get_string())
89           other_len = tok.get_uint16()
90           if other_len > 0:
91               other = base64.b64decode(tok.get_string())
92               if len(other) != other_len:
93                   raise SyntaxError('invalid other data')
94           else:
95               other = b''
96           return cls(rdclass, rdtype, algorithm, time_signed, fudge, mac,
97                      original_id, error, other)
98   
99       def _to_wire(self, file, compress=None, origin=None, canonicalize=False):
100          self.algorithm.to_wire(file, None, origin, False)
101          file.write(struct.pack('!HIHH',
102                                 (self.time_signed >> 32) & 0xffff,
103                                 self.time_signed & 0xffffffff,
104                                 self.fudge,
105                                 len(self.mac)))
106          file.write(self.mac)
107          file.write(struct.pack('!HHH', self.original_id, self.error,
108                                 len(self.other)))
109          file.write(self.other)
110  
111      @classmethod
112      def from_wire_parser(cls, rdclass, rdtype, parser, origin=None):
113          algorithm = parser.get_name()
114          time_signed = parser.get_uint48()
115          fudge = parser.get_uint16()
116          mac = parser.get_counted_bytes(2)
117          (original_id, error) = parser.get_struct('!HH')
118          other = parser.get_counted_bytes(2)
119          return cls(rdclass, rdtype, algorithm, time_signed, fudge, mac,
120                     original_id, error, other)
