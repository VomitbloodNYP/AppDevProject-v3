103  def _compute_times(timeout):
104      now = time.time()
105      if timeout is None:
106          return (now, None)
107      else:
108          return (now, now + timeout)

111  def _wait_for(fd, readable, writable, _, expiration):
112      # Use the selected selector class to wait for any of the specified
113      # events.  An "expiration" absolute time is converted into a relative
114      # timeout.
115      #
116      # The unused parameter is 'error', which is always set when
117      # selecting for read or write, and we have no error-only selects.
118  
119      if readable and isinstance(fd, ssl.SSLSocket) and fd.pending() > 0:
120          return True
121      sel = _selector_class()
122      events = 0
123      if readable:
124          events |= selectors.EVENT_READ
125      if writable:
126          events |= selectors.EVENT_WRITE
127      if events:
128          sel.register(fd, events)
129      if expiration is None:
130          timeout = None
131      else:
132          timeout = expiration - time.time()
133          if timeout <= 0.0:
134              raise dns.exception.Timeout
135      if not sel.select(timeout):
136          raise dns.exception.Timeout

163  def _addresses_equal(af, a1, a2):
164      # Convert the first value of the tuple, which is a textual format
165      # address into binary form, so that we are not confused by different
166      # textual representations of the same address
167      try:
168          n1 = dns.inet.inet_pton(af, a1[0])
169          n2 = dns.inet.inet_pton(af, a2[0])
170      except dns.exception.SyntaxError:
171          return False
172      return n1 == n2 and a1[1:] == a2[1:]

175  def _matches_destination(af, from_address, destination, ignore_unexpected):
176      # Check that from_address is appropriate for a response to a query
177      # sent to destination.
178      if not destination:
179          return True
180      if _addresses_equal(af, from_address, destination) or \
181         (dns.inet.is_multicast(destination[0]) and
182          from_address[1:] == destination[1:]):
183          return True
184      elif ignore_unexpected:
185          return False
186      raise UnexpectedSource(f'got a response from {from_address} instead of '
187                             f'{destination}')

190  def _destination_and_source(where, port, source, source_port,
191                              where_must_be_address=True):
192      # Apply defaults and compute destination and source tuples
193      # suitable for use in connect(), sendto(), or bind().
194      af = None
195      destination = None
196      try:
197          af = dns.inet.af_for_address(where)
198          destination = where
199      except Exception:
200          if where_must_be_address:
201              raise
202          # URLs are ok so eat the exception
203      if source:
204          saf = dns.inet.af_for_address(source)
205          if af:
206              # We know the destination af, so source had better agree!
207              if saf != af:
208                  raise ValueError('different address families for source ' +
209                                   'and destination')
210          else:
211              # We didn't know the destination af, but we know the source,
212              # so that's our af.
213              af = saf
214      if source_port and not source:
215          # Caller has specified a source_port but not an address, so we
216          # need to return a source, and we need to use the appropriate
217          # wildcard address as the address.
218          if af == socket.AF_INET:
219              source = '0.0.0.0'
220          elif af == socket.AF_INET6:
221              source = '::'
222          else:
223              raise ValueError('source_port specified but address family is '
224                               'unknown')
225      # Convert high-level (address, port) tuples into low-level address
226      # tuples.
227      if destination:
228          destination = dns.inet.low_level_address_tuple((destination, port), af)
229      if source:
230          source = dns.inet.low_level_address_tuple((source, source_port), af)
231      return (af, destination, source)

233  def _make_socket(af, type, source, ssl_context=None, server_hostname=None):
234      s = socket_factory(af, type)
235      try:
236          s.setblocking(False)
237          if source is not None:
238              s.bind(source)
239          if ssl_context:
240              return ssl_context.wrap_socket(s, do_handshake_on_connect=False,
241                                             server_hostname=server_hostname)
242          else:
243              return s
244      except Exception:
245          s.close()
246          raise

248  def https(q, where, timeout=None, port=443, source=None, source_port=0,
249            one_rr_per_rrset=False, ignore_trailing=False,
250            session=None, path='/dns-query', post=True,
251            bootstrap_address=None, verify=True):
252      """Return the response obtained after sending a query via DNS-over-HTTPS. 
253   
254      *q*, a ``dns.message.Message``, the query to send. 
255   
256      *where*, a ``str``, the nameserver IP address or the full URL. If an IP 
257      address is given, the URL will be constructed using the following schema: 
258      https://<IP-address>:<port>/<path>. 
259   
260      *timeout*, a ``float`` or ``None``, the number of seconds to 
261      wait before the query times out. If ``None``, the default, wait forever. 
262   
263      *port*, a ``int``, the port to send the query to. The default is 443. 
264   
265      *source*, a ``str`` containing an IPv4 or IPv6 address, specifying 
266      the source address.  The default is the wildcard address. 
267   
268      *source_port*, an ``int``, the port from which to send the message. 
269      The default is 0. 
270   
271      *one_rr_per_rrset*, a ``bool``. If ``True``, put each RR into its own 
272      RRset. 
273   
274      *ignore_trailing*, a ``bool``. If ``True``, ignore trailing 
275      junk at end of the received message. 
276   
277      *session*, an ``httpx.Client`` or ``requests.session.Session``.  If 
278      provided, the client/session to use to send the queries. 
279   
280      *path*, a ``str``. If *where* is an IP address, then *path* will be used to 
281      construct the URL to send the DNS query to. 
282   
283      *post*, a ``bool``. If ``True``, the default, POST method will be used. 
284   
285      *bootstrap_address*, a ``str``, the IP address to use to bypass the 
286      system's DNS resolver. 
287   
288      *verify*, a ``str``, containing a path to a certificate file or directory. 
289   
290      Returns a ``dns.message.Message``. 
291      """
292  
293      if not have_doh:
294          raise NoDOH('Neither httpx nor requests is available.')  # pragma: no cover
295  
296      _httpx_ok = _have_httpx
297  
298      wire = q.to_wire()
299      (af, _, source) = _destination_and_source(where, port, source, source_port,
300                                                False)
301      transport_adapter = None
302      transport = None
303      headers = {
304          "accept": "application/dns-message"
305      }
306      if af is not None:
307          if af == socket.AF_INET:
308              url = 'https://{}:{}{}'.format(where, port, path)
309          elif af == socket.AF_INET6:
310              url = 'https://[{}]:{}{}'.format(where, port, path)
311      elif bootstrap_address is not None:
312          _httpx_ok = False
313          split_url = urllib.parse.urlsplit(where)
314          headers['Host'] = split_url.hostname
315          url = where.replace(split_url.hostname, bootstrap_address)
316          if _have_requests:
317              transport_adapter = HostHeaderSSLAdapter()
318      else:
319          url = where
320      if source is not None:
321          # set source port and source address
322          if _have_httpx:
323              if source_port == 0:
324                  transport = httpx.HTTPTransport(local_address=source[0])
325              else:
326                  _httpx_ok = False
327          if _have_requests:
328              transport_adapter = SourceAddressAdapter(source)
329  
330      if session:
331          if _have_httpx:
332              _is_httpx = isinstance(session, httpx.Client)
333          else:
334              _is_httpx = False
335          if _is_httpx and not _httpx_ok:
336              raise NoDOH('Session is httpx, but httpx cannot be used for '
337                          'the requested operation.')
338      else:
339          _is_httpx = _httpx_ok
340  
341      if not _httpx_ok and not _have_requests:
342          raise NoDOH('Cannot use httpx for this operation, and '
343                      'requests is not available.')
344  
345      with contextlib.ExitStack() as stack:
346          if not session:
347              if _is_httpx:
348                  session = stack.enter_context(httpx.Client(http1=True,
349                                                             http2=_have_http2,
350                                                             verify=verify,
351                                                             transport=transport))
352              else:
353                  session = stack.enter_context(requests.sessions.Session())
354  
355          if transport_adapter:
356              session.mount(url, transport_adapter)
357  
358          # see https://tools.ietf.org/html/rfc8484#section-4.1.1 for DoH
359          # GET and POST examples
360          if post:
361              headers.update({
362                  "content-type": "application/dns-message",
363                  "content-length": str(len(wire))
364              })
365              if _is_httpx:
366                  response = session.post(url, headers=headers, content=wire,
367                                          timeout=timeout)
368              else:
369                  response = session.post(url, headers=headers, data=wire,
370                                          timeout=timeout, verify=verify)
371          else:
372              wire = base64.urlsafe_b64encode(wire).rstrip(b"=")
373              if _is_httpx:
374                  wire = wire.decode()  # httpx does a repr() if we give it bytes
375                  response = session.get(url, headers=headers,
376                                         timeout=timeout,
377                                         params={"dns": wire})
378              else:
379                  response = session.get(url, headers=headers,
380                                         timeout=timeout, verify=verify,
381                                         params={"dns": wire})
382  
383      # see https://tools.ietf.org/html/rfc8484#section-4.2.1 for info about DoH
384      # status codes
385      if response.status_code < 200 or response.status_code > 299:
386          raise ValueError('{} responded with status code {}'
387                           '\nResponse body: {}'.format(where,
388                                                        response.status_code,
389                                                        response.content))
390      r = dns.message.from_wire(response.content,
391                                keyring=q.keyring,
392                                request_mac=q.request_mac,
393                                one_rr_per_rrset=one_rr_per_rrset,
394                                ignore_trailing=ignore_trailing)
395      r.time = response.elapsed
396      if not q.is_response(r):
397          raise BadResponse
398      return r

400  def _udp_recv(sock, max_size, expiration):
401      """Reads a datagram from the socket. 
402      A Timeout exception will be raised if the operation is not completed 
403      by the expiration time. 
404      """
405      while True:
406          try:
407              return sock.recvfrom(max_size)
408          except BlockingIOError:
409              _wait_for_readable(sock, expiration)
410

412  def _udp_send(sock, data, destination, expiration):
413      """Sends the specified datagram to destination over the socket. 
414      A Timeout exception will be raised if the operation is not completed 
415      by the expiration time. 
416      """
417      while True:
418          try:
419              if destination:
420                  return sock.sendto(data, destination)
421              else:
422                  return sock.send(data)
423          except BlockingIOError:  # pragma: no cover
424              _wait_for_writable(sock, expiration)

427  def send_udp(sock, what, destination, expiration=None):
428      """Send a DNS message to the specified UDP socket. 
429   
430      *sock*, a ``socket``. 
431   
432      *what*, a ``bytes`` or ``dns.message.Message``, the message to send. 
433   
434      *destination*, a destination tuple appropriate for the address family 
435      of the socket, specifying where to send the query. 
436   
437      *expiration*, a ``float`` or ``None``, the absolute time at which 
438      a timeout exception should be raised.  If ``None``, no timeout will 
439      occur. 
440   
441      Returns an ``(int, float)`` tuple of bytes sent and the sent time. 
442      """
443  
444      if isinstance(what, dns.message.Message):
445          what = what.to_wire()
446      sent_time = time.time()
447      n = _udp_send(sock, what, destination, expiration)
448      return (n, sent_time)

451  def receive_udp(sock, destination=None, expiration=None,
452                  ignore_unexpected=False, one_rr_per_rrset=False,
453                  keyring=None, request_mac=b'', ignore_trailing=False,
454                  raise_on_truncation=False):
455      """Read a DNS message from a UDP socket. 
456   
457      *sock*, a ``socket``. 
458   
459      *destination*, a destination tuple appropriate for the address family 
460      of the socket, specifying where the message is expected to arrive from. 
461      When receiving a response, this would be where the associated query was 
462      sent. 
463   
464      *expiration*, a ``float`` or ``None``, the absolute time at which 
465      a timeout exception should be raised.  If ``None``, no timeout will 
466      occur. 
467   
468      *ignore_unexpected*, a ``bool``.  If ``True``, ignore responses from 
469      unexpected sources. 
470   
471      *one_rr_per_rrset*, a ``bool``.  If ``True``, put each RR into its own 
472      RRset. 
473   
474      *keyring*, a ``dict``, the keyring to use for TSIG. 
475   
476      *request_mac*, a ``bytes``, the MAC of the request (for TSIG). 
477   
478      *ignore_trailing*, a ``bool``.  If ``True``, ignore trailing 
479      junk at end of the received message. 
480   
481      *raise_on_truncation*, a ``bool``.  If ``True``, raise an exception if 
482      the TC bit is set. 
483   
484      Raises if the message is malformed, if network errors occur, of if 
485      there is a timeout. 
486   
487      If *destination* is not ``None``, returns a ``(dns.message.Message, float)`` 
488      tuple of the received message and the received time. 
489   
490      If *destination* is ``None``, returns a 
491      ``(dns.message.Message, float, tuple)`` 
492      tuple of the received message, the received time, and the address where 
493      the message arrived from. 
494      """
495  
496      wire = b''
497      while True:
498          (wire, from_address) = _udp_recv(sock, 65535, expiration)
499          if _matches_destination(sock.family, from_address, destination,
500                                  ignore_unexpected):
501              break
502      received_time = time.time()
503      r = dns.message.from_wire(wire, keyring=keyring, request_mac=request_mac,
504                                one_rr_per_rrset=one_rr_per_rrset,
505                                ignore_trailing=ignore_trailing,
506                                raise_on_truncation=raise_on_truncation)
507      if destination:
508          return (r, received_time)
509      else:
510          return (r, received_time, from_address)

512  def udp(q, where, timeout=None, port=53, source=None, source_port=0,
513          ignore_unexpected=False, one_rr_per_rrset=False, ignore_trailing=False,
514          raise_on_truncation=False, sock=None):
515      """Return the response obtained after sending a query via UDP. 
516   
517      *q*, a ``dns.message.Message``, the query to send 
518   
519      *where*, a ``str`` containing an IPv4 or IPv6 address,  where 
520      to send the message. 
521   
522      *timeout*, a ``float`` or ``None``, the number of seconds to wait before the 
523      query times out.  If ``None``, the default, wait forever. 
524   
525      *port*, an ``int``, the port send the message to.  The default is 53. 
526   
527      *source*, a ``str`` containing an IPv4 or IPv6 address, specifying 
528      the source address.  The default is the wildcard address. 
529   
530      *source_port*, an ``int``, the port from which to send the message. 
531      The default is 0. 
532   
533      *ignore_unexpected*, a ``bool``.  If ``True``, ignore responses from 
534      unexpected sources. 
535   
536      *one_rr_per_rrset*, a ``bool``.  If ``True``, put each RR into its own 
537      RRset. 
538   
539      *ignore_trailing*, a ``bool``.  If ``True``, ignore trailing 
540      junk at end of the received message. 
541   
542      *raise_on_truncation*, a ``bool``.  If ``True``, raise an exception if 
543      the TC bit is set. 
544   
545      *sock*, a ``socket.socket``, or ``None``, the socket to use for the 
546      query.  If ``None``, the default, a socket is created.  Note that 
547      if a socket is provided, it must be a nonblocking datagram socket, 
548      and the *source* and *source_port* are ignored. 
549   
550      Returns a ``dns.message.Message``. 
551      """
552  
553      wire = q.to_wire()
554      (af, destination, source) = _destination_and_source(where, port,
555                                                          source, source_port)
556      (begin_time, expiration) = _compute_times(timeout)
557      with contextlib.ExitStack() as stack:
558          if sock:
559              s = sock
560          else:
561              s = stack.enter_context(_make_socket(af, socket.SOCK_DGRAM, source))
562          send_udp(s, wire, destination, expiration)
563          (r, received_time) = receive_udp(s, destination, expiration,
564                                           ignore_unexpected, one_rr_per_rrset,
565                                           q.keyring, q.mac, ignore_trailing,
566                                           raise_on_truncation)
567          r.time = received_time - begin_time
568          if not q.is_response(r):
569              raise BadResponse
570          return r

572  def udp_with_fallback(q, where, timeout=None, port=53, source=None,
573                        source_port=0, ignore_unexpected=False,
574                        one_rr_per_rrset=False, ignore_trailing=False,
575                        udp_sock=None, tcp_sock=None):
576      """Return the response to the query, trying UDP first and falling back 
577      to TCP if UDP results in a truncated response. 
578   
579      *q*, a ``dns.message.Message``, the query to send 
580   
581      *where*, a ``str`` containing an IPv4 or IPv6 address,  where 
582      to send the message. 
583   
584      *timeout*, a ``float`` or ``None``, the number of seconds to wait before the 
585      query times out.  If ``None``, the default, wait forever. 
586   
587      *port*, an ``int``, the port send the message to.  The default is 53. 
588   
589      *source*, a ``str`` containing an IPv4 or IPv6 address, specifying 
590      the source address.  The default is the wildcard address. 
591   
592      *source_port*, an ``int``, the port from which to send the message. 
593      The default is 0. 
594   
595      *ignore_unexpected*, a ``bool``.  If ``True``, ignore responses from 
596      unexpected sources. 
597   
598      *one_rr_per_rrset*, a ``bool``.  If ``True``, put each RR into its own 
599      RRset. 
600   
601      *ignore_trailing*, a ``bool``.  If ``True``, ignore trailing 
602      junk at end of the received message. 
603   
604      *udp_sock*, a ``socket.socket``, or ``None``, the socket to use for the 
605      UDP query.  If ``None``, the default, a socket is created.  Note that 
606      if a socket is provided, it must be a nonblocking datagram socket, 
607      and the *source* and *source_port* are ignored for the UDP query. 
608   
609      *tcp_sock*, a ``socket.socket``, or ``None``, the connected socket to use for the 
610      TCP query.  If ``None``, the default, a socket is created.  Note that 
611      if a socket is provided, it must be a nonblocking connected stream 
612      socket, and *where*, *source* and *source_port* are ignored for the TCP 
613      query. 
614   
615      Returns a (``dns.message.Message``, tcp) tuple where tcp is ``True`` 
616      if and only if TCP was used. 
617      """
618      try:
619          response = udp(q, where, timeout, port, source, source_port,
620                         ignore_unexpected, one_rr_per_rrset,
621                         ignore_trailing, True, udp_sock)
622          return (response, False)
623      except dns.message.Truncated:
624          response = tcp(q, where, timeout, port, source, source_port,
625                         one_rr_per_rrset, ignore_trailing, tcp_sock)
626          return (response, True)

628  def _net_read(sock, count, expiration):
629      """Read the specified number of bytes from sock.  Keep trying until we 
630      either get the desired amount, or we hit EOF. 
631      A Timeout exception will be raised if the operation is not completed 
632      by the expiration time. 
633      """
634      s = b''
635      while count > 0:
636          try:
637              n = sock.recv(count)
638              if n == b'':
639                  raise EOFError
640              count -= len(n)
641              s += n
642          except (BlockingIOError, ssl.SSLWantReadError):
643              _wait_for_readable(sock, expiration)
644          except ssl.SSLWantWriteError:  # pragma: no cover
645              _wait_for_writable(sock, expiration)
646      return s

649  def _net_write(sock, data, expiration):
650      """Write the specified data to the socket. 
651      A Timeout exception will be raised if the operation is not completed 
652      by the expiration time. 
653      """
654      current = 0
655      l = len(data)
656      while current < l:
657          try:
658              current += sock.send(data[current:])
659          except (BlockingIOError, ssl.SSLWantWriteError):
660              _wait_for_writable(sock, expiration)
661          except ssl.SSLWantReadError:  # pragma: no cover
662              _wait_for_readable(sock, expiration)

665  def send_tcp(sock, what, expiration=None):
666      """Send a DNS message to the specified TCP socket. 
667   
668      *sock*, a ``socket``. 
669   
670      *what*, a ``bytes`` or ``dns.message.Message``, the message to send. 
671   
672      *expiration*, a ``float`` or ``None``, the absolute time at which 
673      a timeout exception should be raised.  If ``None``, no timeout will 
674      occur. 
675   
676      Returns an ``(int, float)`` tuple of bytes sent and the sent time. 
677      """
678  
679      if isinstance(what, dns.message.Message):
680          what = what.to_wire()
681      l = len(what)
682      # copying the wire into tcpmsg is inefficient, but lets us
683      # avoid writev() or doing a short write that would get pushed
684      # onto the net
685      tcpmsg = struct.pack("!H", l) + what
686      sent_time = time.time()
687      _net_write(sock, tcpmsg, expiration)
688      return (len(tcpmsg), sent_time)
689  
690  def receive_tcp(sock, expiration=None, one_rr_per_rrset=False,
691                  keyring=None, request_mac=b'', ignore_trailing=False):
692      """Read a DNS message from a TCP socket. 
693   
694      *sock*, a ``socket``. 
695   
696      *expiration*, a ``float`` or ``None``, the absolute time at which 
697      a timeout exception should be raised.  If ``None``, no timeout will 
698      occur. 
699   
700      *one_rr_per_rrset*, a ``bool``.  If ``True``, put each RR into its own 
701      RRset. 
702   
703      *keyring*, a ``dict``, the keyring to use for TSIG. 
704   
705      *request_mac*, a ``bytes``, the MAC of the request (for TSIG). 
706   
707      *ignore_trailing*, a ``bool``.  If ``True``, ignore trailing 
708      junk at end of the received message. 
709   
710      Raises if the message is malformed, if network errors occur, of if 
711      there is a timeout. 
712   
713      Returns a ``(dns.message.Message, float)`` tuple of the received message 
714      and the received time. 
715      """
716  
717      ldata = _net_read(sock, 2, expiration)
718      (l,) = struct.unpack("!H", ldata)
719      wire = _net_read(sock, l, expiration)
720      received_time = time.time()
721      r = dns.message.from_wire(wire, keyring=keyring, request_mac=request_mac,
722                                one_rr_per_rrset=one_rr_per_rrset,
723                                ignore_trailing=ignore_trailing)
724      return (r, received_time)
725  
726  def _connect(s, address, expiration):
727      err = s.connect_ex(address)
728      if err == 0:
729          return
730      if err in (errno.EINPROGRESS, errno.EWOULDBLOCK, errno.EALREADY):
731          _wait_for_writable(s, expiration)
732          err = s.getsockopt(socket.SOL_SOCKET, socket.SO_ERROR)
733      if err != 0:
734          raise OSError(err, os.strerror(err))

737  def tcp(q, where, timeout=None, port=53, source=None, source_port=0,
738          one_rr_per_rrset=False, ignore_trailing=False, sock=None):
739      """Return the response obtained after sending a query via TCP. 
740   
741      *q*, a ``dns.message.Message``, the query to send 
742   
743      *where*, a ``str`` containing an IPv4 or IPv6 address, where 
744      to send the message. 
745   
746      *timeout*, a ``float`` or ``None``, the number of seconds to wait before the 
747      query times out.  If ``None``, the default, wait forever. 
748   
749      *port*, an ``int``, the port send the message to.  The default is 53. 
750   
751      *source*, a ``str`` containing an IPv4 or IPv6 address, specifying 
752      the source address.  The default is the wildcard address. 
753   
754      *source_port*, an ``int``, the port from which to send the message. 
755      The default is 0. 
756   
757      *one_rr_per_rrset*, a ``bool``.  If ``True``, put each RR into its own 
758      RRset. 
759   
760      *ignore_trailing*, a ``bool``.  If ``True``, ignore trailing 
761      junk at end of the received message. 
762   
763      *sock*, a ``socket.socket``, or ``None``, the connected socket to use for the 
764      query.  If ``None``, the default, a socket is created.  Note that 
765      if a socket is provided, it must be a nonblocking connected stream 
766      socket, and *where*, *port*, *source* and *source_port* are ignored. 
767   
768      Returns a ``dns.message.Message``. 
769      """
770  
771      wire = q.to_wire()
772      (begin_time, expiration) = _compute_times(timeout)
773      with contextlib.ExitStack() as stack:
774          if sock:
775              s = sock
776          else:
777              (af, destination, source) = _destination_and_source(where, port,
778                                                                  source,
779                                                                  source_port)
780              s = stack.enter_context(_make_socket(af, socket.SOCK_STREAM,
781                                                   source))
782              _connect(s, destination, expiration)
783          send_tcp(s, wire, expiration)
784          (r, received_time) = receive_tcp(s, expiration, one_rr_per_rrset,
785                                           q.keyring, q.mac, ignore_trailing)
786          r.time = received_time - begin_time
787          if not q.is_response(r):
788              raise BadResponse
789          return r
790  
791  
792  def _tls_handshake(s, expiration):
793      while True:
794          try:
795              s.do_handshake()
796              return
797          except ssl.SSLWantReadError:
798              _wait_for_readable(s, expiration)
799          except ssl.SSLWantWriteError:  # pragma: no cover
800              _wait_for_writable(s, expiration)

803  def tls(q, where, timeout=None, port=853, source=None, source_port=0,
804          one_rr_per_rrset=False, ignore_trailing=False, sock=None,
805          ssl_context=None, server_hostname=None):
806      """Return the response obtained after sending a query via TLS. 
807   
808      *q*, a ``dns.message.Message``, the query to send 
809   
810      *where*, a ``str`` containing an IPv4 or IPv6 address,  where 
811      to send the message. 
812   
813      *timeout*, a ``float`` or ``None``, the number of seconds to wait before the 
814      query times out.  If ``None``, the default, wait forever. 
815   
816      *port*, an ``int``, the port send the message to.  The default is 853. 
817   
818      *source*, a ``str`` containing an IPv4 or IPv6 address, specifying 
819      the source address.  The default is the wildcard address. 
820   
821      *source_port*, an ``int``, the port from which to send the message. 
822      The default is 0. 
823   
824      *one_rr_per_rrset*, a ``bool``.  If ``True``, put each RR into its own 
825      RRset. 
826   
827      *ignore_trailing*, a ``bool``.  If ``True``, ignore trailing 
828      junk at end of the received message. 
829   
830      *sock*, an ``ssl.SSLSocket``, or ``None``, the socket to use for 
831      the query.  If ``None``, the default, a socket is created.  Note 
832      that if a socket is provided, it must be a nonblocking connected 
833      SSL stream socket, and *where*, *port*, *source*, *source_port*, 
834      and *ssl_context* are ignored. 
835   
836      *ssl_context*, an ``ssl.SSLContext``, the context to use when establishing 
837      a TLS connection. If ``None``, the default, creates one with the default 
838      configuration. 
839   
840      *server_hostname*, a ``str`` containing the server's hostname.  The 
841      default is ``None``, which means that no hostname is known, and if an 
842      SSL context is created, hostname checking will be disabled. 
843   
844      Returns a ``dns.message.Message``. 
845   
846      """
847  
848      if sock:
849          #
850          # If a socket was provided, there's no special TLS handling needed.
851          #
852          return tcp(q, where, timeout, port, source, source_port,
853                     one_rr_per_rrset, ignore_trailing, sock)
854  
855      wire = q.to_wire()
856      (begin_time, expiration) = _compute_times(timeout)
857      (af, destination, source) = _destination_and_source(where, port,
858                                                          source, source_port)
859      if ssl_context is None and not sock:
860          ssl_context = ssl.create_default_context()
861          if server_hostname is None:
862              ssl_context.check_hostname = False
863  
864      with _make_socket(af, socket.SOCK_STREAM, source, ssl_context=ssl_context,
865                        server_hostname=server_hostname) as s:
866          _connect(s, destination, expiration)
867          _tls_handshake(s, expiration)
868          send_tcp(s, wire, expiration)
869          (r, received_time) = receive_tcp(s, expiration, one_rr_per_rrset,
870                                           q.keyring, q.mac, ignore_trailing)
871          r.time = received_time - begin_time
872          if not q.is_response(r):
873              raise BadResponse
874          return r

877  def xfr(where, zone, rdtype=dns.rdatatype.AXFR, rdclass=dns.rdataclass.IN,
878          timeout=None, port=53, keyring=None, keyname=None, relativize=True,
879          lifetime=None, source=None, source_port=0, serial=0,
880          use_udp=False, keyalgorithm=dns.tsig.default_algorithm):
881      """Return a generator for the responses to a zone transfer. 
882   
883      *where*, a ``str`` containing an IPv4 or IPv6 address,  where 
884      to send the message. 
885   
886      *zone*, a ``dns.name.Name`` or ``str``, the name of the zone to transfer. 
887   
888      *rdtype*, an ``int`` or ``str``, the type of zone transfer.  The 
889      default is ``dns.rdatatype.AXFR``.  ``dns.rdatatype.IXFR`` can be 
890      used to do an incremental transfer instead. 
891   
892      *rdclass*, an ``int`` or ``str``, the class of the zone transfer. 
893      The default is ``dns.rdataclass.IN``. 
894   
895      *timeout*, a ``float``, the number of seconds to wait for each 
896      response message.  If None, the default, wait forever. 
897   
898      *port*, an ``int``, the port send the message to.  The default is 53. 
899   
900      *keyring*, a ``dict``, the keyring to use for TSIG. 
901   
902      *keyname*, a ``dns.name.Name`` or ``str``, the name of the TSIG 
903      key to use. 
904   
905      *relativize*, a ``bool``.  If ``True``, all names in the zone will be 
906      relativized to the zone origin.  It is essential that the 
907      relativize setting matches the one specified to 
908      ``dns.zone.from_xfr()`` if using this generator to make a zone. 
909   
910      *lifetime*, a ``float``, the total number of seconds to spend 
911      doing the transfer.  If ``None``, the default, then there is no 
912      limit on the time the transfer may take. 
913   
914      *source*, a ``str`` containing an IPv4 or IPv6 address, specifying 
915      the source address.  The default is the wildcard address. 
916   
917      *source_port*, an ``int``, the port from which to send the message. 
918      The default is 0. 
919   
920      *serial*, an ``int``, the SOA serial number to use as the base for 
921      an IXFR diff sequence (only meaningful if *rdtype* is 
922      ``dns.rdatatype.IXFR``). 
923   
924      *use_udp*, a ``bool``.  If ``True``, use UDP (only meaningful for IXFR). 
925   
926      *keyalgorithm*, a ``dns.name.Name`` or ``str``, the TSIG algorithm to use. 
927   
928      Raises on errors, and so does the generator. 
929   
930      Returns a generator of ``dns.message.Message`` objects. 
931      """
932  
933      if isinstance(zone, str):
934          zone = dns.name.from_text(zone)
935      rdtype = dns.rdatatype.RdataType.make(rdtype)
936      q = dns.message.make_query(zone, rdtype, rdclass)
937      if rdtype == dns.rdatatype.IXFR:
938          rrset = dns.rrset.from_text(zone, 0, 'IN', 'SOA',
939                                      '. . %u 0 0 0 0' % serial)
940          q.authority.append(rrset)
941      if keyring is not None:
942          q.use_tsig(keyring, keyname, algorithm=keyalgorithm)
943      wire = q.to_wire()
944      (af, destination, source) = _destination_and_source(where, port,
945                                                          source, source_port)
946      if use_udp and rdtype != dns.rdatatype.IXFR:
947          raise ValueError('cannot do a UDP AXFR')
948      sock_type = socket.SOCK_DGRAM if use_udp else socket.SOCK_STREAM
949      with _make_socket(af, sock_type, source) as s:
950          (_, expiration) = _compute_times(lifetime)
951          _connect(s, destination, expiration)
952          l = len(wire)
953          if use_udp:
954              _udp_send(s, wire, None, expiration)
955          else:
956              tcpmsg = struct.pack("!H", l) + wire
957              _net_write(s, tcpmsg, expiration)
958          done = False
959          delete_mode = True
960          expecting_SOA = False
961          soa_rrset = None
962          if relativize:
963              origin = zone
964              oname = dns.name.empty
965          else:
966              origin = None
967              oname = zone
968          tsig_ctx = None
969          while not done:
970              (_, mexpiration) = _compute_times(timeout)
971              if mexpiration is None or \
972                 (expiration is not None and mexpiration > expiration):
973                  mexpiration = expiration
974              if use_udp:
975                  (wire, _) = _udp_recv(s, 65535, mexpiration)
976              else:
977                  ldata = _net_read(s, 2, mexpiration)
978                  (l,) = struct.unpack("!H", ldata)
979                  wire = _net_read(s, l, mexpiration)
980              is_ixfr = (rdtype == dns.rdatatype.IXFR)
981              r = dns.message.from_wire(wire, keyring=q.keyring,
982                                        request_mac=q.mac, xfr=True,
983                                        origin=origin, tsig_ctx=tsig_ctx,
984                                        multi=True, one_rr_per_rrset=is_ixfr)
985              rcode = r.rcode()
986              if rcode != dns.rcode.NOERROR:
987                  raise TransferError(rcode)
988              tsig_ctx = r.tsig_ctx
989              answer_index = 0
990              if soa_rrset is None:
991                  if not r.answer or r.answer[0].name != oname:
992                      raise dns.exception.FormError(
993                          "No answer or RRset not for qname")
994                  rrset = r.answer[0]
995                  if rrset.rdtype != dns.rdatatype.SOA:
996                      raise dns.exception.FormError("first RRset is not an SOA")
997                  answer_index = 1
998                  soa_rrset = rrset.copy()
999                  if rdtype == dns.rdatatype.IXFR:
1000                     if dns.serial.Serial(soa_rrset[0].serial) <= serial:
1001                         #
1002                         # We're already up-to-date.
1003                         #
1004                         done = True
1005                     else:
1006                         expecting_SOA = True
1007             #
1008             # Process SOAs in the answer section (other than the initial
1009             # SOA in the first message).
1010             #
1011             for rrset in r.answer[answer_index:]:
1012                 if done:
1013                     raise dns.exception.FormError("answers after final SOA")
1014                 if rrset.rdtype == dns.rdatatype.SOA and rrset.name == oname:
1015                     if expecting_SOA:
1016                         if rrset[0].serial != serial:
1017                             raise dns.exception.FormError(
1018                                 "IXFR base serial mismatch")
1019                         expecting_SOA = False
1020                     elif rdtype == dns.rdatatype.IXFR:
1021                         delete_mode = not delete_mode
1022                     #
1023                     # If this SOA RRset is equal to the first we saw then we're
1024                     # finished. If this is an IXFR we also check that we're
1025                     # seeing the record in the expected part of the response.
1026                     #
1027                     if rrset == soa_rrset and \
1028                             (rdtype == dns.rdatatype.AXFR or
1029                              (rdtype == dns.rdatatype.IXFR and delete_mode)):
1030                         done = True
1031                 elif expecting_SOA:
1032                     #
1033                     # We made an IXFR request and are expecting another
1034                     # SOA RR, but saw something else, so this must be an
1035                     # AXFR response.
1036                     #
1037                     rdtype = dns.rdatatype.AXFR
1038                     expecting_SOA = False
1039             if done and q.keyring and not r.had_tsig:
1040                 raise dns.exception.FormError("missing TSIG")
1041             yield r

1044 class UDPMode(enum.IntEnum):
1045     """How should UDP be used in an IXFR from :py:func:`inbound_xfr()`? 
1046  
1047     NEVER means "never use UDP; always use TCP" 
1048     TRY_FIRST means "try to use UDP but fall back to TCP if needed" 
1049     ONLY means "raise ``dns.xfr.UseTCP`` if trying UDP does not succeed" 
1050     """
1051     NEVER = 0
1052     TRY_FIRST = 1
1053     ONLY = 2
1054 
1055 
1056 def inbound_xfr(where, txn_manager, query=None,
1057                 port=53, timeout=None, lifetime=None, source=None,
1058                 source_port=0, udp_mode=UDPMode.NEVER):
1059     """Conduct an inbound transfer and apply it via a transaction from the 
1060     txn_manager. 
1061  
1062     *where*, a ``str`` containing an IPv4 or IPv6 address,  where 
1063     to send the message. 
1064  
1065     *txn_manager*, a ``dns.transaction.TransactionManager``, the txn_manager 
1066     for this transfer (typically a ``dns.zone.Zone``). 
1067  
1068     *query*, the query to send.  If not supplied, a default query is 
1069     constructed using information from the *txn_manager*. 
1070  
1071     *port*, an ``int``, the port send the message to.  The default is 53. 
1072  
1073     *timeout*, a ``float``, the number of seconds to wait for each 
1074     response message.  If None, the default, wait forever. 
1075  
1076     *lifetime*, a ``float``, the total number of seconds to spend 
1077     doing the transfer.  If ``None``, the default, then there is no 
1078     limit on the time the transfer may take. 
1079  
1080     *source*, a ``str`` containing an IPv4 or IPv6 address, specifying 
1081     the source address.  The default is the wildcard address. 
1082  
1083     *source_port*, an ``int``, the port from which to send the message. 
1084     The default is 0. 
1085  
1086     *udp_mode*, a ``dns.query.UDPMode``, determines how UDP is used 
1087     for IXFRs.  The default is ``dns.UDPMode.NEVER``, i.e. only use 
1088     TCP.  Other possibilities are ``dns.UDPMode.TRY_FIRST``, which 
1089     means "try UDP but fallback to TCP if needed", and 
1090     ``dns.UDPMode.ONLY``, which means "try UDP and raise 
1091     ``dns.xfr.UseTCP`` if it does not succeed. 
1092  
1093     Raises on errors. 
1094     """
1095     if query is None:
1096         (query, serial) = dns.xfr.make_query(txn_manager)
1097     else:
1098         serial = dns.xfr.extract_serial_from_query(query)
1099     rdtype = query.question[0].rdtype
1100     is_ixfr = rdtype == dns.rdatatype.IXFR
1101     origin = txn_manager.from_wire_origin()
1102     wire = query.to_wire()
1103     (af, destination, source) = _destination_and_source(where, port,
1104                                                         source, source_port)
1105     (_, expiration) = _compute_times(lifetime)
1106     retry = True
1107     while retry:
1108         retry = False
1109         if is_ixfr and udp_mode != UDPMode.NEVER:
1110             sock_type = socket.SOCK_DGRAM
1111             is_udp = True
1112         else:
1113             sock_type = socket.SOCK_STREAM
1114             is_udp = False
1115         with _make_socket(af, sock_type, source) as s:
1116             _connect(s, destination, expiration)
1117             if is_udp:
1118                 _udp_send(s, wire, None, expiration)
1119             else:
1120                 tcpmsg = struct.pack("!H", len(wire)) + wire
1121                 _net_write(s, tcpmsg, expiration)
1122             with dns.xfr.Inbound(txn_manager, rdtype, serial,
1123                                  is_udp) as inbound:
1124                 done = False
1125                 tsig_ctx = None
1126                 while not done:
1127                     (_, mexpiration) = _compute_times(timeout)
1128                     if mexpiration is None or \
1129                        (expiration is not None and mexpiration > expiration):
1130                         mexpiration = expiration
1131                     if is_udp:
1132                         (rwire, _) = _udp_recv(s, 65535, mexpiration)
1133                     else:
1134                         ldata = _net_read(s, 2, mexpiration)
1135                         (l,) = struct.unpack("!H", ldata)
1136                         rwire = _net_read(s, l, mexpiration)
1137                     r = dns.message.from_wire(rwire, keyring=query.keyring,
1138                                               request_mac=query.mac, xfr=True,
1139                                               origin=origin, tsig_ctx=tsig_ctx,
1140                                               multi=(not is_udp),
1141                                               one_rr_per_rrset=is_ixfr)
1142                     try:
1143                         done = inbound.process_message(r)
1144                     except dns.xfr.UseTCP:
1145                         assert is_udp  # should not happen if we used TCP!
1146                         if udp_mode == UDPMode.ONLY:
1147                             raise
1148                         done = True
1149                         retry = True
1150                         udp_mode = UDPMode.NEVER
1151                         continue
1152                     tsig_ctx = r.tsig_ctx
1153                 if not retry and query.keyring and not r.had_tsig:
1154                     raise dns.exception.FormError("missing TSIG")
1155
