44   def _source_tuple(af, address, port):
45       # Make a high level source tuple, or return None if address and port
46       # are both None
47       if address or port:
48           if address is None:
49               if af == socket.AF_INET:
50                   address = '0.0.0.0'
51               elif af == socket.AF_INET6:
52                   address = '::'
53               else:
54                   raise NotImplementedError(f'unknown address family {af}')
55           return (address, port)
56       else:
57           return None

60   def _timeout(expiration, now=None):
61       if expiration:
62           if not now:
63               now = time.time()
64           return max(expiration - now, 0)
65       else:
66           return None

69   async def send_udp(sock, what, destination, expiration=None):
70       """Send a DNS message to the specified UDP socket. 
71    
72       *sock*, a ``dns.asyncbackend.DatagramSocket``. 
73    
74       *what*, a ``bytes`` or ``dns.message.Message``, the message to send. 
75    
76       *destination*, a destination tuple appropriate for the address family 
77       of the socket, specifying where to send the query. 
78    
79       *expiration*, a ``float`` or ``None``, the absolute time at which 
80       a timeout exception should be raised.  If ``None``, no timeout will 
81       occur. 
82    
83       Returns an ``(int, float)`` tuple of bytes sent and the sent time. 
84       """
85   
86       if isinstance(what, dns.message.Message):
87           what = what.to_wire()
88       sent_time = time.time()
89       n = await sock.sendto(what, destination, _timeout(expiration, sent_time))
90       return (n, sent_time)

93   async def receive_udp(sock, destination=None, expiration=None,
94                         ignore_unexpected=False, one_rr_per_rrset=False,
95                         keyring=None, request_mac=b'', ignore_trailing=False,
96                         raise_on_truncation=False):
97       """Read a DNS message from a UDP socket. 
98    
99       *sock*, a ``dns.asyncbackend.DatagramSocket``. 
100   
101      See :py:func:`dns.query.receive_udp()` for the documentation of the other 
102      parameters, exceptions, and return type of this method. 
103      """
104  
105      wire = b''
106      while 1:
107          (wire, from_address) = await sock.recvfrom(65535, _timeout(expiration))
108          if _matches_destination(sock.family, from_address, destination,
109                                  ignore_unexpected):
110              break
111      received_time = time.time()
112      r = dns.message.from_wire(wire, keyring=keyring, request_mac=request_mac,
113                                one_rr_per_rrset=one_rr_per_rrset,
114                                ignore_trailing=ignore_trailing,
115                                raise_on_truncation=raise_on_truncation)
116      return (r, received_time, from_address)

118  async def udp(q, where, timeout=None, port=53, source=None, source_port=0,
119                ignore_unexpected=False, one_rr_per_rrset=False,
120                ignore_trailing=False, raise_on_truncation=False, sock=None,
121                backend=None):
122      """Return the response obtained after sending a query via UDP. 
123   
124      *sock*, a ``dns.asyncbackend.DatagramSocket``, or ``None``, 
125      the socket to use for the query.  If ``None``, the default, a 
126      socket is created.  Note that if a socket is provided, the 
127      *source*, *source_port*, and *backend* are ignored. 
128   
129      *backend*, a ``dns.asyncbackend.Backend``, or ``None``.  If ``None``, 
130      the default, then dnspython will use the default backend. 
131   
132      See :py:func:`dns.query.udp()` for the documentation of the other 
133      parameters, exceptions, and return type of this method. 
134      """
135      wire = q.to_wire()
136      (begin_time, expiration) = _compute_times(timeout)
137      s = None
138      # After 3.6 is no longer supported, this can use an AsyncExitStack.
139      try:
140          af = dns.inet.af_for_address(where)
141          destination = _lltuple((where, port), af)
142          if sock:
143              s = sock
144          else:
145              if not backend:
146                  backend = dns.asyncbackend.get_default_backend()
147              stuple = _source_tuple(af, source, source_port)
148              if backend.datagram_connection_required():
149                  dtuple = (where, port)
150              else:
151                  dtuple = None
152              s = await backend.make_socket(af, socket.SOCK_DGRAM, 0, stuple,
153                                            dtuple)
154          await send_udp(s, wire, destination, expiration)
155          (r, received_time, _) = await receive_udp(s, destination, expiration,
156                                                    ignore_unexpected,
157                                                    one_rr_per_rrset,
158                                                    q.keyring, q.mac,
159                                                    ignore_trailing,
160                                                    raise_on_truncation)
161          r.time = received_time - begin_time
162          if not q.is_response(r):
163              raise BadResponse
164          return r
165      finally:
166          if not sock and s:
167              await s.close()

169  async def udp_with_fallback(q, where, timeout=None, port=53, source=None,
170                              source_port=0, ignore_unexpected=False,
171                              one_rr_per_rrset=False, ignore_trailing=False,
172                              udp_sock=None, tcp_sock=None, backend=None):
173      """Return the response to the query, trying UDP first and falling back 
174      to TCP if UDP results in a truncated response. 
175   
176      *udp_sock*, a ``dns.asyncbackend.DatagramSocket``, or ``None``, 
177      the socket to use for the UDP query.  If ``None``, the default, a 
178      socket is created.  Note that if a socket is provided the *source*, 
179      *source_port*, and *backend* are ignored for the UDP query. 
180   
181      *tcp_sock*, a ``dns.asyncbackend.StreamSocket``, or ``None``, the 
182      socket to use for the TCP query.  If ``None``, the default, a 
183      socket is created.  Note that if a socket is provided *where*, 
184      *source*, *source_port*, and *backend*  are ignored for the TCP query. 
185   
186      *backend*, a ``dns.asyncbackend.Backend``, or ``None``.  If ``None``, 
187      the default, then dnspython will use the default backend. 
188   
189      See :py:func:`dns.query.udp_with_fallback()` for the documentation 
190      of the other parameters, exceptions, and return type of this 
191      method. 
192      """
193      try:
194          response = await udp(q, where, timeout, port, source, source_port,
195                               ignore_unexpected, one_rr_per_rrset,
196                               ignore_trailing, True, udp_sock, backend)
197          return (response, False)
198      except dns.message.Truncated:
199          response = await tcp(q, where, timeout, port, source, source_port,
200                               one_rr_per_rrset, ignore_trailing, tcp_sock,
201                               backend)
202          return (response, True)

205  async def send_tcp(sock, what, expiration=None):
206      """Send a DNS message to the specified TCP socket. 
207   
208      *sock*, a ``dns.asyncbackend.StreamSocket``. 
209   
210      See :py:func:`dns.query.send_tcp()` for the documentation of the other 
211      parameters, exceptions, and return type of this method. 
212      """
213  
214      if isinstance(what, dns.message.Message):
215          what = what.to_wire()
216      l = len(what)
217      # copying the wire into tcpmsg is inefficient, but lets us
218      # avoid writev() or doing a short write that would get pushed
219      # onto the net
220      tcpmsg = struct.pack("!H", l) + what
221      sent_time = time.time()
222      await sock.sendall(tcpmsg, _timeout(expiration, sent_time))
223      return (len(tcpmsg), sent_time)

226  async def _read_exactly(sock, count, expiration):
227      """Read the specified number of bytes from stream.  Keep trying until we 
228      either get the desired amount, or we hit EOF. 
229      """
230      s = b''
231      while count > 0:
232          n = await sock.recv(count, _timeout(expiration))
233          if n == b'':
234              raise EOFError
235          count = count - len(n)
236          s = s + n
237      return s

240  async def receive_tcp(sock, expiration=None, one_rr_per_rrset=False,
241                        keyring=None, request_mac=b'', ignore_trailing=False):
242      """Read a DNS message from a TCP socket. 
243   
244      *sock*, a ``dns.asyncbackend.StreamSocket``. 
245   
246      See :py:func:`dns.query.receive_tcp()` for the documentation of the other 
247      parameters, exceptions, and return type of this method. 
248      """
249  
250      ldata = await _read_exactly(sock, 2, expiration)
251      (l,) = struct.unpack("!H", ldata)
252      wire = await _read_exactly(sock, l, expiration)
253      received_time = time.time()
254      r = dns.message.from_wire(wire, keyring=keyring, request_mac=request_mac,
255                                one_rr_per_rrset=one_rr_per_rrset,
256                                ignore_trailing=ignore_trailing)
257      return (r, received_time)

260  async def tcp(q, where, timeout=None, port=53, source=None, source_port=0,
261                one_rr_per_rrset=False, ignore_trailing=False, sock=None,
262                backend=None):
263      """Return the response obtained after sending a query via TCP. 
264   
265      *sock*, a ``dns.asyncbacket.StreamSocket``, or ``None``, the 
266      socket to use for the query.  If ``None``, the default, a socket 
267      is created.  Note that if a socket is provided 
268      *where*, *port*, *source*, *source_port*, and *backend* are ignored. 
269   
270      *backend*, a ``dns.asyncbackend.Backend``, or ``None``.  If ``None``, 
271      the default, then dnspython will use the default backend. 
272   
273      See :py:func:`dns.query.tcp()` for the documentation of the other 
274      parameters, exceptions, and return type of this method. 
275      """
276  
277      wire = q.to_wire()
278      (begin_time, expiration) = _compute_times(timeout)
279      s = None
280      # After 3.6 is no longer supported, this can use an AsyncExitStack.
281      try:
282          if sock:
283              # Verify that the socket is connected, as if it's not connected,
284              # it's not writable, and the polling in send_tcp() will time out or
285              # hang forever.
286              await sock.getpeername()
287              s = sock
288          else:
289              # These are simple (address, port) pairs, not
290              # family-dependent tuples you pass to lowlevel socket
291              # code.
292              af = dns.inet.af_for_address(where)
293              stuple = _source_tuple(af, source, source_port)
294              dtuple = (where, port)
295              if not backend:
296                  backend = dns.asyncbackend.get_default_backend()
297              s = await backend.make_socket(af, socket.SOCK_STREAM, 0, stuple,
298                                            dtuple, timeout)
299          await send_tcp(s, wire, expiration)
300          (r, received_time) = await receive_tcp(s, expiration, one_rr_per_rrset,
301                                                 q.keyring, q.mac,
302                                                 ignore_trailing)
303          r.time = received_time - begin_time
304          if not q.is_response(r):
305              raise BadResponse
306          return r
307      finally:
308          if not sock and s:
309              await s.close()

311  async def tls(q, where, timeout=None, port=853, source=None, source_port=0,
312                one_rr_per_rrset=False, ignore_trailing=False, sock=None,
313                backend=None, ssl_context=None, server_hostname=None):
314      """Return the response obtained after sending a query via TLS. 
315   
316      *sock*, an ``asyncbackend.StreamSocket``, or ``None``, the socket 
317      to use for the query.  If ``None``, the default, a socket is 
318      created.  Note that if a socket is provided, it must be a 
319      connected SSL stream socket, and *where*, *port*, 
320      *source*, *source_port*, *backend*, *ssl_context*, and *server_hostname* 
321      are ignored. 
322   
323      *backend*, a ``dns.asyncbackend.Backend``, or ``None``.  If ``None``, 
324      the default, then dnspython will use the default backend. 
325   
326      See :py:func:`dns.query.tls()` for the documentation of the other 
327      parameters, exceptions, and return type of this method. 
328      """
329      # After 3.6 is no longer supported, this can use an AsyncExitStack.
330      (begin_time, expiration) = _compute_times(timeout)
331      if not sock:
332          if ssl_context is None:
333              ssl_context = ssl.create_default_context()
334              if server_hostname is None:
335                  ssl_context.check_hostname = False
336          else:
337              ssl_context = None
338              server_hostname = None
339          af = dns.inet.af_for_address(where)
340          stuple = _source_tuple(af, source, source_port)
341          dtuple = (where, port)
342          if not backend:
343              backend = dns.asyncbackend.get_default_backend()
344          s = await backend.make_socket(af, socket.SOCK_STREAM, 0, stuple,
345                                        dtuple, timeout, ssl_context,
346                                        server_hostname)
347      else:
348          s = sock
349      try:
350          timeout = _timeout(expiration)
351          response = await tcp(q, where, timeout, port, source, source_port,
352                               one_rr_per_rrset, ignore_trailing, s, backend)
353          end_time = time.time()
354          response.time = end_time - begin_time
355          return response
356      finally:
357          if not sock and s:
358              await s.close()

360  async def https(q, where, timeout=None, port=443, source=None, source_port=0,
361                  one_rr_per_rrset=False, ignore_trailing=False, client=None,
362                  path='/dns-query', post=True, verify=True):
363      """Return the response obtained after sending a query via DNS-over-HTTPS. 
364   
365      *client*, a ``httpx.AsyncClient``.  If provided, the client to use for 
366      the query. 
367   
368      Unlike the other dnspython async functions, a backend cannot be provided 
369      in this function because httpx always auto-detects the async backend. 
370   
371      See :py:func:`dns.query.https()` for the documentation of the other 
372      parameters, exceptions, and return type of this method. 
373      """
374  
375      if not _have_httpx:
376          raise NoDOH('httpx is not available.')  # pragma: no cover
377  
378      wire = q.to_wire()
379      try:
380          af = dns.inet.af_for_address(where)
381      except ValueError:
382          af = None
383      transport = None
384      headers = {
385          "accept": "application/dns-message"
386      }
387      if af is not None:
388          if af == socket.AF_INET:
389              url = 'https://{}:{}{}'.format(where, port, path)
390          elif af == socket.AF_INET6:
391              url = 'https://[{}]:{}{}'.format(where, port, path)
392      else:
393          url = where
394      if source is not None:
395          transport = httpx.AsyncHTTPTransport(local_address=source[0])
396  
397      # After 3.6 is no longer supported, this can use an AsyncExitStack
398      client_to_close = None
399      try:
400          if not client:
401              client = httpx.AsyncClient(http1=True, http2=_have_http2,
402                                         verify=verify, transport=transport)
403              client_to_close = client
404  
405          # see https://tools.ietf.org/html/rfc8484#section-4.1.1 for DoH
406          # GET and POST examples
407          if post:
408              headers.update({
409                  "content-type": "application/dns-message",
410                  "content-length": str(len(wire))
411              })
412              response = await client.post(url, headers=headers, content=wire,
413                                           timeout=timeout)
414          else:
415              wire = base64.urlsafe_b64encode(wire).rstrip(b"=")
416              wire = wire.decode()  # httpx does a repr() if we give it bytes
417              response = await client.get(url, headers=headers, timeout=timeout,
418                                          params={"dns": wire})
419      finally:
420          if client_to_close:
421              await client.aclose()
422  
423      # see https://tools.ietf.org/html/rfc8484#section-4.2.1 for info about DoH
424      # status codes
425      if response.status_code < 200 or response.status_code > 299:
426          raise ValueError('{} responded with status code {}'
427                           '\nResponse body: {}'.format(where,
428                                                        response.status_code,
429                                                        response.content))
430      r = dns.message.from_wire(response.content,
431                                keyring=q.keyring,
432                                request_mac=q.request_mac,
433                                one_rr_per_rrset=one_rr_per_rrset,
434                                ignore_trailing=ignore_trailing)
435      r.time = response.elapsed
436      if not q.is_response(r):
437          raise BadResponse
438      return r

440  async def inbound_xfr(where, txn_manager, query=None,
441                        port=53, timeout=None, lifetime=None, source=None,
442                        source_port=0, udp_mode=UDPMode.NEVER, backend=None):
443      """Conduct an inbound transfer and apply it via a transaction from the 
444      txn_manager. 
445   
446      *backend*, a ``dns.asyncbackend.Backend``, or ``None``.  If ``None``, 
447      the default, then dnspython will use the default backend. 
448   
449      See :py:func:`dns.query.inbound_xfr()` for the documentation of 
450      the other parameters, exceptions, and return type of this method. 
451      """
452      if query is None:
453          (query, serial) = dns.xfr.make_query(txn_manager)
454      else:
455          serial = dns.xfr.extract_serial_from_query(query)
456      rdtype = query.question[0].rdtype
457      is_ixfr = rdtype == dns.rdatatype.IXFR
458      origin = txn_manager.from_wire_origin()
459      wire = query.to_wire()
460      af = dns.inet.af_for_address(where)
461      stuple = _source_tuple(af, source, source_port)
462      dtuple = (where, port)
463      (_, expiration) = _compute_times(lifetime)
464      retry = True
465      while retry:
466          retry = False
467          if is_ixfr and udp_mode != UDPMode.NEVER:
468              sock_type = socket.SOCK_DGRAM
469              is_udp = True
470          else:
471              sock_type = socket.SOCK_STREAM
472              is_udp = False
473          if not backend:
474              backend = dns.asyncbackend.get_default_backend()
475          s = await backend.make_socket(af, sock_type, 0, stuple, dtuple,
476                                        _timeout(expiration))
477          async with s:
478              if is_udp:
479                  await s.sendto(wire, dtuple, _timeout(expiration))
480              else:
481                  tcpmsg = struct.pack("!H", len(wire)) + wire
482                  await s.sendall(tcpmsg, expiration)
483              with dns.xfr.Inbound(txn_manager, rdtype, serial,
484                                   is_udp) as inbound:
485                  done = False
486                  tsig_ctx = None
487                  while not done:
488                      (_, mexpiration) = _compute_times(timeout)
489                      if mexpiration is None or \
490                         (expiration is not None and mexpiration > expiration):
491                          mexpiration = expiration
492                      if is_udp:
493                          destination = _lltuple((where, port), af)
494                          while True:
495                              timeout = _timeout(mexpiration)
496                              (rwire, from_address) = await s.recvfrom(65535,
497                                                                       timeout)
498                              if _matches_destination(af, from_address,
499                                                      destination, True):
500                                  break
501                      else:
502                          ldata = await _read_exactly(s, 2, mexpiration)
503                          (l,) = struct.unpack("!H", ldata)
504                          rwire = await _read_exactly(s, l, mexpiration)
505                      is_ixfr = (rdtype == dns.rdatatype.IXFR)
506                      r = dns.message.from_wire(rwire, keyring=query.keyring,
507                                                request_mac=query.mac, xfr=True,
508                                                origin=origin, tsig_ctx=tsig_ctx,
509                                                multi=(not is_udp),
510                                                one_rr_per_rrset=is_ixfr)
511                      try:
512                          done = inbound.process_message(r)
513                      except dns.xfr.UseTCP:
514                          assert is_udp  # should not happen if we used TCP!
515                          if udp_mode == UDPMode.ONLY:
516                              raise
517                          done = True
518                          retry = True
519                          udp_mode = UDPMode.NEVER
520                          continue
521                      tsig_ctx = r.tsig_ctx
522                  if not retry and query.keyring and not r.had_tsig:
523                      raise dns.exception.FormError("missing TSIG")
524
