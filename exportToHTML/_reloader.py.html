13   # The various system prefixes where imports are found. Base values are
14   # different when running in a virtualenv. All reloaders will ignore the
15   # base paths (usually the system installation). The stat reloader won't
16   # scan the virtualenv paths, it will only include modules that are
17   # already imported.
18   _ignore_always = tuple({sys.base_prefix, sys.base_exec_prefix})
19   prefix = {*_ignore_always, sys.prefix, sys.exec_prefix}
20   
21   if hasattr(sys, "real_prefix"):
22       # virtualenv < 20
23       prefix.add(sys.real_prefix)  # type: ignore[attr-defined]
24   
25   _stat_ignore_scan = tuple(prefix)
26   del prefix
27   _ignore_common_dirs = {
28       "__pycache__",
29       ".git",
30       ".hg",
31       ".tox",
32       ".nox",
33       ".pytest_cache",
34       ".mypy_cache",
35   }

38   def _iter_module_paths() -> t.Iterator[str]:
39       """Find the filesystem paths associated with imported modules."""
40       # List is in case the value is modified by the app while updating.
41       for module in list(sys.modules.values()):
42           name = getattr(module, "__file__", None)
43   
44           if name is None or name.startswith(_ignore_always):
45               continue
46   
47           while not os.path.isfile(name):
48               # Zip file, find the base file without the module path.
49               old = name
50               name = os.path.dirname(name)
51   
52               if name == old:  # skip if it was all directories somehow
53                   break
54           else:
55               yield name

58   def _remove_by_pattern(paths: t.Set[str], exclude_patterns: t.Set[str]) -> None:
59       for pattern in exclude_patterns:
60           paths.difference_update(fnmatch.filter(paths, pattern))

63   def _find_stat_paths(
64       extra_files: t.Set[str], exclude_patterns: t.Set[str]
65   ) -> t.Iterable[str]:
66       """Find paths for the stat reloader to watch. Returns imported 
67       module files, Python files under non-system paths. Extra files and 
68       Python files under extra directories can also be scanned. 
69    
70       System paths have to be excluded for efficiency. Non-system paths, 
71       such as a project root or ``sys.path.insert``, should be the paths 
72       of interest to the user anyway. 
73       """
74       paths = set()
75   
76       for path in chain(list(sys.path), extra_files):
77           path = os.path.abspath(path)
78   
79           if os.path.isfile(path):
80               # zip file on sys.path, or extra file
81               paths.add(path)
82               continue
83   
84           parent_has_py = {os.path.dirname(path): True}
85   
86           for root, dirs, files in os.walk(path):
87               # Optimizations: ignore system prefixes, __pycache__ will
88               # have a py or pyc module at the import path, ignore some
89               # common known dirs such as version control and tool caches.
90               if (
91                   root.startswith(_stat_ignore_scan)
92                   or os.path.basename(root) in _ignore_common_dirs
93               ):
94                   dirs.clear()
95                   continue
96   
97               has_py = False
98   
99               for name in files:
100                  if name.endswith((".py", ".pyc")):
101                      has_py = True
102                      paths.add(os.path.join(root, name))
103  
104              # Optimization: stop scanning a directory if neither it nor
105              # its parent contained Python files.
106              if not (has_py or parent_has_py[os.path.dirname(root)]):
107                  dirs.clear()
108                  continue
109  
110              parent_has_py[root] = has_py
111  
112      paths.update(_iter_module_paths())
113      _remove_by_pattern(paths, exclude_patterns)
114      return paths

117  def _find_watchdog_paths(
118      extra_files: t.Set[str], exclude_patterns: t.Set[str]
119  ) -> t.Iterable[str]:
120      """Find paths for the stat reloader to watch. Looks at the same 
121      sources as the stat reloader, but watches everything under 
122      directories instead of individual files. 
123      """
124      dirs = set()
125  
126      for name in chain(list(sys.path), extra_files):
127          name = os.path.abspath(name)
128  
129          if os.path.isfile(name):
130              name = os.path.dirname(name)
131  
132          dirs.add(name)
133  
134      for name in _iter_module_paths():
135          dirs.add(os.path.dirname(name))
136  
137      _remove_by_pattern(dirs, exclude_patterns)
138      return _find_common_roots(dirs)

141  def _find_common_roots(paths: t.Iterable[str]) -> t.Iterable[str]:
142      root: t.Dict[str, dict] = {}
143  
144      for chunks in sorted((PurePath(x).parts for x in paths), key=len, reverse=True):
145          node = root
146  
147          for chunk in chunks:
148              node = node.setdefault(chunk, {})
149  
150          node.clear()
151  
152      rv = set()
153  
154      def _walk(node: t.Mapping[str, dict], path: t.Tuple[str, ...]) -> None:
155          for prefix, child in node.items():
156              _walk(child, path + (prefix,))
157  
158          if not node:
159              rv.add(os.path.join(*path))
160  
161      _walk(root, ())
162      return rv

165  def _get_args_for_reloading() -> t.List[str]:
166      """Determine how the script was executed, and return the args needed 
167      to execute it again in a new process. 
168      """
169      rv = [sys.executable]
170      py_script = sys.argv[0]
171      args = sys.argv[1:]
172      # Need to look at main module to determine how it was executed.
173      __main__ = sys.modules["__main__"]
174  
175      # The value of __package__ indicates how Python was called. It may
176      # not exist if a setuptools script is installed as an egg. It may be
177      # set incorrectly for entry points created with pip on Windows.
178      if getattr(__main__, "__package__", None) is None or (
179          os.name == "nt"
180          and __main__.__package__ == ""
181          and not os.path.exists(py_script)
182          and os.path.exists(f"{py_script}.exe")
183      ):
184          # Executed a file, like "python app.py".
185          py_script = os.path.abspath(py_script)
186  
187          if os.name == "nt":
188              # Windows entry points have ".exe" extension and should be
189              # called directly.
190              if not os.path.exists(py_script) and os.path.exists(f"{py_script}.exe"):
191                  py_script += ".exe"
192  
193              if (
194                  os.path.splitext(sys.executable)[1] == ".exe"
195                  and os.path.splitext(py_script)[1] == ".exe"
196              ):
197                  rv.pop(0)
198  
199          rv.append(py_script)
200      else:
201          # Executed a module, like "python -m werkzeug.serving".
202          if os.path.isfile(py_script):
203              # Rewritten by Python from "-m script" to "/path/to/script.py".
204              py_module = t.cast(str, __main__.__package__)
205              name = os.path.splitext(os.path.basename(py_script))[0]
206  
207              if name != "__main__":
208                  py_module += f".{name}"
209          else:
210              # Incorrectly rewritten by pydevd debugger from "-m script" to "script".
211              py_module = py_script
212  
213          rv.extend(("-m", py_module.lstrip(".")))
214  
215      rv.extend(args)
216      return rv

219  class ReloaderLoop:
220      name = ""
221  
222      def __init__(
223          self,
224          extra_files: t.Optional[t.Iterable[str]] = None,
225          exclude_patterns: t.Optional[t.Iterable[str]] = None,
226          interval: t.Union[int, float] = 1,
227      ) -> None:
228          self.extra_files: t.Set[str] = {os.path.abspath(x) for x in extra_files or ()}
229          self.exclude_patterns: t.Set[str] = set(exclude_patterns or ())
230          self.interval = interval
231  
232      def __enter__(self) -> "ReloaderLoop":
233          """Do any setup, then run one step of the watch to populate the 
234          initial filesystem state. 
235          """
236          self.run_step()
237          return self
238  
239      def __exit__(self, exc_type, exc_val, exc_tb):  # type: ignore
240          """Clean up any resources associated with the reloader."""
241          pass
242  
243      def run(self) -> None:
244          """Continually run the watch step, sleeping for the configured 
245          interval after each step. 
246          """
247          while True:
248              self.run_step()
249              time.sleep(self.interval)
250  
251      def run_step(self) -> None:
252          """Run one step for watching the filesystem. Called once to set 
253          up initial state, then repeatedly to update it. 
254          """
255          pass
256  
257      def restart_with_reloader(self) -> int:
258          """Spawn a new Python interpreter with the same arguments as the 
259          current one, but running the reloader thread. 
260          """
261          while True:
262              _log("info", f" * Restarting with {self.name}")
263              args = _get_args_for_reloading()
264              new_environ = os.environ.copy()
265              new_environ["WERKZEUG_RUN_MAIN"] = "true"
266              exit_code = subprocess.call(args, env=new_environ, close_fds=False)
267  
268              if exit_code != 3:
269                  return exit_code
270  
271      def trigger_reload(self, filename: str) -> None:
272          self.log_reload(filename)
273          sys.exit(3)
274  
275      def log_reload(self, filename: str) -> None:
276          filename = os.path.abspath(filename)
277          _log("info", f" * Detected change in {filename!r}, reloading")

280  class StatReloaderLoop(ReloaderLoop):
281      name = "stat"
282  
283      def __enter__(self) -> ReloaderLoop:
284          self.mtimes: t.Dict[str, float] = {}
285          return super().__enter__()
286  
287      def run_step(self) -> None:
288          for name in _find_stat_paths(self.extra_files, self.exclude_patterns):
289              try:
290                  mtime = os.stat(name).st_mtime
291              except OSError:
292                  continue
293  
294              old_time = self.mtimes.get(name)
295  
296              if old_time is None:
297                  self.mtimes[name] = mtime
298                  continue
299  
300              if mtime > old_time:
301                  self.trigger_reload(name)

304  class WatchdogReloaderLoop(ReloaderLoop):
305      def __init__(self, *args: t.Any, **kwargs: t.Any) -> None:
306          from watchdog.observers import Observer
307          from watchdog.events import PatternMatchingEventHandler
308  
309          super().__init__(*args, **kwargs)
310          trigger_reload = self.trigger_reload
311  
312          class EventHandler(PatternMatchingEventHandler):  # type: ignore
313              def on_any_event(self, event):  # type: ignore
314                  trigger_reload(event.src_path)
315  
316          reloader_name = Observer.__name__.lower()
317  
318          if reloader_name.endswith("observer"):
319              reloader_name = reloader_name[:-8]
320  
321          self.name = f"watchdog ({reloader_name})"
322          self.observer = Observer()
323          # Extra patterns can be non-Python files, match them in addition
324          # to all Python files in default and extra directories. Ignore
325          # __pycache__ since a change there will always have a change to
326          # the source file (or initial pyc file) as well. Ignore Git and
327          # Mercurial internal changes.
328          extra_patterns = [p for p in self.extra_files if not os.path.isdir(p)]
329          self.event_handler = EventHandler(
330              patterns=["*.py", "*.pyc", "*.zip", *extra_patterns],
331              ignore_patterns=[
332                  *[f"*/{d}/*" for d in _ignore_common_dirs],
333                  *self.exclude_patterns,
334              ],
335          )
336          self.should_reload = False
337  
338      def trigger_reload(self, filename: str) -> None:
339          # This is called inside an event handler, which means throwing
340          # SystemExit has no effect.
341          # https://github.com/gorakhargosh/watchdog/issues/294
342          self.should_reload = True
343          self.log_reload(filename)
344  
345      def __enter__(self) -> ReloaderLoop:
346          self.watches: t.Dict[str, t.Any] = {}
347          self.observer.start()
348          return super().__enter__()
349  
350      def __exit__(self, exc_type, exc_val, exc_tb):  # type: ignore
351          self.observer.stop()
352          self.observer.join()
353  
354      def run(self) -> None:
355          while not self.should_reload:
356              self.run_step()
357              time.sleep(self.interval)
358  
359          sys.exit(3)
360  
361      def run_step(self) -> None:
362          to_delete = set(self.watches)
363  
364          for path in _find_watchdog_paths(self.extra_files, self.exclude_patterns):
365              if path not in self.watches:
366                  try:
367                      self.watches[path] = self.observer.schedule(
368                          self.event_handler, path, recursive=True
369                      )
370                  except OSError:
371                      # Clear this path from list of watches We don't want
372                      # the same error message showing again in the next
373                      # iteration.
374                      self.watches[path] = None
375  
376              to_delete.discard(path)
377  
378          for path in to_delete:
379              watch = self.watches.pop(path, None)
380  
381              if watch is not None:
382                  self.observer.unschedule(watch)

398  def ensure_echo_on() -> None:
399      """Ensure that echo mode is enabled. Some tools such as PDB disable 
400      it which causes usability issues after a reload."""
401      # tcgetattr will fail if stdin isn't a tty
402      if sys.stdin is None or not sys.stdin.isatty():
403          return
404  
405      try:
406          import termios
407      except ImportError:
408          return
409  
410      attributes = termios.tcgetattr(sys.stdin)
411  
412      if not attributes[3] & termios.ECHO:
413          attributes[3] |= termios.ECHO
414          termios.tcsetattr(sys.stdin, termios.TCSANOW, attributes)

417  def run_with_reloader(
418      main_func: t.Callable[[], None],
419      extra_files: t.Optional[t.Iterable[str]] = None,
420      exclude_patterns: t.Optional[t.Iterable[str]] = None,
421      interval: t.Union[int, float] = 1,
422      reloader_type: str = "auto",
423  ) -> None:
424      """Run the given function in an independent Python interpreter."""
425      import signal
426  
427      signal.signal(signal.SIGTERM, lambda *args: sys.exit(0))
428      reloader = reloader_loops[reloader_type](
429          extra_files=extra_files, exclude_patterns=exclude_patterns, interval=interval
430      )
431  
432      try:
433          if os.environ.get("WERKZEUG_RUN_MAIN") == "true":
434              ensure_echo_on()
435              t = threading.Thread(target=main_func, args=())
436              t.daemon = True
437  
438              # Enter the reloader to set up initial state, then start
439              # the app thread and reloader update loop.
440              with reloader:
441                  t.start()
442                  reloader.run()
443          else:
444              sys.exit(reloader.restart_with_reloader())
445      except KeyboardInterrupt:
446          pass
