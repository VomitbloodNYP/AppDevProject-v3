18   """IPv4 helper functions."""
19   
20   import struct
21   
22   import dns.exception
23   
24   def inet_ntoa(address):
25       """Convert an IPv4 address in binary form to text form. 
26    
27       *address*, a ``bytes``, the IPv4 address in binary form. 
28    
29       Returns a ``str``. 
30       """
31   
32       if len(address) != 4:
33           raise dns.exception.SyntaxError
34       return ('%u.%u.%u.%u' % (address[0], address[1],
35                                address[2], address[3]))
36   
37   def inet_aton(text):
38       """Convert an IPv4 address in text form to binary form. 
39    
40       *text*, a ``str``, the IPv4 address in textual form. 
41    
42       Returns a ``bytes``. 
43       """
44   
45       if not isinstance(text, bytes):
46           text = text.encode()
47       parts = text.split(b'.')
48       if len(parts) != 4:
49           raise dns.exception.SyntaxError
50       for part in parts:
51           if not part.isdigit():
52               raise dns.exception.SyntaxError
53           if len(part) > 1 and part[0] == ord('0'):
54               # No leading zeros
55               raise dns.exception.SyntaxError
56       try:
57           b = [int(part) for part in parts]
58           return struct.pack('BBBB', *b)
59       except Exception:
60           raise dns.exception.SyntaxError
