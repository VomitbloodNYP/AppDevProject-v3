18   # Based on RFC 2822 section 3.2.4 / RFC 5322 section 3.2.3, these
19   # characters are permitted in email addresses (not taking into
20   # account internationalization):
21   ATEXT = r'a-zA-Z0-9_!#\$%&\'\*\+\-/=\?\^`\{\|\}~'
22   
23   # A "dot atom text", per RFC 2822 3.2.4:
24   DOT_ATOM_TEXT = '[' + ATEXT + ']+(?:\\.[' + ATEXT + ']+)*'
25   
26   # RFC 6531 section 3.3 extends the allowed characters in internationalized
27   # addresses to also include three specific ranges of UTF8 defined in
28   # RFC3629 section 4, which appear to be the Unicode code points from
29   # U+0080 to U+10FFFF.
30   ATEXT_INTL = ATEXT + u"\u0080-\U0010FFFF"
31   DOT_ATOM_TEXT_INTL = '[' + ATEXT_INTL + ']+(?:\\.[' + ATEXT_INTL + ']+)*'
32   
33   # The domain part of the email address, after IDNA (ASCII) encoding,
34   # must also satisfy the requirements of RFC 952/RFC 1123 which restrict
35   # the allowed characters of hostnames further. The hyphen cannot be at
36   # the beginning or end of a *dot-atom component* of a hostname either.
37   ATEXT_HOSTNAME = r'(?:(?:[a-zA-Z0-9][a-zA-Z0-9\-]*)?[a-zA-Z0-9])'
38   
39   # Length constants
40   # RFC 3696 + errata 1003 + errata 1690 (https://www.rfc-editor.org/errata_search.php?rfc=3696&eid=1690)
41   # explains the maximum length of an email address is 254 octets.
42   EMAIL_MAX_LENGTH = 254
43   LOCAL_PART_MAX_LENGTH = 64
44   DOMAIN_MAX_LENGTH = 255

46   # IANA Special Use Domain Names
47   # Last Updated 2021-09-21
48   # https://www.iana.org/assignments/special-use-domain-names/special-use-domain-names.txt
49   #
50   # The domain names without dots would be caught by the check that the domain
51   # name in an email address must have a period, but this list will also catch
52   # subdomains of these domains, which are also reserved.
53   SPECIAL_USE_DOMAIN_NAMES = [
54       # The "arpa" entry here is consolidated from a lot of arpa subdomains
55       # for private address (i.e. non-routable IP addresses like 172.16.x.x)
56       # reverse mapping, plus some other subdomains. Although RFC 6761 says
57       # that application software should not treat these domains as special,
58       # they are private-use domains and so cannot have globally deliverable
59       # email addresses, which is an assumption of this library, and probably
60       # all of arpa is similarly special-use, so we reject it all.
61       "arpa",
62   
63       # RFC 6761 says applications "SHOULD NOT" treat the "example" domains
64       # as special, i.e. applications should accept these domains.
65       #
66       # The domain "example" alone fails our syntax validation because it
67       # lacks a dot (we assume no one has an email address on a TLD directly).
68       # "@example.com/net/org" will currently fail DNS-based deliverability
69       # checks because IANA publishes a NULL MX for these domains, and
70       # "@mail.example[.com/net/org]" and other subdomains will fail DNS-
71       # based deliverability checks because IANA does not publish MX or A
72       # DNS records for these subdomains.
73       # "example", # i.e. "wwww.example"
74       # "example.com",
75       # "example.net",
76       # "example.org",
77   
78       # RFC 6761 says that applications are permitted to treat this domain
79       # as special and that DNS should return an immediate negative response,
80       # so we also immediately reject this domain, which also follows the
81       # purpose of the domain.
82       "invalid",
83   
84       # RFC 6762 says that applications "may" treat ".local" as special and
85       # that "name resolution APIs and libraries SHOULD recognize these names
86       # as special," and since ".local" has no global definition, we reject
87       # it, as we expect email addresses to be gloally routable.
88       "local",
89   
90       # RFC 6761 says that applications (like this library) are permitted
91       # to treat "localhost" as special, and since it cannot have a globally
92       # deliverable email address, we reject it.
93       "localhost",
94   
95       # RFC 7686 says "applications that do not implement the Tor protocol
96       # SHOULD generate an error upon the use of .onion and SHOULD NOT
97       # perform a DNS lookup.
98       "onion",
99   
100      # Although RFC 6761 says that application software should not treat
101      # these domains as special, it also warns users that the address may
102      # resolve differently in different systems, and therefore it cannot
103      # have a globally routable email address, which is an assumption of
104      # this library, so we reject "@test" and "@*.test" addresses, unless
105      # the test_environment keyword argument is given, to allow their use
106      # in application-level test environments. These domains will generally
107      # fail deliverability checks because "test" is not an actual TLD.
108      "test",
109  ]

138  class ValidatedEmail(object):
139      """The validate_email function returns objects of this type holding the normalized form of the email address 
140      and other information."""
141  
142      """The email address that was passed to validate_email. (If passed as bytes, this will be a string.)"""
143      original_email = None
144  
145      """The normalized email address, which should always be used in preferance to the original address. 
146      The normalized address converts an IDNA ASCII domain name to Unicode, if possible, and performs 
147      Unicode normalization on the local part and on the domain (if originally Unicode). It is the 
148      concatenation of the local_part and domain attributes, separated by an @-sign."""
149      email = None
150  
151      """The local part of the email address after Unicode normalization."""
152      local_part = None
153  
154      """The domain part of the email address after Unicode normalization or conversion to 
155      Unicode from IDNA ascii."""
156      domain = None
157  
158      """If not None, a form of the email address that uses 7-bit ASCII characters only."""
159      ascii_email = None
160  
161      """If not None, the local part of the email address using 7-bit ASCII characters only."""
162      ascii_local_part = None
163  
164      """If not None, a form of the domain name that uses 7-bit ASCII characters only."""
165      ascii_domain = None
166  
167      """If True, the SMTPUTF8 feature of your mail relay will be required to transmit messages 
168      to this address. This flag is True just when ascii_local_part is missing. Otherwise it 
169      is False."""
170      smtputf8 = None
171  
172      """If a deliverability check is performed and if it succeeds, a list of (priority, domain) 
173      tuples of MX records specified in the DNS for the domain."""
174      mx = None
175  
176      """If no MX records are actually specified in DNS and instead are inferred, through an obsolete 
177      mechanism, from A or AAAA records, the value is the type of DNS record used instead (`A` or `AAAA`)."""
178      mx_fallback_type = None
179  
180      """Tests use this constructor."""
181      def __init__(self, **kwargs):
182          for k, v in kwargs.items():
183              setattr(self, k, v)
184  
185      """As a convenience, str(...) on instances of this class return the normalized address."""
186      def __self__(self):
187          return self.normalized_email
188  
189      def __repr__(self):
190          return "<ValidatedEmail {}>".format(self.email)
191  
192      """For backwards compatibility, some fields are also exposed through a dict-like interface. Note 
193      that some of the names changed when they became attributes."""
194      def __getitem__(self, key):
195          if key == "email":
196              return self.email
197          if key == "email_ascii":
198              return self.ascii_email
199          if key == "local":
200              return self.local_part
201          if key == "domain":
202              return self.ascii_domain
203          if key == "domain_i18n":
204              return self.domain
205          if key == "smtputf8":
206              return self.smtputf8
207          if key == "mx":
208              return self.mx
209          if key == "mx-fallback":
210              return self.mx_fallback_type
211          raise KeyError()
212  
213      """Tests use this."""
214      def __eq__(self, other):
215          if not isinstance(other, ValidatedEmail):
216              return False
217          return (
218              self.email == other.email
219              and self.local_part == other.local_part
220              and self.domain == other.domain
221              and self.ascii_email == other.ascii_email
222              and self.ascii_local_part == other.ascii_local_part
223              and self.ascii_domain == other.ascii_domain
224              and self.smtputf8 == other.smtputf8
225              and repr(sorted(self.mx) if self.mx else self.mx)
226              == repr(sorted(other.mx) if other.mx else other.mx)
227              and self.mx_fallback_type == other.mx_fallback_type
228          )
229  
230      """This helps producing the README."""
231      def as_constructor(self):
232          return "ValidatedEmail(" \
233              + ",".join("\n  {}={}".format(
234                         key,
235                         repr(getattr(self, key)))
236                         for key in ('email', 'local_part', 'domain',
237                                     'ascii_email', 'ascii_local_part', 'ascii_domain',
238                                     'smtputf8', 'mx', 'mx_fallback_type')
239                         ) \
240              + ")"
241  
242      """Convenience method for accessing ValidatedEmail as a dict"""
243      def as_dict(self):
244          return self.__dict__

247  def __get_length_reason(addr, utf8=False, limit=EMAIL_MAX_LENGTH):
248      diff = len(addr) - limit
249      reason = "({}{} character{} too many)"
250      prefix = "at least " if utf8 else ""
251      suffix = "s" if diff > 1 else ""
252      return reason.format(prefix, diff, suffix)

255  def caching_resolver(*, timeout=None, cache=None):
256      if timeout is None:
257          timeout = DEFAULT_TIMEOUT
258      resolver = dns.resolver.Resolver()
259      resolver.cache = cache or dns.resolver.LRUCache()
260      resolver.lifetime = timeout  # timeout, in seconds

264  def validate_email(
265      email,
266      # /, # not supported in Python 3.6, 3.7
267      *,
268      allow_smtputf8=None,
269      allow_empty_local=False,
270      check_deliverability=None,
271      test_environment=None,
272      globally_deliverable=GLOBALLY_DELIVERABLE,
273      timeout=None,
274      dns_resolver=None
275  ):
276      """ 
277      Validates an email address, raising an EmailNotValidError if the address is not valid or returning a dict of 
278      information when the address is valid. The email argument can be a str or a bytes instance, 
279      but if bytes it must be ASCII-only. 
280      """
281  
282      # Fill in default values of arguments.
283      if allow_smtputf8 is None:
284          allow_smtputf8 = ALLOW_SMTPUTF8
285      if check_deliverability is None:
286          check_deliverability = CHECK_DELIVERABILITY
287      if test_environment is None:
288          test_environment = TEST_ENVIRONMENT
289      if timeout is None:
290          timeout = DEFAULT_TIMEOUT
291  
292      # Allow email to be a str or bytes instance. If bytes,
293      # it must be ASCII because that's how the bytes work
294      # on the wire with SMTP.
295      if not isinstance(email, (str, unicode_class)):
296          try:
297              email = email.decode("ascii")
298          except ValueError:
299              raise EmailSyntaxError("The email address is not valid ASCII.")
300  
301      # At-sign.
302      parts = email.split('@')
303      if len(parts) != 2:
304          raise EmailSyntaxError("The email address is not valid. It must have exactly one @-sign.")
305  
306      # Collect return values in this instance.
307      ret = ValidatedEmail()
308      ret.original_email = email
309  
310      # Validate the email address's local part syntax and get a normalized form.
311      local_part_info = validate_email_local_part(parts[0],
312                                                  allow_smtputf8=allow_smtputf8,
313                                                  allow_empty_local=allow_empty_local)
314      ret.local_part = local_part_info["local_part"]
315      ret.ascii_local_part = local_part_info["ascii_local_part"]
316      ret.smtputf8 = local_part_info["smtputf8"]
317  
318      # Validate the email address's domain part syntax and get a normalized form.
319      domain_part_info = validate_email_domain_part(parts[1], test_environment=test_environment, globally_deliverable=globally_deliverable)
320      ret.domain = domain_part_info["domain"]
321      ret.ascii_domain = domain_part_info["ascii_domain"]
322  
323      # Construct the complete normalized form.
324      ret.email = ret.local_part + "@" + ret.domain
325  
326      # If the email address has an ASCII form, add it.
327      if not ret.smtputf8:
328          ret.ascii_email = ret.ascii_local_part + "@" + ret.ascii_domain
329  
330      # If the email address has an ASCII representation, then we assume it may be
331      # transmitted in ASCII (we can't assume SMTPUTF8 will be used on all hops to
332      # the destination) and the length limit applies to ASCII characters (which is
333      # the same as octets). The number of characters in the internationalized form
334      # may be many fewer (because IDNA ASCII is verbose) and could be less than 254
335      # Unicode characters, and of course the number of octets over the limit may
336      # not be the number of characters over the limit, so if the email address is
337      # internationalized, we can't give any simple information about why the address
338      # is too long.
339      #
340      # In addition, check that the UTF-8 encoding (i.e. not IDNA ASCII and not
341      # Unicode characters) is at most 254 octets. If the addres is transmitted using
342      # SMTPUTF8, then the length limit probably applies to the UTF-8 encoded octets.
343      # If the email address has an ASCII form that differs from its internationalized
344      # form, I don't think the internationalized form can be longer, and so the ASCII
345      # form length check would be sufficient. If there is no ASCII form, then we have
346      # to check the UTF-8 encoding. The UTF-8 encoding could be up to about four times
347      # longer than the number of characters.
348      #
349      # See the length checks on the local part and the domain.
350      if ret.ascii_email and len(ret.ascii_email) > EMAIL_MAX_LENGTH:
351          if ret.ascii_email == ret.email:
352              reason = __get_length_reason(ret.ascii_email)
353          elif len(ret.email) > EMAIL_MAX_LENGTH:
354              # If there are more than 254 characters, then the ASCII
355              # form is definitely going to be too long.
356              reason = __get_length_reason(ret.email, utf8=True)
357          else:
358              reason = "(when converted to IDNA ASCII)"
359          raise EmailSyntaxError("The email address is too long {}.".format(reason))
360      if len(ret.email.encode("utf8")) > EMAIL_MAX_LENGTH:
361          if len(ret.email) > EMAIL_MAX_LENGTH:
362              # If there are more than 254 characters, then the UTF-8
363              # encoding is definitely going to be too long.
364              reason = __get_length_reason(ret.email, utf8=True)
365          else:
366              reason = "(when encoded in bytes)"
367          raise EmailSyntaxError("The email address is too long {}.".format(reason))
368  
369      if check_deliverability and not test_environment:
370          # Validate the email address's deliverability using DNS
371          # and update the return dict with metadata.
372          deliverability_info = validate_email_deliverability(
373              ret["domain"], ret["domain_i18n"], timeout, dns_resolver
374          )
375          for key, value in deliverability_info.items():
376              setattr(ret, key, value)
377  
378      return ret

381  def validate_email_local_part(local, allow_smtputf8=True, allow_empty_local=False):
382      # Validates the local part of an email address.
383  
384      if len(local) == 0:
385          if not allow_empty_local:
386              raise EmailSyntaxError("There must be something before the @-sign.")
387          else:
388              # The caller allows an empty local part. Useful for validating certain
389              # Postfix aliases.
390              return {
391                  "local_part": local,
392                  "ascii_local_part": local,
393                  "smtputf8": False,
394              }
395  
396      # RFC 5321 4.5.3.1.1
397      # We're checking the number of characters here. If the local part
398      # is ASCII-only, then that's the same as bytes (octets). If it's
399      # internationalized, then the UTF-8 encoding may be longer, but
400      # that may not be relevant. We will check the total address length
401      # instead.
402      if len(local) > LOCAL_PART_MAX_LENGTH:
403          reason = __get_length_reason(local, limit=LOCAL_PART_MAX_LENGTH)
404          raise EmailSyntaxError("The email address is too long before the @-sign {}.".format(reason))
405  
406      # Check the local part against the regular expression for the older ASCII requirements.
407      m = re.match(DOT_ATOM_TEXT + "\\Z", local)
408      if m:
409          # Return the local part unchanged and flag that SMTPUTF8 is not needed.
410          return {
411              "local_part": local,
412              "ascii_local_part": local,
413              "smtputf8": False,
414          }
415  
416      else:
417          # The local part failed the ASCII check. Now try the extended internationalized requirements.
418          m = re.match(DOT_ATOM_TEXT_INTL + "\\Z", local)
419          if not m:
420              # It's not a valid internationalized address either. Report which characters were not valid.
421              bad_chars = ', '.join(sorted(set(
422                  unicodedata.name(c, repr(c)) for c in local if not re.match(u"[" + (ATEXT if not allow_smtputf8 else ATEXT_INTL) + u"]", c)
423              )))
424              raise EmailSyntaxError("The email address contains invalid characters before the @-sign: %s." % bad_chars)
425  
426          # It would be valid if internationalized characters were allowed by the caller.
427          if not allow_smtputf8:
428              raise EmailSyntaxError("Internationalized characters before the @-sign are not supported.")
429  
430          # It's valid.
431  
432          # RFC 6532 section 3.1 also says that Unicode NFC normalization should be applied,
433          # so we'll return the normalized local part in the return value.
434          local = unicodedata.normalize("NFC", local)
435  
436          # Check for unsafe characters.
437          # Some of this may be redundant with the range U+0080 to U+10FFFF that is checked
438          # by DOT_ATOM_TEXT_INTL.
439          for i, c in enumerate(local):
440              category = unicodedata.category(c)
441              if category[0] in ("L", "N", "P", "S"):
442                  # letters, numbers, punctuation, and symbols are permitted
443                  pass
444              elif category[0] == "M":
445                  # combining character in first position would combine with something
446                  # outside of the email address if concatenated to the right, but are
447                  # otherwise permitted
448                  if i == 0:
449                      raise EmailSyntaxError("The email address contains an initial invalid character (%s)."
450                                             % unicodedata.name(c, repr(c)))
451              elif category[0] in ("Z", "C"):
452                  # spaces and line/paragraph characters (Z) and
453                  # control, format, surrogate, private use, and unassigned code points (C)
454                  raise EmailSyntaxError("The email address contains an invalid character (%s)."
455                                         % unicodedata.name(c, repr(c)))
456              else:
457                  # All categories should be handled above, but in case there is something new
458                  # in the future.
459                  raise EmailSyntaxError("The email address contains a character (%s; category %s) that may not be safe."
460                                         % (unicodedata.name(c, repr(c)), category))
461  
462          # Try encoding to UTF-8. Failure is possible with some characters like
463          # surrogate code points, but those are checked above. Still, we don't
464          # want to have an unhandled exception later.
465          try:
466              local.encode("utf8")
467          except ValueError:
468              raise EmailSyntaxError("The email address contains an invalid character.")
469  
470          # Flag that SMTPUTF8 will be required for deliverability.
471          return {
472              "local_part": local,
473              "ascii_local_part": None,  # no ASCII form is possible
474              "smtputf8": True,
475          }

478  def validate_email_domain_part(domain, test_environment=False, globally_deliverable=True):
479      # Empty?
480      if len(domain) == 0:
481          raise EmailSyntaxError("There must be something after the @-sign.")
482  
483      # Perform UTS-46 normalization, which includes casefolding, NFC normalization,
484      # and converting all label separators (the period/full stop, fullwidth full stop,
485      # ideographic full stop, and halfwidth ideographic full stop) to basic periods.
486      # It will also raise an exception if there is an invalid character in the input,
487      # such as "⒈" which is invalid because it would expand to include a period.
488      try:
489          domain = idna.uts46_remap(domain, std3_rules=False, transitional=False)
490      except idna.IDNAError as e:
491          raise EmailSyntaxError("The domain name %s contains invalid characters (%s)." % (domain, str(e)))
492  
493      # Now we can perform basic checks on the use of periods (since equivalent
494      # symbols have been mapped to periods). These checks are needed because the
495      # IDNA library doesn't handle well domains that have empty labels (i.e. initial
496      # dot, trailing dot, or two dots in a row).
497      if domain.endswith("."):
498          raise EmailSyntaxError("An email address cannot end with a period.")
499      if domain.startswith("."):
500          raise EmailSyntaxError("An email address cannot have a period immediately after the @-sign.")
501      if ".." in domain:
502          raise EmailSyntaxError("An email address cannot have two periods in a row.")
503  
504      # Regardless of whether international characters are actually used,
505      # first convert to IDNA ASCII. For ASCII-only domains, the transformation
506      # does nothing. If internationalized characters are present, the MTA
507      # must either support SMTPUTF8 or the mail client must convert the
508      # domain name to IDNA before submission.
509      #
510      # Unfortunately this step incorrectly 'fixes' domain names with leading
511      # periods by removing them, so we have to check for this above. It also gives
512      # a funky error message ("No input") when there are two periods in a
513      # row, also checked separately above.
514      try:
515          ascii_domain = idna.encode(domain, uts46=False).decode("ascii")
516      except idna.IDNAError as e:
517          if "Domain too long" in str(e):
518              # We can't really be more specific because UTS-46 normalization means
519              # the length check is applied to a string that is different from the
520              # one the user supplied. Also I'm not sure if the length check applies
521              # to the internationalized form, the IDNA ASCII form, or even both!
522              raise EmailSyntaxError("The email address is too long after the @-sign.")
523          raise EmailSyntaxError("The domain name %s contains invalid characters (%s)." % (domain, str(e)))
524  
525      # We may have been given an IDNA ASCII domain to begin with. Check
526      # that the domain actually conforms to IDNA. It could look like IDNA
527      # but not be actual IDNA. For ASCII-only domains, the conversion out
528      # of IDNA just gives the same thing back.
529      #
530      # This gives us the canonical internationalized form of the domain,
531      # which we should use in all error messages.
532      try:
533          domain_i18n = idna.decode(ascii_domain.encode('ascii'))
534      except idna.IDNAError as e:
535          raise EmailSyntaxError("The domain name %s is not valid IDNA (%s)." % (ascii_domain, str(e)))
536  
537      # RFC 5321 4.5.3.1.2
538      # We're checking the number of bytes (octets) here, which can be much
539      # higher than the number of characters in internationalized domains,
540      # on the assumption that the domain may be transmitted without SMTPUTF8
541      # as IDNA ASCII. This is also checked by idna.encode, so this exception
542      # is never reached.
543      if len(ascii_domain) > DOMAIN_MAX_LENGTH:
544          raise EmailSyntaxError("The email address is too long after the @-sign.")
545  
546      # A "dot atom text", per RFC 2822 3.2.4, but using the restricted
547      # characters allowed in a hostname (see ATEXT_HOSTNAME above).
548      DOT_ATOM_TEXT = ATEXT_HOSTNAME + r'(?:\.' + ATEXT_HOSTNAME + r')*'
549  
550      # Check the regular expression. This is probably entirely redundant
551      # with idna.decode, which also checks this format.
552      m = re.match(DOT_ATOM_TEXT + "\\Z", ascii_domain)
553      if not m:
554          raise EmailSyntaxError("The email address contains invalid characters after the @-sign.")
555  
556      if globally_deliverable:
557          # All publicly deliverable addresses have domain named with at least
558          # one period, and we'll consider the lack of a period a syntax error
559          # since that will match people's sense of what an email address looks
560          # like. We'll skip this in test environments to allow '@test' email
561          # addresses.
562          if "." not in ascii_domain and not (ascii_domain == "test" and test_environment):
563              raise EmailSyntaxError("The domain name %s is not valid. It should have a period." % domain_i18n)
564  
565          # We also know that all TLDs currently end with a letter.
566          if not re.search(r"[A-Za-z]\Z", ascii_domain):
567              raise EmailSyntaxError(
568                  "The domain name %s is not valid. It is not within a valid top-level domain." % domain_i18n
569              )
570  
571      # Check special-use and reserved domain names.
572      # Some might fail DNS-based deliverability checks, but that
573      # can be turned off, so we should fail them all sooner.
574      for d in SPECIAL_USE_DOMAIN_NAMES:
575          # See the note near the definition of SPECIAL_USE_DOMAIN_NAMES.
576          if d == "test" and test_environment:
577              continue
578  
579          if ascii_domain == d or ascii_domain.endswith("." + d):
580              raise EmailSyntaxError("The domain name %s is a special-use or reserved name that cannot be used with email." % domain_i18n)
581  
582      # Return the IDNA ASCII-encoded form of the domain, which is how it
583      # would be transmitted on the wire (except when used with SMTPUTF8
584      # possibly), as well as the canonical Unicode form of the domain,
585      # which is better for display purposes. This should also take care
586      # of RFC 6532 section 3.1's suggestion to apply Unicode NFC
587      # normalization to addresses.
588      return {
589          "ascii_domain": ascii_domain,
590          "domain": domain_i18n,
591      }

594  def validate_email_deliverability(domain, domain_i18n, timeout=DEFAULT_TIMEOUT, dns_resolver=None):
595      # Check that the domain resolves to an MX record. If there is no MX record,
596      # try an A or AAAA record which is a deprecated fallback for deliverability.
597      # (Note that changing the DEFAULT_TIMEOUT module-level attribute
598      #  will not change the default value of this method's timeout argument.)
599  
600      # If no dns.resolver.Resolver was given, get dnspython's default resolver.
601      # Override the default resolver's timeout. This may affect other uses of
602      # dnspython in this process.
603      if dns_resolver is None:
604          dns_resolver = dns.resolver.get_default_resolver()
605          dns_resolver.lifetime = timeout
606  
607      deliverability_info = {}
608  
609      def dns_resolver_resolve_shim(domain, record):
610          try:
611              # dns.resolver.Resolver.resolve is new to dnspython 2.x.
612              # https://dnspython.readthedocs.io/en/latest/resolver-class.html#dns.resolver.Resolver.resolve
613              return dns_resolver.resolve(domain, record)
614          except AttributeError:
615              # dnspython 2.x is only available in Python 3.6 and later. For earlier versions
616              # of Python, we maintain compatibility with dnspython 1.x which has a
617              # dnspython.resolver.Resolver.query method instead. The only difference is that
618              # query may treat the domain as relative and use the system's search domains,
619              # which we prevent by adding a "." to the domain name to make it absolute.
620              # dns.resolver.Resolver.query is deprecated in dnspython version 2.x.
621              # https://dnspython.readthedocs.io/en/latest/resolver-class.html#dns.resolver.Resolver.query
622              return dns_resolver.query(domain + ".", record)
623  
624      try:
625          # We need a way to check how timeouts are handled in the tests. So we
626          # have a secret variable that if set makes this method always test the
627          # handling of a timeout.
628          if getattr(validate_email_deliverability, 'TEST_CHECK_TIMEOUT', False):
629              raise dns.exception.Timeout()
630  
631          try:
632              # Try resolving for MX records.
633              response = dns_resolver_resolve_shim(domain, "MX")
634  
635              # For reporting, put them in priority order and remove the trailing dot in the qnames.
636              mtas = sorted([(r.preference, str(r.exchange).rstrip('.')) for r in response])
637  
638              # Remove "null MX" records from the list (their value is (0, ".") but we've stripped
639              # trailing dots, so the 'exchange' is just ""). If there was only a null MX record,
640              # email is not deliverable.
641              mtas = [(preference, exchange) for preference, exchange in mtas
642                      if exchange != ""]
643              if len(mtas) == 0:
644                  raise EmailUndeliverableError("The domain name %s does not accept email." % domain_i18n)
645  
646              deliverability_info["mx"] = mtas
647              deliverability_info["mx_fallback_type"] = None
648  
649          except (dns.resolver.NoNameservers, dns.resolver.NXDOMAIN, dns.resolver.NoAnswer):
650  
651              # If there was no MX record, fall back to an A record.
652              try:
653                  response = dns_resolver_resolve_shim(domain, "A")
654                  deliverability_info["mx"] = [(0, str(r)) for r in response]
655                  deliverability_info["mx_fallback_type"] = "A"
656              except (dns.resolver.NoNameservers, dns.resolver.NXDOMAIN, dns.resolver.NoAnswer):
657  
658                  # If there was no A record, fall back to an AAAA record.
659                  try:
660                      response = dns_resolver_resolve_shim(domain, "AAAA")
661                      deliverability_info["mx"] = [(0, str(r)) for r in response]
662                      deliverability_info["mx_fallback_type"] = "AAAA"
663                  except (dns.resolver.NoNameservers, dns.resolver.NXDOMAIN, dns.resolver.NoAnswer):
664  
665                      # If there was no MX, A, or AAAA record, then mail to
666                      # this domain is not deliverable.
667                      raise EmailUndeliverableError("The domain name %s does not exist." % domain_i18n)
668  
669          try:
670              # Check for a SPF reject all ("v=spf1 -all") record which indicates
671              # no emails are sent from this domain, which like a NULL MX record
672              # would indicate that the domain is not used for email.
673              response = dns_resolver_resolve_shim(domain, "TXT")
674              for rec in response:
675                  value = b"".join(rec.strings)
676                  if value.startswith(b"v=spf1 "):
677                      deliverability_info["spf"] = value.decode("ascii", errors='replace')
678                      if value == b"v=spf1 -all":
679                          raise EmailUndeliverableError("The domain name %s does not send email." % domain_i18n)
680          except dns.resolver.NoAnswer:
681              # No TXT records means there is no SPF policy, so we cannot take any action.
682              pass
683          except (dns.resolver.NoNameservers, dns.resolver.NXDOMAIN):
684              # Failure to resolve at this step will be ignored.
685              pass
686  
687      except dns.exception.Timeout:
688          # A timeout could occur for various reasons, so don't treat it as a failure.
689          return {
690              "unknown-deliverability": "timeout",
691          }
692  
693      except EmailUndeliverableError:
694          # Don't let these get clobbered by the wider except block below.
695          raise
696  
697      except Exception as e:
698          # Unhandled conditions should not propagate.
699          raise EmailUndeliverableError(
700              "There was an error while checking if the domain name in the email address is deliverable: " + str(e)
701          )
702  
703      return deliverability_info

706  def main():
707      import json
708  
709      def __utf8_input_shim(input_str):
710          if sys.version_info < (3,):
711              return input_str.decode("utf-8")
712          return input_str
713  
714      def __utf8_output_shim(output_str):
715          if sys.version_info < (3,):
716              return unicode_class(output_str).encode("utf-8")
717          return output_str
718  
719      if len(sys.argv) == 1:
720          # Validate the email addresses pased line-by-line on STDIN.
721          dns_resolver = caching_resolver()
722          for line in sys.stdin:
723              email = __utf8_input_shim(line.strip())
724              try:
725                  validate_email(email, dns_resolver=dns_resolver)
726              except EmailNotValidError as e:
727                  print(__utf8_output_shim("{} {}".format(email, e)))
728      else:
729          # Validate the email address passed on the command line.
730          email = __utf8_input_shim(sys.argv[1])
731          try:
732              result = validate_email(email)
733              print(json.dumps(result.as_dict(), indent=2, sort_keys=True, ensure_ascii=False))
734          except EmailNotValidError as e:
735              print(__utf8_output_shim(e))

737  
738  if __name__ == "__main__":
739      main()
