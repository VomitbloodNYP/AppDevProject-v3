20   class BooleanField(Field):
21       """ 
22       Represents an ``<input type="checkbox">``. Set the ``checked``-status by using the 
23       ``default``-option. Any value for ``default``, e.g. ``default="checked"`` puts 
24       ``checked`` into the html-element and sets the ``data`` to ``True`` 
25    
26       :param false_values: 
27           If provided, a sequence of strings each of which is an exact match 
28           string of what is considered a "false" value. Defaults to the tuple 
29           ``(False, "false", "")`` 
30       """
31   
32       widget = widgets.CheckboxInput()
33       false_values = (False, "false", "")
34   
35       def __init__(self, label=None, validators=None, false_values=None, **kwargs):
36           super().__init__(label, validators, **kwargs)
37           if false_values is not None:
38               self.false_values = false_values
39   
40       def process_data(self, value):
41           self.data = bool(value)
42   
43       def process_formdata(self, valuelist):
44           if not valuelist or valuelist[0] in self.false_values:
45               self.data = False
46           else:
47               self.data = True
48   
49       def _value(self):
50           if self.raw_data:
51               return str(self.raw_data[0])
52           return "y"

55   class StringField(Field):
56       """ 
57       This field is the base for most of the more complicated fields, and 
58       represents an ``<input type="text">``. 
59       """
60   
61       widget = widgets.TextInput()
62   
63       def process_formdata(self, valuelist):
64           if valuelist:
65               self.data = valuelist[0]
66   
67       def _value(self):
68           return str(self.data) if self.data is not None else ""

91   class FileField(Field):
92       """Renders a file upload field. 
93    
94       By default, the value will be the filename sent in the form data. 
95       WTForms **does not** deal with frameworks' file handling capabilities. 
96       A WTForms extension for a framework may replace the filename value 
97       with an object representing the uploaded data. 
98       """
99   
100      widget = widgets.FileInput()
101  
102      def _value(self):
103          # browser ignores value of file input for security
104          return False
105
