18   """A place to store TSIG keys."""
19   
20   import base64
21   
22   import dns.name
23   import dns.tsig
24   
25   
26   def from_text(textring):
27       """Convert a dictionary containing (textual DNS name, base64 secret) 
28       pairs into a binary keyring which has (dns.name.Name, bytes) pairs, or 
29       a dictionary containing (textual DNS name, (algorithm, base64 secret)) 
30       pairs into a binary keyring which has (dns.name.Name, dns.tsig.Key) pairs. 
31       @rtype: dict"""
32   
33       keyring = {}
34       for (name, value) in textring.items():
35           name = dns.name.from_text(name)
36           if isinstance(value, str):
37               keyring[name] = dns.tsig.Key(name, value).secret
38           else:
39               (algorithm, secret) = value
40               keyring[name] = dns.tsig.Key(name, secret, algorithm)
41       return keyring
42   
43   
44   def to_text(keyring):
45       """Convert a dictionary containing (dns.name.Name, dns.tsig.Key) pairs 
46       into a text keyring which has (textual DNS name, (textual algorithm, 
47       base64 secret)) pairs, or a dictionary containing (dns.name.Name, bytes) 
48       pairs into a text keyring which has (textual DNS name, base64 secret) pairs. 
49       @rtype: dict"""
50   
51       textring = {}
52       def b64encode(secret):
53           return base64.encodebytes(secret).decode().rstrip()
54       for (name, key) in keyring.items():
55           name = name.to_text()
56           if isinstance(key, bytes):
57               textring[name] = b64encode(key)
58           else:
59               if isinstance(key.secret, bytes):
60                   text_secret = b64encode(key.secret)
61               else:
62                   text_secret = str(key.secret)
63   
64               textring[name] = (key.algorithm.to_text(), text_secret)
65       return textring
