18   import socket
19   import struct
20   
21   import dns.ipv4
22   import dns.immutable
23   import dns.rdata
24   
25   try:
26       _proto_tcp = socket.getprotobyname('tcp')
27       _proto_udp = socket.getprotobyname('udp')
28   except OSError:
29       # Fall back to defaults in case /etc/protocols is unavailable.
30       _proto_tcp = 6
31       _proto_udp = 17
32   
33   @dns.immutable.immutable
34   class WKS(dns.rdata.Rdata):
35   
36       """WKS record"""
37   
38       # see: RFC 1035
39   
40       __slots__ = ['address', 'protocol', 'bitmap']
41   
42       def __init__(self, rdclass, rdtype, address, protocol, bitmap):
43           super().__init__(rdclass, rdtype)
44           self.address = self._as_ipv4_address(address)
45           self.protocol = self._as_uint8(protocol)
46           self.bitmap = self._as_bytes(bitmap)
47   
48       def to_text(self, origin=None, relativize=True, **kw):
49           bits = []
50           for i, byte in enumerate(self.bitmap):
51               for j in range(0, 8):
52                   if byte & (0x80 >> j):
53                       bits.append(str(i * 8 + j))
54           text = ' '.join(bits)
55           return '%s %d %s' % (self.address, self.protocol, text)
56   
57       @classmethod
58       def from_text(cls, rdclass, rdtype, tok, origin=None, relativize=True,
59                     relativize_to=None):
60           address = tok.get_string()
61           protocol = tok.get_string()
62           if protocol.isdigit():
63               protocol = int(protocol)
64           else:
65               protocol = socket.getprotobyname(protocol)
66           bitmap = bytearray()
67           for token in tok.get_remaining():
68               value = token.unescape().value
69               if value.isdigit():
70                   serv = int(value)
71               else:
72                   if protocol != _proto_udp and protocol != _proto_tcp:
73                       raise NotImplementedError("protocol must be TCP or UDP")
74                   if protocol == _proto_udp:
75                       protocol_text = "udp"
76                   else:
77                       protocol_text = "tcp"
78                   serv = socket.getservbyname(value, protocol_text)
79               i = serv // 8
80               l = len(bitmap)
81               if l < i + 1:
82                   for _ in range(l, i + 1):
83                       bitmap.append(0)
84               bitmap[i] = bitmap[i] | (0x80 >> (serv % 8))
85           bitmap = dns.rdata._truncate_bitmap(bitmap)
86           return cls(rdclass, rdtype, address, protocol, bitmap)
87   
88       def _to_wire(self, file, compress=None, origin=None, canonicalize=False):
89           file.write(dns.ipv4.inet_aton(self.address))
90           protocol = struct.pack('!B', self.protocol)
91           file.write(protocol)
92           file.write(self.bitmap)
93   
94       @classmethod
95       def from_wire_parser(cls, rdclass, rdtype, parser, origin=None):
96           address = parser.get_bytes(4)
97           protocol = parser.get_uint8()
98           bitmap = parser.get_remaining()
99           return cls(rdclass, rdtype, address, protocol, bitmap)
