15   class TransactionManager:
16       def reader(self):
17           """Begin a read-only transaction."""
18           raise NotImplementedError  # pragma: no cover
19   
20       def writer(self, replacement=False):
21           """Begin a writable transaction. 
22    
23           *replacement*, a ``bool``.  If `True`, the content of the 
24           transaction completely replaces any prior content.  If False, 
25           the default, then the content of the transaction updates the 
26           existing content. 
27           """
28           raise NotImplementedError  # pragma: no cover
29   
30       def origin_information(self):
31           """Returns a tuple 
32    
33               (absolute_origin, relativize, effective_origin) 
34    
35           giving the absolute name of the default origin for any 
36           relative domain names, the "effective origin", and whether 
37           names should be relativized.  The "effective origin" is the 
38           absolute origin if relativize is False, and the empty name if 
39           relativize is true.  (The effective origin is provided even 
40           though it can be computed from the absolute_origin and 
41           relativize setting because it avoids a lot of code 
42           duplication.) 
43    
44           If the returned names are `None`, then no origin information is 
45           available. 
46    
47           This information is used by code working with transactions to 
48           allow it to coordinate relativization.  The transaction code 
49           itself takes what it gets (i.e. does not change name 
50           relativity). 
51    
52           """
53           raise NotImplementedError  # pragma: no cover
54   
55       def get_class(self):
56           """The class of the transaction manager. 
57           """
58           raise NotImplementedError  # pragma: no cover
59   
60       def from_wire_origin(self):
61           """Origin to use in from_wire() calls. 
62           """
63           (absolute_origin, relativize, _) = self.origin_information()
64           if relativize:
65               return absolute_origin
66           else:
67               return None

82   def _ensure_immutable_rdataset(rdataset):
83       if rdataset is None or isinstance(rdataset, dns.rdataset.ImmutableRdataset):
84           return rdataset
85       return dns.rdataset.ImmutableRdataset(rdataset)
86   
87   def _ensure_immutable_node(node):
88       if node is None or node.is_immutable():
89           return node
90       return dns.node.ImmutableNode(node)

95       def __init__(self, manager, replacement=False, read_only=False):
96           self.manager = manager
97           self.replacement = replacement
98           self.read_only = read_only
99           self._ended = False
100          self._check_put_rdataset = []
101          self._check_delete_rdataset = []
102          self._check_delete_name = []

108      def get(self, name, rdtype, covers=dns.rdatatype.NONE):
109          """Return the rdataset associated with *name*, *rdtype*, and *covers*, 
110          or `None` if not found. 
111   
112          Note that the returned rdataset is immutable. 
113          """
114          self._check_ended()
115          if isinstance(name, str):
116              name = dns.name.from_text(name, None)
117          rdtype = dns.rdatatype.RdataType.make(rdtype)
118          rdataset = self._get_rdataset(name, rdtype, covers)
119          return _ensure_immutable_rdataset(rdataset)
120  
121      def get_node(self, name):
122          """Return the node at *name*, if any. 
123   
124          Returns an immutable node or ``None``. 
125          """
126          return _ensure_immutable_node(self._get_node(name))
127  
128      def _check_read_only(self):
129          if self.read_only:
130              raise ReadOnly
131  
132      def add(self, *args):
133          """Add records. 
134   
135          The arguments may be: 
136   
137              - rrset 
138   
139              - name, rdataset... 
140   
141              - name, ttl, rdata... 
142          """
143          self._check_ended()
144          self._check_read_only()
145          return self._add(False, args)
146  
147      def replace(self, *args):
148          """Replace the existing rdataset at the name with the specified 
149          rdataset, or add the specified rdataset if there was no existing 
150          rdataset. 
151   
152          The arguments may be: 
153   
154              - rrset 
155   
156              - name, rdataset... 
157   
158              - name, ttl, rdata... 
159   
160          Note that if you want to replace the entire node, you should do 
161          a delete of the name followed by one or more calls to add() or 
162          replace(). 
163          """
164          self._check_ended()
165          self._check_read_only()
166          return self._add(True, args)
167  
168      def delete(self, *args):
169          """Delete records. 
170   
171          It is not an error if some of the records are not in the existing 
172          set. 
173   
174          The arguments may be: 
175   
176              - rrset 
177   
178              - name 
179   
180              - name, rdataclass, rdatatype, [covers] 
181   
182              - name, rdataset... 
183   
184              - name, rdata... 
185          """
186          self._check_ended()
187          self._check_read_only()
188          return self._delete(False, args)
189  
190      def delete_exact(self, *args):
191          """Delete records. 
192   
193          The arguments may be: 
194   
195              - rrset 
196   
197              - name 
198   
199              - name, rdataclass, rdatatype, [covers] 
200   
201              - name, rdataset... 
202   
203              - name, rdata... 
204   
205          Raises dns.transaction.DeleteNotExact if some of the records 
206          are not in the existing set. 
207   
208          """
209          self._check_ended()
210          self._check_read_only()
211          return self._delete(True, args)

213      def name_exists(self, name):
214          """Does the specified name exist?"""
215          self._check_ended()
216          if isinstance(name, str):
217              name = dns.name.from_text(name, None)
218          return self._name_exists(name)

220      def update_serial(self, value=1, relative=True, name=dns.name.empty):
221          """Update the serial number. 
222   
223          *value*, an `int`, is an increment if *relative* is `True`, or the 
224          actual value to set if *relative* is `False`. 
225   
226          Raises `KeyError` if there is no SOA rdataset at *name*. 
227   
228          Raises `ValueError` if *value* is negative or if the increment is 
229          so large that it would cause the new serial to be less than the 
230          prior value. 
231          """
232          self._check_ended()
233          if value < 0:
234              raise ValueError('negative update_serial() value')
235          if isinstance(name, str):
236              name = dns.name.from_text(name, None)
237          rdataset = self._get_rdataset(name, dns.rdatatype.SOA,
238                                        dns.rdatatype.NONE)
239          if rdataset is None or len(rdataset) == 0:
240              raise KeyError
241          if relative:
242              serial = dns.serial.Serial(rdataset[0].serial) + value
243          else:
244              serial = dns.serial.Serial(value)
245          serial = serial.value  # convert back to int
246          if serial == 0:
247              serial = 1
248          rdata = rdataset[0].replace(serial=serial)
249          new_rdataset = dns.rdataset.from_rdata(rdataset.ttl, rdata)
250          self.replace(name, new_rdataset)
251  
252      def __iter__(self):
253          self._check_ended()
254          return self._iterate_rdatasets()
255  
256      def changed(self):
257          """Has this transaction changed anything? 
258   
259          For read-only transactions, the result is always `False`. 
260   
261          For writable transactions, the result is `True` if at some time 
262          during the life of the transaction, the content was changed. 
263          """
264          self._check_ended()
265          return self._changed()
266  
267      def commit(self):
268          """Commit the transaction. 
269   
270          Normally transactions are used as context managers and commit 
271          or rollback automatically, but it may be done explicitly if needed. 
272          A ``dns.transaction.Ended`` exception will be raised if you try 
273          to use a transaction after it has been committed or rolled back. 
274   
275          Raises an exception if the commit fails (in which case the transaction 
276          is also rolled back. 
277          """
278          self._end(True)
279  
280      def rollback(self):
281          """Rollback the transaction. 
282   
283          Normally transactions are used as context managers and commit 
284          or rollback automatically, but it may be done explicitly if needed. 
285          A ``dns.transaction.AlreadyEnded`` exception will be raised if you try 
286          to use a transaction after it has been committed or rolled back. 
287   
288          Rollback cannot otherwise fail. 
289          """
290          self._end(False)

292      def check_put_rdataset(self, check):
293          """Call *check* before putting (storing) an rdataset. 
294   
295          The function is called with the transaction, the name, and the rdataset. 
296   
297          The check function may safely make non-mutating transaction method 
298          calls, but behavior is undefined if mutating transaction methods are 
299          called.  The check function should raise an exception if it objects to 
300          the put, and otherwise should return ``None``. 
301          """
302          self._check_put_rdataset.append(check)
303  
304      def check_delete_rdataset(self, check):
305          """Call *check* before deleting an rdataset. 
306   
307          The function is called with the transaction, the name, the rdatatype, 
308          and the covered rdatatype. 
309   
310          The check function may safely make non-mutating transaction method 
311          calls, but behavior is undefined if mutating transaction methods are 
312          called.  The check function should raise an exception if it objects to 
313          the put, and otherwise should return ``None``. 
314          """
315          self._check_delete_rdataset.append(check)
316  
317      def check_delete_name(self, check):
318          """Call *check* before putting (storing) an rdataset. 
319   
320          The function is called with the transaction and the name. 
321   
322          The check function may safely make non-mutating transaction method 
323          calls, but behavior is undefined if mutating transaction methods are 
324          called.  The check function should raise an exception if it objects to 
325          the put, and otherwise should return ``None``. 
326          """
327          self._check_delete_name.append(check)

333      def _raise_if_not_empty(self, method, args):
334          if len(args) != 0:
335              raise TypeError(f'extra parameters to {method}')
336  
337      def _rdataset_from_args(self, method, deleting, args):
338          try:
339              arg = args.popleft()
340              if isinstance(arg, dns.rrset.RRset):
341                  rdataset = arg.to_rdataset()
342              elif isinstance(arg, dns.rdataset.Rdataset):
343                  rdataset = arg
344              else:
345                  if deleting:
346                      ttl = 0
347                  else:
348                      if isinstance(arg, int):
349                          ttl = arg
350                          if ttl > dns.ttl.MAX_TTL:
351                              raise ValueError(f'{method}: TTL value too big')
352                      else:
353                          raise TypeError(f'{method}: expected a TTL')
354                      arg = args.popleft()
355                  if isinstance(arg, dns.rdata.Rdata):
356                      rdataset = dns.rdataset.from_rdata(ttl, arg)
357                  else:
358                      raise TypeError(f'{method}: expected an Rdata')
359              return rdataset
360          except IndexError:
361              if deleting:
362                  return None
363              else:
364                  # reraise
365                  raise TypeError(f'{method}: expected more arguments')
366  
367      def _add(self, replace, args):
368          try:
369              args = collections.deque(args)
370              if replace:
371                  method = 'replace()'
372              else:
373                  method = 'add()'
374              arg = args.popleft()
375              if isinstance(arg, str):
376                  arg = dns.name.from_text(arg, None)
377              if isinstance(arg, dns.name.Name):
378                  name = arg
379                  rdataset = self._rdataset_from_args(method, False, args)
380              elif isinstance(arg, dns.rrset.RRset):
381                  rrset = arg
382                  name = rrset.name
383                  # rrsets are also rdatasets, but they don't print the
384                  # same and can't be stored in nodes, so convert.
385                  rdataset = rrset.to_rdataset()
386              else:
387                  raise TypeError(f'{method} requires a name or RRset ' +
388                                  'as the first argument')
389              if rdataset.rdclass != self.manager.get_class():
390                  raise ValueError(f'{method} has objects of wrong RdataClass')
391              if rdataset.rdtype == dns.rdatatype.SOA:
392                  (_, _, origin) = self._origin_information()
393                  if name != origin:
394                      raise ValueError(f'{method} has non-origin SOA')
395              self._raise_if_not_empty(method, args)
396              if not replace:
397                  existing = self._get_rdataset(name, rdataset.rdtype,
398                                                rdataset.covers)
399                  if existing is not None:
400                      if isinstance(existing, dns.rdataset.ImmutableRdataset):
401                          trds = dns.rdataset.Rdataset(existing.rdclass,
402                                                       existing.rdtype,
403                                                       existing.covers)
404                          trds.update(existing)
405                          existing = trds
406                      rdataset = existing.union(rdataset)
407              self._checked_put_rdataset(name, rdataset)
408          except IndexError:
409              raise TypeError(f'not enough parameters to {method}')
410  
411      def _delete(self, exact, args):
412          try:
413              args = collections.deque(args)
414              if exact:
415                  method = 'delete_exact()'
416              else:
417                  method = 'delete()'
418              arg = args.popleft()
419              if isinstance(arg, str):
420                  arg = dns.name.from_text(arg, None)
421              if isinstance(arg, dns.name.Name):
422                  name = arg
423                  if len(args) > 0 and (isinstance(args[0], int) or
424                                        isinstance(args[0], str)):
425                      # deleting by type and (optionally) covers
426                      rdtype = dns.rdatatype.RdataType.make(args.popleft())
427                      if len(args) > 0:
428                          covers = dns.rdatatype.RdataType.make(args.popleft())
429                      else:
430                          covers = dns.rdatatype.NONE
431                      self._raise_if_not_empty(method, args)
432                      existing = self._get_rdataset(name, rdtype, covers)
433                      if existing is None:
434                          if exact:
435                              raise DeleteNotExact(f'{method}: missing rdataset')
436                      else:
437                          self._delete_rdataset(name, rdtype, covers)
438                      return
439                  else:
440                      rdataset = self._rdataset_from_args(method, True, args)
441              elif isinstance(arg, dns.rrset.RRset):
442                  rdataset = arg  # rrsets are also rdatasets
443                  name = rdataset.name
444              else:
445                  raise TypeError(f'{method} requires a name or RRset ' +
446                                  'as the first argument')
447              self._raise_if_not_empty(method, args)
448              if rdataset:
449                  if rdataset.rdclass != self.manager.get_class():
450                      raise ValueError(f'{method} has objects of wrong '
451                                       'RdataClass')
452                  existing = self._get_rdataset(name, rdataset.rdtype,
453                                                rdataset.covers)
454                  if existing is not None:
455                      if exact:
456                          intersection = existing.intersection(rdataset)
457                          if intersection != rdataset:
458                              raise DeleteNotExact(f'{method}: missing rdatas')
459                      rdataset = existing.difference(rdataset)
460                      if len(rdataset) == 0:
461                          self._checked_delete_rdataset(name, rdataset.rdtype,
462                                                        rdataset.covers)
463                      else:
464                          self._checked_put_rdataset(name, rdataset)
465                  elif exact:
466                      raise DeleteNotExact(f'{method}: missing rdataset')
467              else:
468                  if exact and not self._name_exists(name):
469                      raise DeleteNotExact(f'{method}: name not known')
470                  self._checked_delete_name(name)
471          except IndexError:
472              raise TypeError(f'not enough parameters to {method}')
473  
474      def _check_ended(self):
475          if self._ended:
476              raise AlreadyEnded
477  
478      def _end(self, commit):
479          self._check_ended()
480          if self._ended:
481              raise AlreadyEnded
482          try:
483              self._end_transaction(commit)
484          finally:
485              self._ended = True
486  
487      def _checked_put_rdataset(self, name, rdataset):
488          for check in self._check_put_rdataset:
489              check(self, name, rdataset)
490          self._put_rdataset(name, rdataset)
491  
492      def _checked_delete_rdataset(self, name, rdtype, covers):
493          for check in self._check_delete_rdataset:
494              check(self, name, rdtype, covers)
495          self._delete_rdataset(name, rdtype, covers)
496  
497      def _checked_delete_name(self, name):
498          for check in self._check_delete_name:
499              check(self, name)
500          self._delete_name(name)

506      def __enter__(self):
507          return self
508  
509      def __exit__(self, exc_type, exc_val, exc_tb):
510          if not self._ended:
511              if exc_type is None:
512                  self.commit()
513              else:
514                  self.rollback()
515          return False
