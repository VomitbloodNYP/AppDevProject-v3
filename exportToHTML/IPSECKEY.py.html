18   import struct
19   import base64
20   
21   import dns.exception
22   import dns.immutable
23   import dns.rdtypes.util
24   
25   
26   class Gateway(dns.rdtypes.util.Gateway):
27       name = 'IPSECKEY gateway'
28   
29   @dns.immutable.immutable
30   class IPSECKEY(dns.rdata.Rdata):
31   
32       """IPSECKEY record"""
33   
34       # see: RFC 4025
35   
36       __slots__ = ['precedence', 'gateway_type', 'algorithm', 'gateway', 'key']
37   
38       def __init__(self, rdclass, rdtype, precedence, gateway_type, algorithm,
39                    gateway, key):
40           super().__init__(rdclass, rdtype)
41           gateway = Gateway(gateway_type, gateway)
42           self.precedence = self._as_uint8(precedence)
43           self.gateway_type = gateway.type
44           self.algorithm = self._as_uint8(algorithm)
45           self.gateway = gateway.gateway
46           self.key = self._as_bytes(key)
47   
48       def to_text(self, origin=None, relativize=True, **kw):
49           gateway = Gateway(self.gateway_type, self.gateway).to_text(origin,
50                                                                      relativize)
51           return '%d %d %d %s %s' % (self.precedence, self.gateway_type,
52                                      self.algorithm, gateway,
53                                      dns.rdata._base64ify(self.key, **kw))
54   
55       @classmethod
56       def from_text(cls, rdclass, rdtype, tok, origin=None, relativize=True,
57                     relativize_to=None):
58           precedence = tok.get_uint8()
59           gateway_type = tok.get_uint8()
60           algorithm = tok.get_uint8()
61           gateway = Gateway.from_text(gateway_type, tok, origin, relativize,
62                                       relativize_to)
63           b64 = tok.concatenate_remaining_identifiers().encode()
64           key = base64.b64decode(b64)
65           return cls(rdclass, rdtype, precedence, gateway_type, algorithm,
66                      gateway.gateway, key)
67   
68       def _to_wire(self, file, compress=None, origin=None, canonicalize=False):
69           header = struct.pack("!BBB", self.precedence, self.gateway_type,
70                                self.algorithm)
71           file.write(header)
72           Gateway(self.gateway_type, self.gateway).to_wire(file, compress,
73                                                            origin, canonicalize)
74           file.write(self.key)
75   
76       @classmethod
77       def from_wire_parser(cls, rdclass, rdtype, parser, origin=None):
78           header = parser.get_struct('!BBB')
79           gateway_type = header[1]
80           gateway = Gateway.from_wire_parser(gateway_type, parser, origin)
81           key = parser.get_remaining()
82           return cls(rdclass, rdtype, header[0], gateway_type, header[2],
83                      gateway.gateway, key)
