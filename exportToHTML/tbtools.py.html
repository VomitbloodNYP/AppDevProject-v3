124  def _process_traceback(
125      exc: BaseException,
126      te: t.Optional[traceback.TracebackException] = None,
127      *,
128      skip: int = 0,
129      hide: bool = True,
130  ) -> traceback.TracebackException:
131      if te is None:
132          te = traceback.TracebackException.from_exception(exc, lookup_lines=False)
133  
134      # Get the frames the same way StackSummary.extract did, in order
135      # to match each frame with the FrameSummary to augment.
136      frame_gen = traceback.walk_tb(exc.__traceback__)
137      limit = getattr(sys, "tracebacklimit", None)
138  
139      if limit is not None:
140          if limit < 0:
141              limit = 0
142  
143          frame_gen = itertools.islice(frame_gen, limit)
144  
145      if skip:
146          frame_gen = itertools.islice(frame_gen, skip, None)
147          del te.stack[:skip]
148  
149      new_stack: t.List[DebugFrameSummary] = []
150      hidden = False
151  
152      # Match each frame with the FrameSummary that was generated.
153      # Hide frames using Paste's __traceback_hide__ rules. Replace
154      # all visible FrameSummary with DebugFrameSummary.
155      for (f, _), fs in zip(frame_gen, te.stack):
156          if hide:
157              hide_value = f.f_locals.get("__traceback_hide__", False)
158  
159              if hide_value in {"before", "before_and_this"}:
160                  new_stack = []
161                  hidden = False
162  
163                  if hide_value == "before_and_this":
164                      continue
165              elif hide_value in {"reset", "reset_and_this"}:
166                  hidden = False
167  
168                  if hide_value == "reset_and_this":
169                      continue
170              elif hide_value in {"after", "after_and_this"}:
171                  hidden = True
172  
173                  if hide_value == "after_and_this":
174                      continue
175              elif hide_value or hidden:
176                  continue
177  
178          frame_args: t.Dict[str, t.Any] = {
179              "filename": fs.filename,
180              "lineno": fs.lineno,
181              "name": fs.name,
182              "locals": f.f_locals,
183              "globals": f.f_globals,
184          }
185  
186          if hasattr(fs, "colno"):
187              frame_args["colno"] = fs.colno  # type: ignore[attr-defined]
188              frame_args["end_colno"] = fs.end_colno  # type: ignore[attr-defined]
189  
190          new_stack.append(DebugFrameSummary(**frame_args))
191  
192      # The codeop module is used to compile code from the interactive
193      # debugger. Hide any codeop frames from the bottom of the traceback.
194      while new_stack:
195          module = new_stack[0].global_ns.get("__name__")
196  
197          if module is None:
198              module = new_stack[0].local_ns.get("__name__")
199  
200          if module == "codeop":
201              del new_stack[0]
202          else:
203              break
204  
205      te.stack[:] = new_stack
206  
207      if te.__context__:
208          context_exc = t.cast(BaseException, exc.__context__)
209          te.__context__ = _process_traceback(context_exc, te.__context__, hide=hide)
210  
211      if te.__cause__:
212          cause_exc = t.cast(BaseException, exc.__cause__)
213          te.__cause__ = _process_traceback(cause_exc, te.__cause__, hide=hide)
214  
215      return te

218  class DebugTraceback:
219      __slots__ = ("_te", "_cache_all_tracebacks", "_cache_all_frames")
220  
221      def __init__(
222          self,
223          exc: BaseException,
224          te: t.Optional[traceback.TracebackException] = None,
225          *,
226          skip: int = 0,
227          hide: bool = True,
228      ) -> None:
229          self._te = _process_traceback(exc, te, skip=skip, hide=hide)
230  
231      def __str__(self) -> str:
232          return f"<{type(self).__name__} {self._te}>"
233  
234      @cached_property
235      def all_tracebacks(
236          self,
237      ) -> t.List[t.Tuple[t.Optional[str], traceback.TracebackException]]:
238          out = []
239          current = self._te
240  
241          while current is not None:
242              if current.__cause__ is not None:
243                  chained_msg = (
244                      "The above exception was the direct cause of the"
245                      " following exception"
246                  )
247                  chained_exc = current.__cause__
248              elif current.__context__ is not None and not current.__suppress_context__:
249                  chained_msg = (
250                      "During handling of the above exception, another"
251                      " exception occurred"
252                  )
253                  chained_exc = current.__context__
254              else:
255                  chained_msg = None
256                  chained_exc = None
257  
258              out.append((chained_msg, current))
259              current = chained_exc
260  
261          return out
262  
263      @cached_property
264      def all_frames(self) -> t.List["DebugFrameSummary"]:
265          return [
266              f for _, te in self.all_tracebacks for f in te.stack  # type: ignore[misc]
267          ]
268  
269      def render_traceback_text(self) -> str:
270          return "".join(self._te.format())
271  
272      def render_traceback_html(self, include_title: bool = True) -> str:
273          library_frames = [f.is_library for f in self.all_frames]
274          mark_library = 0 < sum(library_frames) < len(library_frames)
275          rows = []
276  
277          if not library_frames:
278              classes = "traceback noframe-traceback"
279          else:
280              classes = "traceback"
281  
282              for msg, current in reversed(self.all_tracebacks):
283                  row_parts = []
284  
285                  if msg is not None:
286                      row_parts.append(f'<li><div class="exc-divider">{msg}:</div>')
287  
288                  for frame in current.stack:
289                      frame = t.cast(DebugFrameSummary, frame)
290                      info = f' title="{escape(frame.info)}"' if frame.info else ""
291                      row_parts.append(f"<li{info}>{frame.render_html(mark_library)}")
292  
293                  rows.append("\n".join(row_parts))
294  
295          is_syntax_error = issubclass(self._te.exc_type, SyntaxError)
296  
297          if include_title:
298              if is_syntax_error:
299                  title = "Syntax Error"
300              else:
301                  title = "Traceback <em>(most recent call last)</em>:"
302          else:
303              title = ""
304  
305          exc_full = escape("".join(self._te.format_exception_only()))
306  
307          if is_syntax_error:
308              description = f"<pre class=syntaxerror>{exc_full}</pre>"
309          else:
310              description = f"<blockquote>{exc_full}</blockquote>"
311  
312          return SUMMARY_HTML % {
313              "classes": classes,
314              "title": f"<h3>{title}</h3>",
315              "frames": "\n".join(rows),
316              "description": description,
317          }

319      def render_debugger_html(
320          self, evalex: bool, secret: str, evalex_trusted: bool
321      ) -> str:
322          exc_lines = list(self._te.format_exception_only())
323          plaintext = "".join(self._te.format())
324          return PAGE_HTML % {
325              "evalex": "true" if evalex else "false",
326              "evalex_trusted": "true" if evalex_trusted else "false",
327              "console": "false",
328              "title": exc_lines[0],
329              "exception": escape("".join(exc_lines)),
330              "exception_type": escape(self._te.exc_type.__name__),
331              "summary": self.render_traceback_html(include_title=False),
332              "plaintext": escape(plaintext),
333              "plaintext_cs": re.sub("-{2,}", "-", plaintext),
334              "secret": secret,
335          }

338  class DebugFrameSummary(traceback.FrameSummary):
339      """A :class:`traceback.FrameSummary` that can evaluate code in the 
340      frame's namespace. 
341      """
342  
343      __slots__ = (
344          "local_ns",
345          "global_ns",
346          "_cache_info",
347          "_cache_is_library",
348          "_cache_console",
349      )
350  
351      def __init__(
352          self,
353          *,
354          locals: t.Dict[str, t.Any],
355          globals: t.Dict[str, t.Any],
356          **kwargs: t.Any,
357      ) -> None:
358          super().__init__(locals=None, **kwargs)
359          self.local_ns = locals
360          self.global_ns = globals
361  
362      @cached_property
363      def info(self) -> t.Optional[str]:
364          return self.local_ns.get("__traceback_info__")
365  
366      @cached_property
367      def is_library(self) -> bool:
368          return any(
369              self.filename.startswith((path, os.path.realpath(path)))
370              for path in sysconfig.get_paths().values()
371          )
372  
373      @cached_property
374      def console(self) -> Console:
375          return Console(self.global_ns, self.local_ns)
376  
377      def eval(self, code: str) -> t.Any:
378          return self.console.eval(code)
379  
380      def render_html(self, mark_library: bool) -> str:
381          context = 5
382          lines = linecache.getlines(self.filename)
383          line_idx = self.lineno - 1  # type: ignore[operator]
384          start_idx = max(0, line_idx - context)
385          stop_idx = min(len(lines), line_idx + context + 1)
386          rendered_lines = []
387  
388          def render_line(line: str, cls: str) -> None:
389              line = line.expandtabs().rstrip()
390              stripped_line = line.strip()
391              prefix = len(line) - len(stripped_line)
392              colno = getattr(self, "colno", 0)
393              end_colno = getattr(self, "end_colno", 0)
394  
395              if cls == "current" and colno and end_colno:
396                  arrow = (
397                      f'\n<span class="ws">{" " * prefix}</span>'
398                      f'{" " * (colno - prefix)}{"^" * (end_colno - colno)}'
399                  )
400              else:
401                  arrow = ""
402  
403              rendered_lines.append(
404                  f'<pre class="line {cls}"><span class="ws">{" " * prefix}</span>'
405                  f"{escape(stripped_line) if stripped_line else ' '}"
406                  f"{arrow if arrow else ''}</pre>"
407              )
408  
409          if lines:
410              for line in lines[start_idx:line_idx]:
411                  render_line(line, "before")
412  
413              render_line(lines[line_idx], "current")
414  
415              for line in lines[line_idx + 1 : stop_idx]:
416                  render_line(line, "after")
417  
418          return FRAME_HTML % {
419              "id": id(self),
420              "filename": escape(self.filename),
421              "lineno": self.lineno,
422              "function_name": escape(self.name),
423              "lines": "\n".join(rendered_lines),
424              "library": "library" if mark_library and self.is_library else "",
425          }
