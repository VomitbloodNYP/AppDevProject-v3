19   import dns.rdtypes.euibase
20   import dns.immutable
21   
22   
23   @dns.immutable.immutable
24   class EUI64(dns.rdtypes.euibase.EUIBase):
25   
26       """EUI64 record"""
27   
28       # see: rfc7043.txt
29   
30       byte_len = 8  # 0123456789abcdef (in hex)
31       text_len = byte_len * 3 - 1  # 01-23-45-67-89-ab-cd-ef
