110  class Zone(dns.transaction.TransactionManager):
111  
112      """A DNS zone. 
113   
114      A ``Zone`` is a mapping from names to nodes.  The zone object may be 
115      treated like a Python dictionary, e.g. ``zone[name]`` will retrieve 
116      the node associated with that name.  The *name* may be a 
117      ``dns.name.Name object``, or it may be a string.  In either case, 
118      if the name is relative it is treated as relative to the origin of 
119      the zone. 
120      """
121  
122      node_factory = dns.node.Node
123  
124      __slots__ = ['rdclass', 'origin', 'nodes', 'relativize']
125  
126      def __init__(self, origin, rdclass=dns.rdataclass.IN, relativize=True):
127          """Initialize a zone object. 
128   
129          *origin* is the origin of the zone.  It may be a ``dns.name.Name``, 
130          a ``str``, or ``None``.  If ``None``, then the zone's origin will 
131          be set by the first ``$ORIGIN`` line in a zone file. 
132   
133          *rdclass*, an ``int``, the zone's rdata class; the default is class IN. 
134   
135          *relativize*, a ``bool``, determine's whether domain names are 
136          relativized to the zone's origin.  The default is ``True``. 
137          """
138  
139          if origin is not None:
140              if isinstance(origin, str):
141                  origin = dns.name.from_text(origin)
142              elif not isinstance(origin, dns.name.Name):
143                  raise ValueError("origin parameter must be convertible to a "
144                                   "DNS name")
145              if not origin.is_absolute():
146                  raise ValueError("origin parameter must be an absolute name")
147          self.origin = origin
148          self.rdclass = rdclass
149          self.nodes = {}
150          self.relativize = relativize
151  
152      def __eq__(self, other):
153          """Two zones are equal if they have the same origin, class, and 
154          nodes. 
155   
156          Returns a ``bool``. 
157          """
158  
159          if not isinstance(other, Zone):
160              return False
161          if self.rdclass != other.rdclass or \
162             self.origin != other.origin or \
163             self.nodes != other.nodes:
164              return False
165          return True
166  
167      def __ne__(self, other):
168          """Are two zones not equal? 
169   
170          Returns a ``bool``. 
171          """
172  
173          return not self.__eq__(other)
174  
175      def _validate_name(self, name):
176          if isinstance(name, str):
177              name = dns.name.from_text(name, None)
178          elif not isinstance(name, dns.name.Name):
179              raise KeyError("name parameter must be convertible to a DNS name")
180          if name.is_absolute():
181              if not name.is_subdomain(self.origin):
182                  raise KeyError(
183                      "name parameter must be a subdomain of the zone origin")
184              if self.relativize:
185                  name = name.relativize(self.origin)
186          elif not self.relativize:
187              # We have a relative name in a non-relative zone, so derelativize.
188              if self.origin is None:
189                  raise KeyError('no zone origin is defined')
190              name = name.derelativize(self.origin)
191          return name
192  
193      def __getitem__(self, key):
194          key = self._validate_name(key)
195          return self.nodes[key]
196  
197      def __setitem__(self, key, value):
198          key = self._validate_name(key)
199          self.nodes[key] = value
200  
201      def __delitem__(self, key):
202          key = self._validate_name(key)
203          del self.nodes[key]
204  
205      def __iter__(self):
206          return self.nodes.__iter__()
207  
208      def keys(self):
209          return self.nodes.keys()
210  
211      def values(self):
212          return self.nodes.values()
213  
214      def items(self):
215          return self.nodes.items()
216  
217      def get(self, key):
218          key = self._validate_name(key)
219          return self.nodes.get(key)
220  
221      def __contains__(self, key):
222          key = self._validate_name(key)
223          return key in self.nodes

225      def find_node(self, name, create=False):
226          """Find a node in the zone, possibly creating it. 
227   
228          *name*: the name of the node to find. 
229          The value may be a ``dns.name.Name`` or a ``str``.  If absolute, the 
230          name must be a subdomain of the zone's origin.  If ``zone.relativize`` 
231          is ``True``, then the name will be relativized. 
232   
233          *create*, a ``bool``.  If true, the node will be created if it does 
234          not exist. 
235   
236          Raises ``KeyError`` if the name is not known and create was 
237          not specified, or if the name was not a subdomain of the origin. 
238   
239          Returns a ``dns.node.Node``. 
240          """
241  
242          name = self._validate_name(name)
243          node = self.nodes.get(name)
244          if node is None:
245              if not create:
246                  raise KeyError
247              node = self.node_factory()
248              self.nodes[name] = node
249          return node
250  
251      def get_node(self, name, create=False):
252          """Get a node in the zone, possibly creating it. 
253   
254          This method is like ``find_node()``, except it returns None instead 
255          of raising an exception if the node does not exist and creation 
256          has not been requested. 
257   
258          *name*: the name of the node to find. 
259          The value may be a ``dns.name.Name`` or a ``str``.  If absolute, the 
260          name must be a subdomain of the zone's origin.  If ``zone.relativize`` 
261          is ``True``, then the name will be relativized. 
262   
263          *create*, a ``bool``.  If true, the node will be created if it does 
264          not exist. 
265   
266          Raises ``KeyError`` if the name is not known and create was 
267          not specified, or if the name was not a subdomain of the origin. 
268   
269          Returns a ``dns.node.Node`` or ``None``. 
270          """
271  
272          try:
273              node = self.find_node(name, create)
274          except KeyError:
275              node = None
276          return node
277  
278      def delete_node(self, name):
279          """Delete the specified node if it exists. 
280   
281          *name*: the name of the node to find. 
282          The value may be a ``dns.name.Name`` or a ``str``.  If absolute, the 
283          name must be a subdomain of the zone's origin.  If ``zone.relativize`` 
284          is ``True``, then the name will be relativized. 
285   
286          It is not an error if the node does not exist. 
287          """
288  
289          name = self._validate_name(name)
290          if name in self.nodes:
291              del self.nodes[name]

293      def find_rdataset(self, name, rdtype, covers=dns.rdatatype.NONE,
294                        create=False):
295          """Look for an rdataset with the specified name and type in the zone, 
296          and return an rdataset encapsulating it. 
297   
298          The rdataset returned is not a copy; changes to it will change 
299          the zone. 
300   
301          KeyError is raised if the name or type are not found. 
302   
303          *name*: the name of the node to find. 
304          The value may be a ``dns.name.Name`` or a ``str``.  If absolute, the 
305          name must be a subdomain of the zone's origin.  If ``zone.relativize`` 
306          is ``True``, then the name will be relativized. 
307   
308          *rdtype*, an ``int`` or ``str``, the rdata type desired. 
309   
310          *covers*, an ``int`` or ``str`` or ``None``, the covered type. 
311          Usually this value is ``dns.rdatatype.NONE``, but if the 
312          rdtype is ``dns.rdatatype.SIG`` or ``dns.rdatatype.RRSIG``, 
313          then the covers value will be the rdata type the SIG/RRSIG 
314          covers.  The library treats the SIG and RRSIG types as if they 
315          were a family of types, e.g. RRSIG(A), RRSIG(NS), RRSIG(SOA). 
316          This makes RRSIGs much easier to work with than if RRSIGs 
317          covering different rdata types were aggregated into a single 
318          RRSIG rdataset. 
319   
320          *create*, a ``bool``.  If true, the node will be created if it does 
321          not exist. 
322   
323          Raises ``KeyError`` if the name is not known and create was 
324          not specified, or if the name was not a subdomain of the origin. 
325   
326          Returns a ``dns.rdataset.Rdataset``. 
327          """
328  
329          name = self._validate_name(name)
330          rdtype = dns.rdatatype.RdataType.make(rdtype)
331          if covers is not None:
332              covers = dns.rdatatype.RdataType.make(covers)
333          node = self.find_node(name, create)
334          return node.find_rdataset(self.rdclass, rdtype, covers, create)
335

337                       create=False):
338          """Look for an rdataset with the specified name and type in the zone. 
339   
340          This method is like ``find_rdataset()``, except it returns None instead 
341          of raising an exception if the rdataset does not exist and creation 
342          has not been requested. 
343   
344          The rdataset returned is not a copy; changes to it will change 
345          the zone. 
346   
347          *name*: the name of the node to find. 
348          The value may be a ``dns.name.Name`` or a ``str``.  If absolute, the 
349          name must be a subdomain of the zone's origin.  If ``zone.relativize`` 
350          is ``True``, then the name will be relativized. 
351   
352          *rdtype*, an ``int`` or ``str``, the rdata type desired. 
353   
354          *covers*, an ``int`` or ``str`` or ``None``, the covered type. 
355          Usually this value is ``dns.rdatatype.NONE``, but if the 
356          rdtype is ``dns.rdatatype.SIG`` or ``dns.rdatatype.RRSIG``, 
357          then the covers value will be the rdata type the SIG/RRSIG 
358          covers.  The library treats the SIG and RRSIG types as if they 
359          were a family of types, e.g. RRSIG(A), RRSIG(NS), RRSIG(SOA). 
360          This makes RRSIGs much easier to work with than if RRSIGs 
361          covering different rdata types were aggregated into a single 
362          RRSIG rdataset. 
363   
364          *create*, a ``bool``.  If true, the node will be created if it does 
365          not exist. 
366   
367          Raises ``KeyError`` if the name is not known and create was 
368          not specified, or if the name was not a subdomain of the origin. 
369   
370          Returns a ``dns.rdataset.Rdataset`` or ``None``. 
371          """
372  
373          try:
374              rdataset = self.find_rdataset(name, rdtype, covers, create)
375          except KeyError:
376              rdataset = None
377          return rdataset

379      def delete_rdataset(self, name, rdtype, covers=dns.rdatatype.NONE):
380          """Delete the rdataset matching *rdtype* and *covers*, if it 
381          exists at the node specified by *name*. 
382   
383          It is not an error if the node does not exist, or if there is no 
384          matching rdataset at the node. 
385   
386          If the node has no rdatasets after the deletion, it will itself 
387          be deleted. 
388   
389          *name*: the name of the node to find. 
390          The value may be a ``dns.name.Name`` or a ``str``.  If absolute, the 
391          name must be a subdomain of the zone's origin.  If ``zone.relativize`` 
392          is ``True``, then the name will be relativized. 
393   
394          *rdtype*, an ``int`` or ``str``, the rdata type desired. 
395   
396          *covers*, an ``int`` or ``str`` or ``None``, the covered type. 
397          Usually this value is ``dns.rdatatype.NONE``, but if the 
398          rdtype is ``dns.rdatatype.SIG`` or ``dns.rdatatype.RRSIG``, 
399          then the covers value will be the rdata type the SIG/RRSIG 
400          covers.  The library treats the SIG and RRSIG types as if they 
401          were a family of types, e.g. RRSIG(A), RRSIG(NS), RRSIG(SOA). 
402          This makes RRSIGs much easier to work with than if RRSIGs 
403          covering different rdata types were aggregated into a single 
404          RRSIG rdataset. 
405          """
406  
407          name = self._validate_name(name)
408          rdtype = dns.rdatatype.RdataType.make(rdtype)
409          if covers is not None:
410              covers = dns.rdatatype.RdataType.make(covers)
411          node = self.get_node(name)
412          if node is not None:
413              node.delete_rdataset(self.rdclass, rdtype, covers)
414              if len(node) == 0:
415                  self.delete_node(name)
416  
417      def replace_rdataset(self, name, replacement):
418          """Replace an rdataset at name. 
419   
420          It is not an error if there is no rdataset matching I{replacement}. 
421   
422          Ownership of the *replacement* object is transferred to the zone; 
423          in other words, this method does not store a copy of *replacement* 
424          at the node, it stores *replacement* itself. 
425   
426          If the node does not exist, it is created. 
427   
428          *name*: the name of the node to find. 
429          The value may be a ``dns.name.Name`` or a ``str``.  If absolute, the 
430          name must be a subdomain of the zone's origin.  If ``zone.relativize`` 
431          is ``True``, then the name will be relativized. 
432   
433          *replacement*, a ``dns.rdataset.Rdataset``, the replacement rdataset. 
434          """
435  
436          if replacement.rdclass != self.rdclass:
437              raise ValueError('replacement.rdclass != zone.rdclass')
438          node = self.find_node(name, True)
439          node.replace_rdataset(replacement)

441      def find_rrset(self, name, rdtype, covers=dns.rdatatype.NONE):
442          """Look for an rdataset with the specified name and type in the zone, 
443          and return an RRset encapsulating it. 
444   
445          This method is less efficient than the similar 
446          ``find_rdataset()`` because it creates an RRset instead of 
447          returning the matching rdataset.  It may be more convenient 
448          for some uses since it returns an object which binds the owner 
449          name to the rdataset. 
450   
451          This method may not be used to create new nodes or rdatasets; 
452          use ``find_rdataset`` instead. 
453   
454          *name*: the name of the node to find. 
455          The value may be a ``dns.name.Name`` or a ``str``.  If absolute, the 
456          name must be a subdomain of the zone's origin.  If ``zone.relativize`` 
457          is ``True``, then the name will be relativized. 
458   
459          *rdtype*, an ``int`` or ``str``, the rdata type desired. 
460   
461          *covers*, an ``int`` or ``str`` or ``None``, the covered type. 
462          Usually this value is ``dns.rdatatype.NONE``, but if the 
463          rdtype is ``dns.rdatatype.SIG`` or ``dns.rdatatype.RRSIG``, 
464          then the covers value will be the rdata type the SIG/RRSIG 
465          covers.  The library treats the SIG and RRSIG types as if they 
466          were a family of types, e.g. RRSIG(A), RRSIG(NS), RRSIG(SOA). 
467          This makes RRSIGs much easier to work with than if RRSIGs 
468          covering different rdata types were aggregated into a single 
469          RRSIG rdataset. 
470   
471          *create*, a ``bool``.  If true, the node will be created if it does 
472          not exist. 
473   
474          Raises ``KeyError`` if the name is not known and create was 
475          not specified, or if the name was not a subdomain of the origin. 
476   
477          Returns a ``dns.rrset.RRset`` or ``None``. 
478          """
479  
480          name = self._validate_name(name)
481          rdtype = dns.rdatatype.RdataType.make(rdtype)
482          if covers is not None:
483              covers = dns.rdatatype.RdataType.make(covers)
484          rdataset = self.nodes[name].find_rdataset(self.rdclass, rdtype, covers)
485          rrset = dns.rrset.RRset(name, self.rdclass, rdtype, covers)
486          rrset.update(rdataset)
487          return rrset
488

533      def iterate_rdatasets(self, rdtype=dns.rdatatype.ANY,
534                            covers=dns.rdatatype.NONE):
535          """Return a generator which yields (name, rdataset) tuples for 
536          all rdatasets in the zone which have the specified *rdtype* 
537          and *covers*.  If *rdtype* is ``dns.rdatatype.ANY``, the default, 
538          then all rdatasets will be matched. 
539   
540          *rdtype*, an ``int`` or ``str``, the rdata type desired. 
541   
542          *covers*, an ``int`` or ``str`` or ``None``, the covered type. 
543          Usually this value is ``dns.rdatatype.NONE``, but if the 
544          rdtype is ``dns.rdatatype.SIG`` or ``dns.rdatatype.RRSIG``, 
545          then the covers value will be the rdata type the SIG/RRSIG 
546          covers.  The library treats the SIG and RRSIG types as if they 
547          were a family of types, e.g. RRSIG(A), RRSIG(NS), RRSIG(SOA). 
548          This makes RRSIGs much easier to work with than if RRSIGs 
549          covering different rdata types were aggregated into a single 
550          RRSIG rdataset. 
551          """
552  
553          rdtype = dns.rdatatype.RdataType.make(rdtype)
554          if covers is not None:
555              covers = dns.rdatatype.RdataType.make(covers)
556          for (name, node) in self.items():
557              for rds in node:
558                  if rdtype == dns.rdatatype.ANY or \
559                     (rds.rdtype == rdtype and rds.covers == covers):
560                      yield (name, rds)
561  
562      def iterate_rdatas(self, rdtype=dns.rdatatype.ANY,
563                         covers=dns.rdatatype.NONE):
564          """Return a generator which yields (name, ttl, rdata) tuples for 
565          all rdatas in the zone which have the specified *rdtype* 
566          and *covers*.  If *rdtype* is ``dns.rdatatype.ANY``, the default, 
567          then all rdatas will be matched. 
568   
569          *rdtype*, an ``int`` or ``str``, the rdata type desired. 
570   
571          *covers*, an ``int`` or ``str`` or ``None``, the covered type. 
572          Usually this value is ``dns.rdatatype.NONE``, but if the 
573          rdtype is ``dns.rdatatype.SIG`` or ``dns.rdatatype.RRSIG``, 
574          then the covers value will be the rdata type the SIG/RRSIG 
575          covers.  The library treats the SIG and RRSIG types as if they 
576          were a family of types, e.g. RRSIG(A), RRSIG(NS), RRSIG(SOA). 
577          This makes RRSIGs much easier to work with than if RRSIGs 
578          covering different rdata types were aggregated into a single 
579          RRSIG rdataset. 
580          """
581  
582          rdtype = dns.rdatatype.RdataType.make(rdtype)
583          if covers is not None:
584              covers = dns.rdatatype.RdataType.make(covers)
585          for (name, node) in self.items():
586              for rds in node:
587                  if rdtype == dns.rdatatype.ANY or \
588                     (rds.rdtype == rdtype and rds.covers == covers):
589                      for rdata in rds:
590                          yield (name, rds.ttl, rdata)

592      def to_file(self, f, sorted=True, relativize=True, nl=None,
593                  want_comments=False, want_origin=False):
594          """Write a zone to a file. 
595   
596          *f*, a file or `str`.  If *f* is a string, it is treated 
597          as the name of a file to open. 
598   
599          *sorted*, a ``bool``.  If True, the default, then the file 
600          will be written with the names sorted in DNSSEC order from 
601          least to greatest.  Otherwise the names will be written in 
602          whatever order they happen to have in the zone's dictionary. 
603   
604          *relativize*, a ``bool``.  If True, the default, then domain 
605          names in the output will be relativized to the zone's origin 
606          if possible. 
607   
608          *nl*, a ``str`` or None.  The end of line string.  If not 
609          ``None``, the output will use the platform's native 
610          end-of-line marker (i.e. LF on POSIX, CRLF on Windows). 
611   
612          *want_comments*, a ``bool``.  If ``True``, emit end-of-line comments 
613          as part of writing the file.  If ``False``, the default, do not 
614          emit them. 
615   
616          *want_origin*, a ``bool``.  If ``True``, emit a $ORIGIN line at 
617          the start of the file.  If ``False``, the default, do not emit 
618          one. 
619          """
620  
621          with contextlib.ExitStack() as stack:
622              if isinstance(f, str):
623                  f = stack.enter_context(open(f, 'wb'))
624  
625              # must be in this way, f.encoding may contain None, or even
626              # attribute may not be there
627              file_enc = getattr(f, 'encoding', None)
628              if file_enc is None:
629                  file_enc = 'utf-8'
630  
631              if nl is None:
632                  # binary mode, '\n' is not enough
633                  nl_b = os.linesep.encode(file_enc)
634                  nl = '\n'
635              elif isinstance(nl, str):
636                  nl_b = nl.encode(file_enc)
637              else:
638                  nl_b = nl
639                  nl = nl.decode()
640  
641              if want_origin:
642                  l = '$ORIGIN ' + self.origin.to_text()
643                  l_b = l.encode(file_enc)
644                  try:
645                      f.write(l_b)
646                      f.write(nl_b)
647                  except TypeError:  # textual mode
648                      f.write(l)
649                      f.write(nl)
650  
651              if sorted:
652                  names = list(self.keys())
653                  names.sort()
654              else:
655                  names = self.keys()
656              for n in names:
657                  l = self[n].to_text(n, origin=self.origin,
658                                      relativize=relativize,
659                                      want_comments=want_comments)
660                  l_b = l.encode(file_enc)
661  
662                  try:
663                      f.write(l_b)
664                      f.write(nl_b)
665                  except TypeError:  # textual mode
666                      f.write(l)
667                      f.write(nl)
668  
669      def to_text(self, sorted=True, relativize=True, nl=None,
670                  want_comments=False, want_origin=False):
671          """Return a zone's text as though it were written to a file. 
672   
673          *sorted*, a ``bool``.  If True, the default, then the file 
674          will be written with the names sorted in DNSSEC order from 
675          least to greatest.  Otherwise the names will be written in 
676          whatever order they happen to have in the zone's dictionary. 
677   
678          *relativize*, a ``bool``.  If True, the default, then domain 
679          names in the output will be relativized to the zone's origin 
680          if possible. 
681   
682          *nl*, a ``str`` or None.  The end of line string.  If not 
683          ``None``, the output will use the platform's native 
684          end-of-line marker (i.e. LF on POSIX, CRLF on Windows). 
685   
686          *want_comments*, a ``bool``.  If ``True``, emit end-of-line comments 
687          as part of writing the file.  If ``False``, the default, do not 
688          emit them. 
689   
690          *want_origin*, a ``bool``.  If ``True``, emit a $ORIGIN line at 
691          the start of the output.  If ``False``, the default, do not emit 
692          one. 
693   
694          Returns a ``str``. 
695          """
696          temp_buffer = io.StringIO()
697          self.to_file(temp_buffer, sorted, relativize, nl, want_comments,
698                       want_origin)
699          return_value = temp_buffer.getvalue()
700          temp_buffer.close()
701          return return_value

703      def check_origin(self):
704          """Do some simple checking of the zone's origin. 
705   
706          Raises ``dns.zone.NoSOA`` if there is no SOA RRset. 
707   
708          Raises ``dns.zone.NoNS`` if there is no NS RRset. 
709   
710          Raises ``KeyError`` if there is no origin node. 
711          """
712          if self.relativize:
713              name = dns.name.empty
714          else:
715              name = self.origin
716          if self.get_rdataset(name, dns.rdatatype.SOA) is None:
717              raise NoSOA
718          if self.get_rdataset(name, dns.rdatatype.NS) is None:
719              raise NoNS
720  
721      def _compute_digest(self, hash_algorithm, scheme=DigestScheme.SIMPLE):
722          hashinfo = _digest_hashers.get(hash_algorithm)
723          if not hashinfo:
724              raise UnsupportedDigestHashAlgorithm
725          if scheme != DigestScheme.SIMPLE:
726              raise UnsupportedDigestScheme
727  
728          if self.relativize:
729              origin_name = dns.name.empty
730          else:
731              origin_name = self.origin
732          hasher = hashinfo()
733          for (name, node) in sorted(self.items()):
734              rrnamebuf = name.to_digestable(self.origin)
735              for rdataset in sorted(node,
736                                     key=lambda rds: (rds.rdtype, rds.covers)):
737                  if name == origin_name and \
738                     dns.rdatatype.ZONEMD in (rdataset.rdtype, rdataset.covers):
739                      continue
740                  rrfixed = struct.pack('!HHI', rdataset.rdtype,
741                                        rdataset.rdclass, rdataset.ttl)
742                  rdatas = [rdata.to_digestable(self.origin)
743                            for rdata in rdataset]
744                  for rdata in sorted(rdatas):
745                      rrlen = struct.pack('!H', len(rdata))
746                      hasher.update(rrnamebuf + rrfixed + rrlen + rdata)
747          return hasher.digest()
748  
749      def compute_digest(self, hash_algorithm, scheme=DigestScheme.SIMPLE):
750          if self.relativize:
751              origin_name = dns.name.empty
752          else:
753              origin_name = self.origin
754          serial = self.get_rdataset(origin_name, dns.rdatatype.SOA)[0].serial
755          digest = self._compute_digest(hash_algorithm, scheme)
756          return dns.rdtypes.ANY.ZONEMD.ZONEMD(self.rdclass,
757                                               dns.rdatatype.ZONEMD,
758                                               serial, scheme, hash_algorithm,
759                                               digest)
760  
761      def verify_digest(self, zonemd=None):
762          if zonemd:
763              digests = [zonemd]
764          else:
765              digests = self.get_rdataset(self.origin, dns.rdatatype.ZONEMD)
766              if digests is None:
767                  raise NoDigest
768          for digest in digests:
769              try:
770                  computed = self._compute_digest(digest.hash_algorithm,
771                                                  digest.scheme)
772                  if computed == digest.digest:
773                      return
774              except Exception:
775                  pass
776          raise DigestVerificationFailure

866  class Version:
867      def __init__(self, zone, id, nodes=None, origin=None):
868          self.zone = zone
869          self.id = id
870          if nodes is not None:
871              self.nodes = nodes
872          else:
873              self.nodes = {}
874          self.origin = origin
875  
876      def _validate_name(self, name):
877          if name.is_absolute():
878              if self.origin is None:
879                  # This should probably never happen as other code (e.g.
880                  # _rr_line) will notice the lack of an origin before us, but
881                  # we check just in case!
882                  raise KeyError('no zone origin is defined')
883              if not name.is_subdomain(self.origin):
884                  raise KeyError("name is not a subdomain of the zone origin")
885              if self.zone.relativize:
886                  name = name.relativize(self.origin)
887          elif not self.zone.relativize:
888              # We have a relative name in a non-relative zone, so derelativize.
889              if self.origin is None:
890                  raise KeyError('no zone origin is defined')
891              name = name.derelativize(self.origin)
892          return name
893  
894      def get_node(self, name):
895          name = self._validate_name(name)
896          return self.nodes.get(name)
897  
898      def get_rdataset(self, name, rdtype, covers):
899          node = self.get_node(name)
900          if node is None:
901              return None
902          return node.get_rdataset(self.zone.rdclass, rdtype, covers)
903  
904      def items(self):
905          return self.nodes.items()

908  class WritableVersion(Version):
909      def __init__(self, zone, replacement=False):
910          # The zone._versions_lock must be held by our caller in a versioned
911          # zone.
912          id = zone._get_next_version_id()
913          super().__init__(zone, id)
914          if not replacement:
915              # We copy the map, because that gives us a simple and thread-safe
916              # way of doing versions, and we have a garbage collector to help
917              # us.  We only make new node objects if we actually change the
918              # node.
919              self.nodes.update(zone.nodes)
920          # We have to copy the zone origin as it may be None in the first
921          # version, and we don't want to mutate the zone until we commit.
922          self.origin = zone.origin
923          self.changed = set()
924  
925      def _maybe_cow(self, name):
926          name = self._validate_name(name)
927          node = self.nodes.get(name)
928          if node is None or name not in self.changed:
929              new_node = self.zone.node_factory()
930              if hasattr(new_node, 'id'):
931                  # We keep doing this for backwards compatibility, as earlier
932                  # code used new_node.id != self.id for the "do we need to CoW?"
933                  # test.  Now we use the changed set as this works with both
934                  # regular zones and versioned zones.
935                  new_node.id = self.id
936              if node is not None:
937                  # moo!  copy on write!
938                  new_node.rdatasets.extend(node.rdatasets)
939              self.nodes[name] = new_node
940              self.changed.add(name)
941              return new_node
942          else:
943              return node
944  
945      def delete_node(self, name):
946          name = self._validate_name(name)
947          if name in self.nodes:
948              del self.nodes[name]
949              self.changed.add(name)
950  
951      def put_rdataset(self, name, rdataset):
952          node = self._maybe_cow(name)
953          node.replace_rdataset(rdataset)
954  
955      def delete_rdataset(self, name, rdtype, covers):
956          node = self._maybe_cow(name)
957          node.delete_rdataset(self.zone.rdclass, rdtype, covers)
958          if len(node) == 0:
959              del self.nodes[name]

962  @dns.immutable.immutable
963  class ImmutableVersion(Version):
964      def __init__(self, version):
965          # We tell super() that it's a replacement as we don't want it
966          # to copy the nodes, as we're about to do that with an
967          # immutable Dict.
968          super().__init__(version.zone, True)
969          # set the right id!
970          self.id = version.id
971          # keep the origin
972          self.origin = version.origin
973          # Make changed nodes immutable
974          for name in version.changed:
975              node = version.nodes.get(name)
976              # it might not exist if we deleted it in the version
977              if node:
978                  version.nodes[name] = ImmutableVersionedNode(node)
979          self.nodes = dns.immutable.Dict(version.nodes, True)

982  class Transaction(dns.transaction.Transaction):
983  
984      def __init__(self, zone, replacement, version=None, make_immutable=False):
985          read_only = version is not None
986          super().__init__(zone, replacement, read_only)
987          self.version = version
988          self.make_immutable = make_immutable
989  
990      @property
991      def zone(self):
992          return self.manager
993  
994      def _setup_version(self):
995          assert self.version is None
996          self.version = WritableVersion(self.zone, self.replacement)
997  
998      def _get_rdataset(self, name, rdtype, covers):
999          return self.version.get_rdataset(name, rdtype, covers)
1000 
1001     def _put_rdataset(self, name, rdataset):
1002         assert not self.read_only
1003         self.version.put_rdataset(name, rdataset)
1004 
1005     def _delete_name(self, name):
1006         assert not self.read_only
1007         self.version.delete_node(name)
1008 
1009     def _delete_rdataset(self, name, rdtype, covers):
1010         assert not self.read_only
1011         self.version.delete_rdataset(name, rdtype, covers)
1012 
1013     def _name_exists(self, name):
1014         return self.version.get_node(name) is not None
1015 
1016     def _changed(self):
1017         if self.read_only:
1018             return False
1019         else:
1020             return len(self.version.changed) > 0
1021 
1022     def _end_transaction(self, commit):
1023         if self.read_only:
1024             self.zone._end_read(self)
1025         elif commit and len(self.version.changed) > 0:
1026             if self.make_immutable:
1027                 version = ImmutableVersion(self.version)
1028             else:
1029                 version = self.version
1030             self.zone._commit_version(self, version, self.version.origin)
1031         else:
1032             # rollback
1033             self.zone._end_write(self)
1034 
1035     def _set_origin(self, origin):
1036         if self.version.origin is None:
1037             self.version.origin = origin
1038 
1039     def _iterate_rdatasets(self):
1040         for (name, node) in self.version.items():
1041             for rdataset in node:
1042                 yield (name, rdataset)
1043 
1044     def _get_node(self, name):
1045         return self.version.get_node(name)
1046 
1047     def _origin_information(self):
1048         (absolute, relativize, effective) = self.manager.origin_information()
1049         if absolute is None and self.version.origin is not None:
1050             # No origin has been committed yet, but we've learned one as part of
1051             # this txn.  Use it.
1052             absolute = self.version.origin
1053             if relativize:
1054                 effective = dns.name.empty
1055             else:
1056                 effective = absolute
1057         return (absolute, relativize, effective)

1060 def from_text(text, origin=None, rdclass=dns.rdataclass.IN,
1061               relativize=True, zone_factory=Zone, filename=None,
1062               allow_include=False, check_origin=True, idna_codec=None):
1063     """Build a zone object from a zone file format string. 
1064  
1065     *text*, a ``str``, the zone file format input. 
1066  
1067     *origin*, a ``dns.name.Name``, a ``str``, or ``None``.  The origin 
1068     of the zone; if not specified, the first ``$ORIGIN`` statement in the 
1069     zone file will determine the origin of the zone. 
1070  
1071     *rdclass*, an ``int``, the zone's rdata class; the default is class IN. 
1072  
1073     *relativize*, a ``bool``, determine's whether domain names are 
1074     relativized to the zone's origin.  The default is ``True``. 
1075  
1076     *zone_factory*, the zone factory to use or ``None``.  If ``None``, then 
1077     ``dns.zone.Zone`` will be used.  The value may be any class or callable 
1078     that returns a subclass of ``dns.zone.Zone``. 
1079  
1080     *filename*, a ``str`` or ``None``, the filename to emit when 
1081     describing where an error occurred; the default is ``'<string>'``. 
1082  
1083     *allow_include*, a ``bool``.  If ``True``, the default, then ``$INCLUDE`` 
1084     directives are permitted.  If ``False``, then encoutering a ``$INCLUDE`` 
1085     will raise a ``SyntaxError`` exception. 
1086  
1087     *check_origin*, a ``bool``.  If ``True``, the default, then sanity 
1088     checks of the origin node will be made by calling the zone's 
1089     ``check_origin()`` method. 
1090

1098  
1099     Raises ``KeyError`` if there is no origin node. 
1100  
1101     Returns a subclass of ``dns.zone.Zone``. 
1102     """
1103

1105     # since it's an implementation detail.  The official file
1106     # interface is from_file().
1107

1109         filename = '<string>'
1110     zone = zone_factory(origin, rdclass, relativize=relativize)
1111     with zone.writer(True) as txn:
1112         tok = dns.tokenizer.Tokenizer(text, filename, idna_codec=idna_codec)
1113         reader = dns.zonefile.Reader(tok, rdclass, txn,
1114                                      allow_include=allow_include)
1115         try:
1116             reader.read()
1117         except dns.zonefile.UnknownOrigin:
1118             # for backwards compatibility
1119             raise dns.zone.UnknownOrigin
1120     # Now that we're done reading, do some basic checking of the zone.
1121     if check_origin:
1122         zone.check_origin()
1123     return zone

1180 def from_xfr(xfr, zone_factory=Zone, relativize=True, check_origin=True):
1181     """Convert the output of a zone transfer generator into a zone object. 
1182  
1183     *xfr*, a generator of ``dns.message.Message`` objects, typically 
1184     ``dns.query.xfr()``. 
1185  
1186     *relativize*, a ``bool``, determine's whether domain names are 
1187     relativized to the zone's origin.  The default is ``True``. 
1188     It is essential that the relativize setting matches the one specified 
1189     to the generator. 
1190  
1191     *check_origin*, a ``bool``.  If ``True``, the default, then sanity 
1192     checks of the origin node will be made by calling the zone's 
1193     ``check_origin()`` method. 
1194  
1195     Raises ``dns.zone.NoSOA`` if there is no SOA RRset. 
1196  
1197     Raises ``dns.zone.NoNS`` if there is no NS RRset. 
1198  
1199     Raises ``KeyError`` if there is no origin node. 
1200  
1201     Returns a subclass of ``dns.zone.Zone``. 
1202     """
1203 
1204     z = None
1205     for r in xfr:
1206         if z is None:
1207             if relativize:
1208                 origin = r.origin
1209             else:
1210                 origin = r.answer[0].name
1211             rdclass = r.answer[0].rdclass
1212             z = zone_factory(origin, rdclass, relativize=relativize)
1213         for rrset in r.answer:
1214             znode = z.nodes.get(rrset.name)
1215             if not znode:
1216                 znode = z.node_factory()
1217                 z.nodes[rrset.name] = znode
1218             zrds = znode.find_rdataset(rrset.rdclass, rrset.rdtype,
1219                                        rrset.covers, True)
1220             zrds.update_ttl(rrset.ttl)
1221             for rd in rrset:
1222                 zrds.add(rd)
1223     if check_origin:
1224         z.check_origin()
1225     return z
