1    """ 
2    Tagged JSON 
3    ~~~~~~~~~~~ 
4     
5    A compact representation for lossless serialization of non-standard JSON 
6    types. :class:`~flask.sessions.SecureCookieSessionInterface` uses this 
7    to serialize the session data, but it may be useful in other places. It 
8    can be extended to support other types. 
9     
10   .. autoclass:: TaggedJSONSerializer 
11       :members: 
12    
13   .. autoclass:: JSONTag 
14       :members: 
15    
16   Let's see an example that adds support for 
17   :class:`~collections.OrderedDict`. Dicts don't have an order in JSON, so 
18   to handle this we will dump the items as a list of ``[key, value]`` 
19   pairs. Subclass :class:`JSONTag` and give it the new key ``' od'`` to 
20   identify the type. The session serializer processes dicts first, so 
21   insert the new tag at the front of the order since ``OrderedDict`` must 
22   be processed before ``dict``. 
23    
24   .. code-block:: python 
25    
26       from flask.json.tag import JSONTag 
27    
28       class TagOrderedDict(JSONTag): 
29           __slots__ = ('serializer',) 
30           key = ' od' 
31    
32           def check(self, value): 
33               return isinstance(value, OrderedDict) 
34    
35           def to_json(self, value): 
36               return [[k, self.serializer.tag(v)] for k, v in iteritems(value)] 
37    
38           def to_python(self, value): 
39               return OrderedDict(value) 
40    
41       app.session_interface.serializer.register(TagOrderedDict, index=0) 
42   """
43   import typing as t
44   from base64 import b64decode

50   from werkzeug.http import http_date
51   from werkzeug.http import parse_date

57   class JSONTag:
58       """Base class for defining type tags for :class:`TaggedJSONSerializer`."""
59   
60       __slots__ = ("serializer",)
61   
62       #: The tag to mark the serialized object with. If ``None``, this tag is
63       #: only used as an intermediate step during tagging.
64       key: t.Optional[str] = None
65   
66       def __init__(self, serializer: "TaggedJSONSerializer") -> None:
67           """Create a tagger for the given serializer."""
68           self.serializer = serializer
69   
70       def check(self, value: t.Any) -> bool:
71           """Check if the given value should be tagged by this tag."""
72           raise NotImplementedError
73   
74       def to_json(self, value: t.Any) -> t.Any:
75           """Convert the Python object to an object that is a valid JSON type. 
76           The tag will be added later."""
77           raise NotImplementedError
78   
79       def to_python(self, value: t.Any) -> t.Any:
80           """Convert the JSON representation back to the correct type. The tag 
81           will already be removed."""
82           raise NotImplementedError
83   
84       def tag(self, value: t.Any) -> t.Any:
85           """Convert the value to a valid JSON type and add the tag structure 
86           around it."""
87           return {self.key: self.to_json(value)}

90   class TagDict(JSONTag):
91       """Tag for 1-item dicts whose only key matches a registered tag. 
92    
93       Internally, the dict key is suffixed with `__`, and the suffix is removed 
94       when deserializing. 
95       """
96   
97       __slots__ = ()
98       key = " di"
99   
100      def check(self, value: t.Any) -> bool:
101          return (
102              isinstance(value, dict)
103              and len(value) == 1
104              and next(iter(value)) in self.serializer.tags
105          )
106  
107      def to_json(self, value: t.Any) -> t.Any:
108          key = next(iter(value))
109          return {f"{key}__": self.serializer.tag(value[key])}
110  
111      def to_python(self, value: t.Any) -> t.Any:
112          key = next(iter(value))
113          return {key[:-2]: value[key]}
114

116  class PassDict(JSONTag):
117      __slots__ = ()
118  
119      def check(self, value: t.Any) -> bool:
120          return isinstance(value, dict)
121  
122      def to_json(self, value: t.Any) -> t.Any:
123          # JSON objects may only have string keys, so don't bother tagging the
124          # key here.
125          return {k: self.serializer.tag(v) for k, v in value.items()}
126  
127      tag = to_json

170  class TagMarkup(JSONTag):
171      """Serialize anything matching the :class:`~markupsafe.Markup` API by 
172      having a ``__html__`` method to the result of that method. Always 
173      deserializes to an instance of :class:`~markupsafe.Markup`."""
174  
175      __slots__ = ()
176      key = " m"
177  
178      def check(self, value: t.Any) -> bool:
179          return callable(getattr(value, "__html__", None))
180  
181      def to_json(self, value: t.Any) -> t.Any:
182          return str(value.__html__())
183  
184      def to_python(self, value: t.Any) -> t.Any:
185          return Markup(value)

216  class TaggedJSONSerializer:
217      """Serializer that uses a tag system to compactly represent objects that 
218      are not JSON types. Passed as the intermediate serializer to 
219      :class:`itsdangerous.Serializer`. 
220   
221      The following extra types are supported: 
222   
223      * :class:`dict` 
224      * :class:`tuple` 
225      * :class:`bytes` 
226      * :class:`~markupsafe.Markup` 
227      * :class:`~uuid.UUID` 
228      * :class:`~datetime.datetime` 
229      """
230  
231      __slots__ = ("tags", "order")
232  
233      #: Tag classes to bind when creating the serializer. Other tags can be
234      #: added later using :meth:`~register`.
235      default_tags = [
236          TagDict,
237          PassDict,
238          TagTuple,
239          PassList,
240          TagBytes,
241          TagMarkup,
242          TagUUID,
243          TagDateTime,
244      ]
245  
246      def __init__(self) -> None:
247          self.tags: t.Dict[str, JSONTag] = {}
248          self.order: t.List[JSONTag] = []
249  
250          for cls in self.default_tags:
251              self.register(cls)
252  
253      def register(
254          self,
255          tag_class: t.Type[JSONTag],
256          force: bool = False,
257          index: t.Optional[int] = None,
258      ) -> None:
259          """Register a new tag with this serializer. 
260   
261          :param tag_class: tag class to register. Will be instantiated with this 
262              serializer instance. 
263          :param force: overwrite an existing tag. If false (default), a 
264              :exc:`KeyError` is raised. 
265          :param index: index to insert the new tag in the tag order. Useful when 
266              the new tag is a special case of an existing tag. If ``None`` 
267              (default), the tag is appended to the end of the order. 
268   
269          :raise KeyError: if the tag key is already registered and ``force`` is 
270              not true. 
271          """
272          tag = tag_class(self)
273          key = tag.key
274  
275          if key is not None:
276              if not force and key in self.tags:
277                  raise KeyError(f"Tag '{key}' is already registered.")
278  
279              self.tags[key] = tag
280  
281          if index is None:
282              self.order.append(tag)
283          else:
284              self.order.insert(index, tag)
285  
286      def tag(self, value: t.Any) -> t.Dict[str, t.Any]:
287          """Convert a value to a tagged representation if necessary."""
288          for tag in self.order:
289              if tag.check(value):
290                  return tag.tag(value)
291  
292          return value
293  
294      def untag(self, value: t.Dict[str, t.Any]) -> t.Any:
295          """Convert a tagged representation back to the original type."""
296          if len(value) != 1:
297              return value
298  
299          key = next(iter(value))
300  
301          if key not in self.tags:
302              return value
303  
304          return self.tags[key].to_python(value[key])
305  
306      def dumps(self, value: t.Any) -> str:
307          """Tag the value and dump it to a compact JSON string."""
308          return dumps(self.tag(value), separators=(",", ":"))
309  
310      def loads(self, value: str) -> t.Any:
311          """Load data from a JSON string and deserialized any tagged objects."""
312          return loads(value, object_hook=self.untag)
