29   _cname_types = {
30       dns.rdatatype.CNAME,
31   }
32   
33   # "neutral" types can coexist with a CNAME and thus are not "other data"
34   _neutral_types = {
35       dns.rdatatype.NSEC,   # RFC 4035 section 2.5
36       dns.rdatatype.NSEC3,  # This is not likely to happen, but not impossible!
37       dns.rdatatype.KEY,    # RFC 4035 section 2.5, RFC 3007
38   }
39   
40   def _matches_type_or_its_signature(rdtypes, rdtype, covers):
41       return rdtype in rdtypes or \
42              (rdtype == dns.rdatatype.RRSIG and covers in rdtypes)
43   
44   
45   @enum.unique
46   class NodeKind(enum.Enum):
47       """Rdatasets in nodes 
48       """
49       REGULAR = 0      # a.k.a "other data"
50       NEUTRAL = 1
51       CNAME = 2
52   
53       @classmethod
54       def classify(cls, rdtype, covers):
55           if _matches_type_or_its_signature(_cname_types, rdtype, covers):
56               return NodeKind.CNAME
57           elif _matches_type_or_its_signature(_neutral_types, rdtype, covers):
58               return NodeKind.NEUTRAL
59           else:
60               return NodeKind.REGULAR
61   
62       @classmethod
63       def classify_rdataset(cls, rdataset):
64           return cls.classify(rdataset.rdtype, rdataset.covers)

66   
67   class Node:
68   
69       """A Node is a set of rdatasets. 
70    
71       A node is either a CNAME node or an "other data" node.  A CNAME 
72       node contains only CNAME, KEY, NSEC, and NSEC3 rdatasets along with their 
73       covering RRSIG rdatasets.  An "other data" node contains any 
74       rdataset other than a CNAME or RRSIG(CNAME) rdataset.  When 
75       changes are made to a node, the CNAME or "other data" state is 
76       always consistent with the update, i.e. the most recent change 
77       wins.  For example, if you have a node which contains a CNAME 
78       rdataset, and then add an MX rdataset to it, then the CNAME 
79       rdataset will be deleted.  Likewise if you have a node containing 
80       an MX rdataset and add a CNAME rdataset, the MX rdataset will be 
81       deleted. 
82       """
83   
84       __slots__ = ['rdatasets']
85   
86       def __init__(self):
87           # the set of rdatasets, represented as a list.
88           self.rdatasets = []
89   
90       def to_text(self, name, **kw):
91           """Convert a node to text format. 
92    
93           Each rdataset at the node is printed.  Any keyword arguments 
94           to this method are passed on to the rdataset's to_text() method. 
95    
96           *name*, a ``dns.name.Name`` or ``str``, the owner name of the 
97           rdatasets. 
98    
99           Returns a ``str``. 
100   
101          """
102  
103          s = io.StringIO()
104          for rds in self.rdatasets:
105              if len(rds) > 0:
106                  s.write(rds.to_text(name, **kw))
107                  s.write('\n')
108          return s.getvalue()[:-1]
109  
110      def __repr__(self):
111          return '<DNS node ' + str(id(self)) + '>'
112  
113      def __eq__(self, other):
114          #
115          # This is inefficient.  Good thing we don't need to do it much.
116          #
117          for rd in self.rdatasets:
118              if rd not in other.rdatasets:
119                  return False
120          for rd in other.rdatasets:
121              if rd not in self.rdatasets:
122                  return False
123          return True
124  
125      def __ne__(self, other):
126          return not self.__eq__(other)
127  
128      def __len__(self):
129          return len(self.rdatasets)
130  
131      def __iter__(self):
132          return iter(self.rdatasets)
133  
134      def _append_rdataset(self, rdataset):
135          """Append rdataset to the node with special handling for CNAME and 
136          other data conditions. 
137   
138          Specifically, if the rdataset being appended has ``NodeKind.CNAME``, 
139          then all rdatasets other than KEY, NSEC, NSEC3, and their covering 
140          RRSIGs are deleted.  If the rdataset being appended has 
141          ``NodeKind.REGULAR`` then CNAME and RRSIG(CNAME) are deleted. 
142          """
143          # Make having just one rdataset at the node fast.
144          if len(self.rdatasets) > 0:
145              kind = NodeKind.classify_rdataset(rdataset)
146              if kind == NodeKind.CNAME:
147                  self.rdatasets = [rds for rds in self.rdatasets if
148                                    NodeKind.classify_rdataset(rds) !=
149                                    NodeKind.REGULAR]
150              elif kind == NodeKind.REGULAR:
151                  self.rdatasets = [rds for rds in self.rdatasets if
152                                    NodeKind.classify_rdataset(rds) !=
153                                    NodeKind.CNAME]
154              # Otherwise the rdataset is NodeKind.NEUTRAL and we do not need to
155              # edit self.rdatasets.
156          self.rdatasets.append(rdataset)
157  
158      def find_rdataset(self, rdclass, rdtype, covers=dns.rdatatype.NONE,
159                        create=False):
160          """Find an rdataset matching the specified properties in the 
161          current node. 
162   
163          *rdclass*, an ``int``, the class of the rdataset. 
164   
165          *rdtype*, an ``int``, the type of the rdataset. 
166   
167          *covers*, an ``int`` or ``None``, the covered type. 
168          Usually this value is ``dns.rdatatype.NONE``, but if the 
169          rdtype is ``dns.rdatatype.SIG`` or ``dns.rdatatype.RRSIG``, 
170          then the covers value will be the rdata type the SIG/RRSIG 
171          covers.  The library treats the SIG and RRSIG types as if they 
172          were a family of types, e.g. RRSIG(A), RRSIG(NS), RRSIG(SOA). 
173          This makes RRSIGs much easier to work with than if RRSIGs 
174          covering different rdata types were aggregated into a single 
175          RRSIG rdataset. 
176   
177          *create*, a ``bool``.  If True, create the rdataset if it is not found. 
178   
179          Raises ``KeyError`` if an rdataset of the desired type and class does 
180          not exist and *create* is not ``True``. 
181   
182          Returns a ``dns.rdataset.Rdataset``. 
183          """
184  
185          for rds in self.rdatasets:
186              if rds.match(rdclass, rdtype, covers):
187                  return rds
188          if not create:
189              raise KeyError
190          rds = dns.rdataset.Rdataset(rdclass, rdtype, covers)
191          self._append_rdataset(rds)
192          return rds
193  
194      def get_rdataset(self, rdclass, rdtype, covers=dns.rdatatype.NONE,
195                       create=False):
196          """Get an rdataset matching the specified properties in the 
197          current node. 
198   
199          None is returned if an rdataset of the specified type and 
200          class does not exist and *create* is not ``True``. 
201   
202          *rdclass*, an ``int``, the class of the rdataset. 
203   
204          *rdtype*, an ``int``, the type of the rdataset. 
205   
206          *covers*, an ``int``, the covered type.  Usually this value is 
207          dns.rdatatype.NONE, but if the rdtype is dns.rdatatype.SIG or 
208          dns.rdatatype.RRSIG, then the covers value will be the rdata 
209          type the SIG/RRSIG covers.  The library treats the SIG and RRSIG 
210          types as if they were a family of 
211          types, e.g. RRSIG(A), RRSIG(NS), RRSIG(SOA).  This makes RRSIGs much 
212          easier to work with than if RRSIGs covering different rdata 
213          types were aggregated into a single RRSIG rdataset. 
214   
215          *create*, a ``bool``.  If True, create the rdataset if it is not found. 
216   
217          Returns a ``dns.rdataset.Rdataset`` or ``None``. 
218          """
219  
220          try:
221              rds = self.find_rdataset(rdclass, rdtype, covers, create)
222          except KeyError:
223              rds = None
224          return rds
225  
226      def delete_rdataset(self, rdclass, rdtype, covers=dns.rdatatype.NONE):
227          """Delete the rdataset matching the specified properties in the 
228          current node. 
229   
230          If a matching rdataset does not exist, it is not an error. 
231   
232          *rdclass*, an ``int``, the class of the rdataset. 
233   
234          *rdtype*, an ``int``, the type of the rdataset. 
235   
236          *covers*, an ``int``, the covered type. 
237          """
238  
239          rds = self.get_rdataset(rdclass, rdtype, covers)
240          if rds is not None:
241              self.rdatasets.remove(rds)
242  
243      def replace_rdataset(self, replacement):
244          """Replace an rdataset. 
245   
246          It is not an error if there is no rdataset matching *replacement*. 
247   
248          Ownership of the *replacement* object is transferred to the node; 
249          in other words, this method does not store a copy of *replacement* 
250          at the node, it stores *replacement* itself. 
251   
252          *replacement*, a ``dns.rdataset.Rdataset``. 
253   
254          Raises ``ValueError`` if *replacement* is not a 
255          ``dns.rdataset.Rdataset``. 
256          """
257  
258          if not isinstance(replacement, dns.rdataset.Rdataset):
259              raise ValueError('replacement is not an rdataset')
260          if isinstance(replacement, dns.rrset.RRset):
261              # RRsets are not good replacements as the match() method
262              # is not compatible.
263              replacement = replacement.to_rdataset()
264          self.delete_rdataset(replacement.rdclass, replacement.rdtype,
265                               replacement.covers)
266          self._append_rdataset(replacement)
267  
268      def classify(self):
269          """Classify a node. 
270   
271          A node which contains a CNAME or RRSIG(CNAME) is a 
272          ``NodeKind.CNAME`` node. 
273   
274          A node which contains only "neutral" types, i.e. types allowed to 
275          co-exist with a CNAME, is a ``NodeKind.NEUTRAL`` node.  The neutral 
276          types are NSEC, NSEC3, KEY, and their associated RRSIGS.  An empty node 
277          is also considered neutral. 
278   
279          A node which contains some rdataset which is not a CNAME, RRSIG(CNAME), 
280          or a neutral type is a a ``NodeKind.REGULAR`` node.  Regular nodes are 
281          also commonly referred to as "other data". 
282          """
283          for rdataset in self.rdatasets:
284              kind = NodeKind.classify(rdataset.rdtype, rdataset.covers)
285              if kind != NodeKind.NEUTRAL:
286                  return kind
287          return NodeKind.NEUTRAL
288  
289      def is_immutable(self):
290          return False

293  @dns.immutable.immutable
294  class ImmutableNode(Node):
295      def __init__(self, node):
296          super().__init__()
297          self.rdatasets = tuple(
298              [dns.rdataset.ImmutableRdataset(rds) for rds in node.rdatasets]
299          )
300  
301      def find_rdataset(self, rdclass, rdtype, covers=dns.rdatatype.NONE,
302                        create=False):
303          if create:
304              raise TypeError("immutable")
305          return super().find_rdataset(rdclass, rdtype, covers, False)
306  
307      def get_rdataset(self, rdclass, rdtype, covers=dns.rdatatype.NONE,
308                       create=False):
309          if create:
310              raise TypeError("immutable")
311          return super().get_rdataset(rdclass, rdtype, covers, False)
312  
313      def delete_rdataset(self, rdclass, rdtype, covers=dns.rdatatype.NONE):
314          raise TypeError("immutable")
315  
316      def replace_rdataset(self, replacement):
317          raise TypeError("immutable")
318  
319      def is_immutable(self):
320          return True
