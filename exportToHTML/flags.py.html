18   """DNS Message Flags."""
19   
20   import enum
21   
22   # Standard DNS flags
23   
24   class Flag(enum.IntFlag):
25       #: Query Response
26       QR = 0x8000
27       #: Authoritative Answer
28       AA = 0x0400
29       #: Truncated Response
30       TC = 0x0200
31       #: Recursion Desired
32       RD = 0x0100
33       #: Recursion Available
34       RA = 0x0080
35       #: Authentic Data
36       AD = 0x0020
37       #: Checking Disabled
38       CD = 0x0010
39   
40   
41   # EDNS flags
42   
43   class EDNSFlag(enum.IntFlag):
44       #: DNSSEC answer OK
45       DO = 0x8000
46   
47   
48   def _from_text(text, enum_class):
49       flags = 0
50       tokens = text.split()
51       for t in tokens:
52           flags |= enum_class[t.upper()]
53       return flags
54   
55   
56   def _to_text(flags, enum_class):
57       text_flags = []
58       for k, v in enum_class.__members__.items():
59           if flags & v != 0:
60               text_flags.append(k)
61       return ' '.join(text_flags)
62   
63   
64   def from_text(text):
65       """Convert a space-separated list of flag text values into a flags 
66       value. 
67    
68       Returns an ``int`` 
69       """
70   
71       return _from_text(text, Flag)
72   
73   
74   def to_text(flags):
75       """Convert a flags value into a space-separated list of flag text 
76       values. 
77    
78       Returns a ``str``. 
79       """
80   
81       return _to_text(flags, Flag)
82   
83   
84   def edns_from_text(text):
85       """Convert a space-separated list of EDNS flag text values into a EDNS 
86       flags value. 
87    
88       Returns an ``int`` 
89       """
90   
91       return _from_text(text, EDNSFlag)
92   
93   
94   def edns_to_text(flags):
95       """Convert an EDNS flags value into a space-separated list of EDNS flag 
96       text values. 
97    
98       Returns a ``str``. 
99       """
100  
101      return _to_text(flags, EDNSFlag)
