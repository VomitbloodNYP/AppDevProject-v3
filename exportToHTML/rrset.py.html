18   """DNS RRsets (an RRset is a named rdataset)"""
19   
20   
21   import dns.name
22   import dns.rdataset
23   import dns.rdataclass
24   import dns.renderer
25   
26   
27   class RRset(dns.rdataset.Rdataset):
28   
29       """A DNS RRset (named rdataset). 
30    
31       RRset inherits from Rdataset, and RRsets can be treated as 
32       Rdatasets in most cases.  There are, however, a few notable 
33       exceptions.  RRsets have different to_wire() and to_text() method 
34       arguments, reflecting the fact that RRsets always have an owner 
35       name. 
36       """
37   
38       __slots__ = ['name', 'deleting']
39   
40       def __init__(self, name, rdclass, rdtype, covers=dns.rdatatype.NONE,
41                    deleting=None):
42           """Create a new RRset."""
43   
44           super().__init__(rdclass, rdtype, covers)
45           self.name = name
46           self.deleting = deleting
47   
48       def _clone(self):
49           obj = super()._clone()
50           obj.name = self.name
51           obj.deleting = self.deleting
52           return obj
53   
54       def __repr__(self):
55           if self.covers == 0:
56               ctext = ''
57           else:
58               ctext = '(' + dns.rdatatype.to_text(self.covers) + ')'
59           if self.deleting is not None:
60               dtext = ' delete=' + dns.rdataclass.to_text(self.deleting)
61           else:
62               dtext = ''
63           return '<DNS ' + str(self.name) + ' ' + \
64                  dns.rdataclass.to_text(self.rdclass) + ' ' + \
65                  dns.rdatatype.to_text(self.rdtype) + ctext + dtext + \
66                  ' RRset: ' + self._rdata_repr() + '>'
67   
68       def __str__(self):
69           return self.to_text()
70   
71       def __eq__(self, other):
72           if isinstance(other, RRset):
73               if self.name != other.name:
74                   return False
75           elif not isinstance(other, dns.rdataset.Rdataset):
76               return False
77           return super().__eq__(other)
78   
79       def match(self, *args, **kwargs):
80           """Does this rrset match the specified attributes? 
81    
82           Behaves as :py:func:`full_match()` if the first argument is a 
83           ``dns.name.Name``, and as :py:func:`dns.rdataset.Rdataset.match()` 
84           otherwise. 
85    
86           (This behavior fixes a design mistake where the signature of this 
87           method became incompatible with that of its superclass.  The fix 
88           makes RRsets matchable as Rdatasets while preserving backwards 
89           compatibility.) 
90           """
91           if isinstance(args[0], dns.name.Name):
92               return self.full_match(*args, **kwargs)
93           else:
94               return super().match(*args, **kwargs)
95   
96       def full_match(self, name, rdclass, rdtype, covers,
97                       deleting=None):
98           """Returns ``True`` if this rrset matches the specified name, class, 
99           type, covers, and deletion state. 
100          """
101          if not super().match(rdclass, rdtype, covers):
102              return False
103          if self.name != name or self.deleting != deleting:
104              return False
105          return True

108  
109      def to_text(self, origin=None, relativize=True, **kw):
110          """Convert the RRset into DNS zone file format. 
111   
112          See ``dns.name.Name.choose_relativity`` for more information 
113          on how *origin* and *relativize* determine the way names 
114          are emitted. 
115   
116          Any additional keyword arguments are passed on to the rdata 
117          ``to_text()`` method. 
118   
119          *origin*, a ``dns.name.Name`` or ``None``, the origin for relative 
120          names. 
121   
122          *relativize*, a ``bool``.  If ``True``, names will be relativized 
123          to *origin*. 
124          """
125  
126          return super().to_text(self.name, origin, relativize,
127                                 self.deleting, **kw)
128  
129      def to_wire(self, file, compress=None, origin=None,
130                  **kw):
131          """Convert the RRset to wire format. 
132   
133          All keyword arguments are passed to ``dns.rdataset.to_wire()``; see 
134          that function for details. 
135   
136          Returns an ``int``, the number of records emitted. 
137          """
138  
139          return super().to_wire(self.name, file, compress, origin,
140                                 self.deleting, **kw)
141  
142      # pylint: enable=arguments-differ
143  
144      def to_rdataset(self):
145          """Convert an RRset into an Rdataset. 
146   
147          Returns a ``dns.rdataset.Rdataset``. 
148          """
149          return dns.rdataset.from_rdata_list(self.ttl, list(self))

152  def from_text_list(name, ttl, rdclass, rdtype, text_rdatas,
153                     idna_codec=None, origin=None, relativize=True,
154                     relativize_to=None):
155      """Create an RRset with the specified name, TTL, class, and type, and with 
156      the specified list of rdatas in text format. 
157   
158      *idna_codec*, a ``dns.name.IDNACodec``, specifies the IDNA 
159      encoder/decoder to use; if ``None``, the default IDNA 2003 
160      encoder/decoder is used. 
161   
162      *origin*, a ``dns.name.Name`` (or ``None``), the 
163      origin to use for relative names. 
164   
165      *relativize*, a ``bool``.  If true, name will be relativized. 
166   
167      *relativize_to*, a ``dns.name.Name`` (or ``None``), the origin to use 
168      when relativizing names.  If not set, the *origin* value will be used. 
169   
170      Returns a ``dns.rrset.RRset`` object. 
171      """
172  
173      if isinstance(name, str):
174          name = dns.name.from_text(name, None, idna_codec=idna_codec)
175      rdclass = dns.rdataclass.RdataClass.make(rdclass)
176      rdtype = dns.rdatatype.RdataType.make(rdtype)
177      r = RRset(name, rdclass, rdtype)
178      r.update_ttl(ttl)
179      for t in text_rdatas:
180          rd = dns.rdata.from_text(r.rdclass, r.rdtype, t, origin, relativize,
181                                   relativize_to, idna_codec)
182          r.add(rd)
183      return r
184  
185  
186  def from_text(name, ttl, rdclass, rdtype, *text_rdatas):
187      """Create an RRset with the specified name, TTL, class, and type and with 
188      the specified rdatas in text format. 
189   
190      Returns a ``dns.rrset.RRset`` object. 
191      """
192  
193      return from_text_list(name, ttl, rdclass, rdtype, text_rdatas)
194  
195  
196  def from_rdata_list(name, ttl, rdatas, idna_codec=None):
197      """Create an RRset with the specified name and TTL, and with 
198      the specified list of rdata objects. 
199   
200      *idna_codec*, a ``dns.name.IDNACodec``, specifies the IDNA 
201      encoder/decoder to use; if ``None``, the default IDNA 2003 
202      encoder/decoder is used. 
203   
204      Returns a ``dns.rrset.RRset`` object. 
205   
206      """
207  
208      if isinstance(name, str):
209          name = dns.name.from_text(name, None, idna_codec=idna_codec)
210  
211      if len(rdatas) == 0:
212          raise ValueError("rdata list must not be empty")
213      r = None
214      for rd in rdatas:
215          if r is None:
216              r = RRset(name, rd.rdclass, rd.rdtype)
217              r.update_ttl(ttl)
218          r.add(rd)
219      return r
220  
221  
222  def from_rdata(name, ttl, *rdatas):
223      """Create an RRset with the specified name and TTL, and with 
224      the specified rdata objects. 
225   
226      Returns a ``dns.rrset.RRset`` object. 
227      """
228  
229      return from_rdata_list(name, ttl, rdatas)
