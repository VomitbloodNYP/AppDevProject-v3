50   class NoRelativeRdataOrdering(dns.exception.DNSException):
51       """An attempt was made to do an ordered comparison of one or more 
52       rdata with relative names.  The only reliable way of sorting rdata 
53       is to use non-relativized rdata.

58   def _wordbreak(data, chunksize=_chunksize, separator=b' '):
59       """Break a binary string into chunks of chunksize characters separated by 
60       a space. 
61       """
62   
63       if not chunksize:
64           return data.decode()
65       return separator.join([data[i:i + chunksize]
66                              for i
67                              in range(0, len(data), chunksize)]).decode()

72   def _hexify(data, chunksize=_chunksize, separator=b' ', **kw):
73       """Convert a binary string into its hex encoding, broken up into chunks 
74       of chunksize characters separated by a separator. 
75       """
76   
77       return _wordbreak(binascii.hexlify(data), chunksize, separator)

80   def _base64ify(data, chunksize=_chunksize, separator=b' ', **kw):
81       """Convert a binary string into its base64 encoding, broken up into chunks 
82       of chunksize characters separated by a separator. 
83       """
84   
85       return _wordbreak(base64.b64encode(data), chunksize, separator)

89   __escaped = b'"\\'

91   def _escapify(qstring):
92       """Escape the characters in a quoted string which need it."""
93   
94       if isinstance(qstring, str):
95           qstring = qstring.encode()
96       if not isinstance(qstring, bytearray):
97           qstring = bytearray(qstring)
98   
99       text = ''
100      for c in qstring:
101          if c in __escaped:
102              text += '\\' + chr(c)
103          elif c >= 0x20 and c < 0x7F:
104              text += chr(c)
105          else:
106              text += '\\%03d' % c
107      return text

110  def _truncate_bitmap(what):
111      """Determine the index of greatest byte that isn't all zeros, and 
112      return the bitmap that contains all the bytes less than that index. 
113      """
114  
115      for i in range(len(what) - 1, -1, -1):
116          if what[i] != 0:
117              return what[0: i + 1]
118      return what[0:1]

124  @dns.immutable.immutable
125  class Rdata:
126      """Base class for all DNS rdata types."""
127  
128      __slots__ = ['rdclass', 'rdtype', 'rdcomment']
129  
130      def __init__(self, rdclass, rdtype):
131          """Initialize an rdata. 
132   
133          *rdclass*, an ``int`` is the rdataclass of the Rdata. 
134   
135          *rdtype*, an ``int`` is the rdatatype of the Rdata. 
136          """
137  
138          self.rdclass = self._as_rdataclass(rdclass)
139          self.rdtype = self._as_rdatatype(rdtype)
140          self.rdcomment = None
141  
142      def _get_all_slots(self):
143          return itertools.chain.from_iterable(getattr(cls, '__slots__', [])
144                                               for cls in self.__class__.__mro__)
145  
146      def __getstate__(self):
147          # We used to try to do a tuple of all slots here, but it
148          # doesn't work as self._all_slots isn't available at
149          # __setstate__() time.  Before that we tried to store a tuple
150          # of __slots__, but that didn't work as it didn't store the
151          # slots defined by ancestors.  This older way didn't fail
152          # outright, but ended up with partially broken objects, e.g.
153          # if you unpickled an A RR it wouldn't have rdclass and rdtype
154          # attributes, and would compare badly.
155          state = {}
156          for slot in self._get_all_slots():
157              state[slot] = getattr(self, slot)
158          return state
159  
160      def __setstate__(self, state):
161          for slot, val in state.items():
162              object.__setattr__(self, slot, val)
163          if not hasattr(self, 'rdcomment'):
164              # Pickled rdata from 2.0.x might not have a rdcomment, so add
165              # it if needed.
166              object.__setattr__(self, 'rdcomment', None)

168      def covers(self):
169          """Return the type a Rdata covers. 
170   
171          DNS SIG/RRSIG rdatas apply to a specific type; this type is 
172          returned by the covers() function.  If the rdata type is not 
173          SIG or RRSIG, dns.rdatatype.NONE is returned.  This is useful when 
174          creating rdatasets, allowing the rdataset to contain only RRSIGs 
175          of a particular type, e.g. RRSIG(NS). 
176   
177          Returns an ``int``. 
178          """
179  
180          return dns.rdatatype.NONE
181  
182      def extended_rdatatype(self):
183          """Return a 32-bit type value, the least significant 16 bits of 
184          which are the ordinary DNS type, and the upper 16 bits of which are 
185          the "covered" type, if any. 
186   
187          Returns an ``int``. 
188          """
189  
190          return self.covers() << 16 | self.rdtype

203      def to_wire(self, file=None, compress=None, origin=None,
204                  canonicalize=False):
205          """Convert an rdata to wire format. 
206   
207          Returns a ``bytes`` or ``None``. 
208          """

212          else:
213              f = io.BytesIO()
214              self._to_wire(f, compress, origin, canonicalize)
215              return f.getvalue()

217      def to_generic(self, origin=None):
218          """Creates a dns.rdata.GenericRdata equivalent of this rdata. 
219   
220          Returns a ``dns.rdata.GenericRdata``.

222          return dns.rdata.GenericRdata(self.rdclass, self.rdtype,

225      def to_digestable(self, origin=None):
226          """Convert rdata to a format suitable for digesting in hashes.  This 
227          is also the DNSSEC canonical form. 
228   
229          Returns a ``bytes``. 
230          """
231  
232          return self.to_wire(origin=origin, canonicalize=True)

234      def __repr__(self):
235          covers = self.covers()
236          if covers == dns.rdatatype.NONE:
237              ctext = ''
238          else:
239              ctext = '(' + dns.rdatatype.to_text(covers) + ')'
240          return '<DNS ' + dns.rdataclass.to_text(self.rdclass) + ' ' + \
241                 dns.rdatatype.to_text(self.rdtype) + ctext + ' rdata: ' + \
242                 str(self) + '>'
243  
244      def __str__(self):
245          return self.to_text()

247      def _cmp(self, other):
248          """Compare an rdata with another rdata of the same rdtype and 
249          rdclass. 
250   
251          For rdata with only absolute names: 
252              Return < 0 if self < other in the DNSSEC ordering, 0 if self 
253              == other, and > 0 if self > other. 
254          For rdata with at least one relative names: 
255              The rdata sorts before any rdata with only absolute names. 
256              When compared with another relative rdata, all names are 
257              made absolute as if they were relative to the root, as the 
258              proper origin is not available.  While this creates a stable 
259              ordering, it is NOT guaranteed to be the DNSSEC ordering. 
260              In the future, all ordering comparisons for rdata with 
261              relative names will be disallowed. 
262          """
263          try:
264              our = self.to_digestable()
265              our_relative = False
266          except dns.name.NeedAbsoluteNameOrOrigin:
267              if _allow_relative_comparisons:
268                  our = self.to_digestable(dns.name.root)
269              our_relative = True
270          try:
271              their = other.to_digestable()
272              their_relative = False
273          except dns.name.NeedAbsoluteNameOrOrigin:
274              if _allow_relative_comparisons:
275                  their = other.to_digestable(dns.name.root)
276              their_relative = True
277          if _allow_relative_comparisons:
278              if our_relative != their_relative:
279                  # For the purpose of comparison, all rdata with at least one
280                  # relative name is less than an rdata with only absolute names.
281                  if our_relative:
282                      return -1
283                  else:
284                      return 1
285          elif our_relative or their_relative:
286              raise NoRelativeRdataOrdering
287          if our == their:
288              return 0
289          elif our > their:
290              return 1
291          else:
292              return -1
293  
294      def __eq__(self, other):
295          if not isinstance(other, Rdata):
296              return False
297          if self.rdclass != other.rdclass or self.rdtype != other.rdtype:
298              return False
299          our_relative = False
300          their_relative = False
301          try:
302              our = self.to_digestable()
303          except dns.name.NeedAbsoluteNameOrOrigin:
304              our = self.to_digestable(dns.name.root)
305              our_relative = True
306          try:
307              their = other.to_digestable()
308          except dns.name.NeedAbsoluteNameOrOrigin:
309              their = other.to_digestable(dns.name.root)
310              their_relative = True
311          if our_relative != their_relative:
312              return False
313          return our == their
314  
315      def __ne__(self, other):
316          if not isinstance(other, Rdata):
317              return True
318          if self.rdclass != other.rdclass or self.rdtype != other.rdtype:
319              return True
320          return not self.__eq__(other)
321  
322      def __lt__(self, other):
323          if not isinstance(other, Rdata) or \
324                  self.rdclass != other.rdclass or self.rdtype != other.rdtype:
325  
326              return NotImplemented
327          return self._cmp(other) < 0
328  
329      def __le__(self, other):
330          if not isinstance(other, Rdata) or \
331                  self.rdclass != other.rdclass or self.rdtype != other.rdtype:
332              return NotImplemented
333          return self._cmp(other) <= 0
334  
335      def __ge__(self, other):
336          if not isinstance(other, Rdata) or \
337                  self.rdclass != other.rdclass or self.rdtype != other.rdtype:
338              return NotImplemented
339          return self._cmp(other) >= 0
340  
341      def __gt__(self, other):
342          if not isinstance(other, Rdata) or \
343                  self.rdclass != other.rdclass or self.rdtype != other.rdtype:
344              return NotImplemented
345          return self._cmp(other) > 0

347      def __hash__(self):
348          return hash(self.to_digestable(dns.name.root))

359      def replace(self, **kwargs):
360          """ 
361          Create a new Rdata instance based on the instance replace was 
362          invoked on. It is possible to pass different parameters to 
363          override the corresponding properties of the base Rdata. 
364   
365          Any field specific to the Rdata type can be replaced, but the 
366          *rdtype* and *rdclass* fields cannot. 
367   
368          Returns an instance of the same Rdata subclass as *self*. 
369          """
370  
371          # Get the constructor parameters.
372          parameters = inspect.signature(self.__init__).parameters
373  
374          # Ensure that all of the arguments correspond to valid fields.
375          # Don't allow rdclass or rdtype to be changed, though.
376          for key in kwargs:
377              if key == 'rdcomment':
378                  continue
379              if key not in parameters:
380                  raise AttributeError("'{}' object has no attribute '{}'"
381                                       .format(self.__class__.__name__, key))
382              if key in ('rdclass', 'rdtype'):
383                  raise AttributeError("Cannot overwrite '{}' attribute '{}'"
384                                       .format(self.__class__.__name__, key))
385  
386          # Construct the parameter list.  For each field, use the value in
387          # kwargs if present, and the current value otherwise.
388          args = (kwargs.get(key, getattr(self, key)) for key in parameters)
389  
390          # Create, validate, and return the new object.
391          rd = self.__class__(*args)
392          # The comment is not set in the constructor, so give it special
393          # handling.
394          rdcomment = kwargs.get('rdcomment', self.rdcomment)
395          if rdcomment is not None:
396              object.__setattr__(rd, 'rdcomment', rdcomment)
397          return rd

410      @classmethod
411      def _as_bytes(cls, value, encode=False, max_length=None, empty_ok=True):
412          if encode and isinstance(value, str):
413              value = value.encode()
414          elif isinstance(value, bytearray):
415              value = bytes(value)
416          elif not isinstance(value, bytes):
417              raise ValueError('not bytes')
418          if max_length is not None and len(value) > max_length:
419              raise ValueError('too long')
420          if not empty_ok and len(value) == 0:
421              raise ValueError('empty bytes not allowed')
422          return value

424      @classmethod
425      def _as_name(cls, value):
426          # Note that proper name conversion (e.g. with origin and IDNA
427          # awareness) is expected to be done via from_text.  This is just
428          # a simple thing for people invoking the constructor directly.
429          if isinstance(value, str):
430              return dns.name.from_text(value)
431          elif not isinstance(value, dns.name.Name):
432              raise ValueError('not a name')
433          return value

535  class GenericRdata(Rdata):
536  
537      """Generic Rdata Class 
538   
539      This class is used for rdata types for which we have no better 
540      implementation.  It implements the DNS "unknown RRs" scheme. 
541      """
542  
543      __slots__ = ['data']
544  
545      def __init__(self, rdclass, rdtype, data):
546          super().__init__(rdclass, rdtype)
547          object.__setattr__(self, 'data', data)
548  
549      def to_text(self, origin=None, relativize=True, **kw):
550          return r'\# %d ' % len(self.data) + _hexify(self.data, **kw)
551  
552      @classmethod
553      def from_text(cls, rdclass, rdtype, tok, origin=None, relativize=True,
554                    relativize_to=None):
555          token = tok.get()
556          if not token.is_identifier() or token.value != r'\#':
557              raise dns.exception.SyntaxError(
558                  r'generic rdata does not start with \#')
559          length = tok.get_int()
560          hex = tok.concatenate_remaining_identifiers(True).encode()
561          data = binascii.unhexlify(hex)
562          if len(data) != length:
563              raise dns.exception.SyntaxError(
564                  'generic rdata hex data has wrong length')
565          return cls(rdclass, rdtype, data)
566  
567      def _to_wire(self, file, compress=None, origin=None, canonicalize=False):
568          file.write(self.data)
569  
570      @classmethod
571      def from_wire_parser(cls, rdclass, rdtype, parser, origin=None):
572          return cls(rdclass, rdtype, parser.get_remaining())

574  _rdata_classes = {}
575  _module_prefix = 'dns.rdtypes'
576  
577  def get_rdata_class(rdclass, rdtype):
578      cls = _rdata_classes.get((rdclass, rdtype))
579      if not cls:
580          cls = _rdata_classes.get((dns.rdatatype.ANY, rdtype))
581          if not cls:
582              rdclass_text = dns.rdataclass.to_text(rdclass)
583              rdtype_text = dns.rdatatype.to_text(rdtype)
584              rdtype_text = rdtype_text.replace('-', '_')
585              try:
586                  mod = import_module('.'.join([_module_prefix,
587                                                rdclass_text, rdtype_text]))
588                  cls = getattr(mod, rdtype_text)
589                  _rdata_classes[(rdclass, rdtype)] = cls
590              except ImportError:
591                  try:
592                      mod = import_module('.'.join([_module_prefix,
593                                                    'ANY', rdtype_text]))
594                      cls = getattr(mod, rdtype_text)
595                      _rdata_classes[(dns.rdataclass.ANY, rdtype)] = cls
596                      _rdata_classes[(rdclass, rdtype)] = cls
597                  except ImportError:
598                      pass
599      if not cls:
600          cls = GenericRdata
601          _rdata_classes[(rdclass, rdtype)] = cls
602      return cls

605  def from_text(rdclass, rdtype, tok, origin=None, relativize=True,
606                relativize_to=None, idna_codec=None):
607      """Build an rdata object from text format. 
608   
609      This function attempts to dynamically load a class which 
610      implements the specified rdata class and type.  If there is no 
611      class-and-type-specific implementation, the GenericRdata class 
612      is used. 
613   
614      Once a class is chosen, its from_text() class method is called 
615      with the parameters to this function. 
616   
617      If *tok* is a ``str``, then a tokenizer is created and the string 
618      is used as its input. 
619   
620      *rdclass*, an ``int``, the rdataclass. 
621   
622      *rdtype*, an ``int``, the rdatatype. 
623   
624      *tok*, a ``dns.tokenizer.Tokenizer`` or a ``str``. 
625   
626      *origin*, a ``dns.name.Name`` (or ``None``), the 
627      origin to use for relative names. 
628   
629      *relativize*, a ``bool``.  If true, name will be relativized. 
630   
631      *relativize_to*, a ``dns.name.Name`` (or ``None``), the origin to use 
632      when relativizing names.  If not set, the *origin* value will be used. 
633   
634      *idna_codec*, a ``dns.name.IDNACodec``, specifies the IDNA 
635      encoder/decoder to use if a tokenizer needs to be created.  If 
636      ``None``, the default IDNA 2003 encoder/decoder is used.  If a 
637      tokenizer is not created, then the codec associated with the tokenizer 
638      is the one that is used. 
639   
640      Returns an instance of the chosen Rdata subclass. 
641   
642      """
643      if isinstance(tok, str):
644          tok = dns.tokenizer.Tokenizer(tok, idna_codec=idna_codec)
645      rdclass = dns.rdataclass.RdataClass.make(rdclass)
646      rdtype = dns.rdatatype.RdataType.make(rdtype)
647      cls = get_rdata_class(rdclass, rdtype)
648      with dns.exception.ExceptionWrapper(dns.exception.SyntaxError):
649          rdata = None
650          if cls != GenericRdata:
651              # peek at first token
652              token = tok.get()
653              tok.unget(token)
654              if token.is_identifier() and \
655                 token.value == r'\#':
656                  #
657                  # Known type using the generic syntax.  Extract the
658                  # wire form from the generic syntax, and then run
659                  # from_wire on it.
660                  #
661                  grdata = GenericRdata.from_text(rdclass, rdtype, tok, origin,
662                                                  relativize, relativize_to)
663                  rdata = from_wire(rdclass, rdtype, grdata.data, 0,
664                                    len(grdata.data), origin)
665                  #
666                  # If this comparison isn't equal, then there must have been
667                  # compressed names in the wire format, which is an error,
668                  # there being no reasonable context to decompress with.
669                  #
670                  rwire = rdata.to_wire()
671                  if rwire != grdata.data:
672                      raise dns.exception.SyntaxError('compressed data in '
673                                                      'generic syntax form '
674                                                      'of known rdatatype')
675          if rdata is None:
676              rdata = cls.from_text(rdclass, rdtype, tok, origin, relativize,
677                                    relativize_to)
678          token = tok.get_eol_as_token()
679          if token.comment is not None:
680              object.__setattr__(rdata, 'rdcomment', token.comment)
681          return rdata

684  def from_wire_parser(rdclass, rdtype, parser, origin=None):
685      """Build an rdata object from wire format 
686   
687      This function attempts to dynamically load a class which 
688      implements the specified rdata class and type.  If there is no 
689      class-and-type-specific implementation, the GenericRdata class 
690      is used. 
691   
692      Once a class is chosen, its from_wire() class method is called 
693      with the parameters to this function. 
694   
695      *rdclass*, an ``int``, the rdataclass. 
696   
697      *rdtype*, an ``int``, the rdatatype. 
698   
699      *parser*, a ``dns.wire.Parser``, the parser, which should be 
700      restricted to the rdata length. 
701   
702      *origin*, a ``dns.name.Name`` (or ``None``).  If not ``None``, 
703      then names will be relativized to this origin. 
704   
705      Returns an instance of the chosen Rdata subclass. 
706      """
707  
708      rdclass = dns.rdataclass.RdataClass.make(rdclass)
709      rdtype = dns.rdatatype.RdataType.make(rdtype)
710      cls = get_rdata_class(rdclass, rdtype)
711      with dns.exception.ExceptionWrapper(dns.exception.FormError):
712          return cls.from_wire_parser(rdclass, rdtype, parser, origin)

715  def from_wire(rdclass, rdtype, wire, current, rdlen, origin=None):
716      """Build an rdata object from wire format 
717   
718      This function attempts to dynamically load a class which 
719      implements the specified rdata class and type.  If there is no 
720      class-and-type-specific implementation, the GenericRdata class 
721      is used. 
722   
723      Once a class is chosen, its from_wire() class method is called 
724      with the parameters to this function. 
725   
726      *rdclass*, an ``int``, the rdataclass. 
727   
728      *rdtype*, an ``int``, the rdatatype. 
729   
730      *wire*, a ``bytes``, the wire-format message. 
731   
732      *current*, an ``int``, the offset in wire of the beginning of 
733      the rdata. 
734   
735      *rdlen*, an ``int``, the length of the wire-format rdata 
736   
737      *origin*, a ``dns.name.Name`` (or ``None``).  If not ``None``, 
738      then names will be relativized to this origin. 
739   
740      Returns an instance of the chosen Rdata subclass. 
741      """
742      parser = dns.wire.Parser(wire, current)
743      with parser.restrict_to(rdlen):
744          return from_wire_parser(rdclass, rdtype, parser, origin)

748      """DNS rdatatype already exists."""
749      supp_kwargs = {'rdclass', 'rdtype'}
750      fmt = "The rdata type with class {rdclass:d} and rdtype {rdtype:d} " + \
751          "already exists."

754  def register_type(implementation, rdtype, rdtype_text, is_singleton=False,
755                    rdclass=dns.rdataclass.IN):
756      """Dynamically register a module to handle an rdatatype. 
757   
758      *implementation*, a module implementing the type in the usual dnspython 
759      way. 
760   
761      *rdtype*, an ``int``, the rdatatype to register. 
762   
763      *rdtype_text*, a ``str``, the textual form of the rdatatype. 
764   
765      *is_singleton*, a ``bool``, indicating if the type is a singleton (i.e. 
766      RRsets of the type can have only one member.) 
767   
768      *rdclass*, the rdataclass of the type, or ``dns.rdataclass.ANY`` if 
769      it applies to all classes. 
770      """
771  
772      existing_cls = get_rdata_class(rdclass, rdtype)
773      if existing_cls != GenericRdata or dns.rdatatype.is_metatype(rdtype):
774          raise RdatatypeExists(rdclass=rdclass, rdtype=rdtype)
775      try:
776          if dns.rdatatype.RdataType(rdtype).name != rdtype_text:
777              raise RdatatypeExists(rdclass=rdclass, rdtype=rdtype)
778      except ValueError:
779          pass
780      _rdata_classes[(rdclass, rdtype)] = getattr(implementation,
781                                                  rdtype_text.replace('-', '_'))
782      dns.rdatatype.register_type(rdtype, rdtype_text, is_singleton)
