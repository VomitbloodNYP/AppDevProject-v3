18   import struct
19   
20   import dns.exception
21   import dns.immutable
22   import dns.rdata
23   import dns.tokenizer
24   
25   
26   @dns.immutable.immutable
27   class CAA(dns.rdata.Rdata):
28   
29       """CAA (Certification Authority Authorization) record"""
30   
31       # see: RFC 6844
32   
33       __slots__ = ['flags', 'tag', 'value']
34   
35       def __init__(self, rdclass, rdtype, flags, tag, value):
36           super().__init__(rdclass, rdtype)
37           self.flags = self._as_uint8(flags)
38           self.tag = self._as_bytes(tag, True, 255)
39           if not tag.isalnum():
40               raise ValueError("tag is not alphanumeric")
41           self.value = self._as_bytes(value)
42   
43       def to_text(self, origin=None, relativize=True, **kw):
44           return '%u %s "%s"' % (self.flags,
45                                  dns.rdata._escapify(self.tag),
46                                  dns.rdata._escapify(self.value))
47   
48       @classmethod
49       def from_text(cls, rdclass, rdtype, tok, origin=None, relativize=True,
50                     relativize_to=None):
51           flags = tok.get_uint8()
52           tag = tok.get_string().encode()
53           value = tok.get_string().encode()
54           return cls(rdclass, rdtype, flags, tag, value)
55   
56       def _to_wire(self, file, compress=None, origin=None, canonicalize=False):
57           file.write(struct.pack('!B', self.flags))
58           l = len(self.tag)
59           assert l < 256
60           file.write(struct.pack('!B', l))
61           file.write(self.tag)
62           file.write(self.value)
63   
64       @classmethod
65       def from_wire_parser(cls, rdclass, rdtype, parser, origin=None):
66           flags = parser.get_uint8()
67           tag = parser.get_counted_bytes()
68           value = parser.get_remaining()
69           return cls(rdclass, rdtype, flags, tag, value)
