53   class StopValidation(Exception):
54       """ 
55       Causes the validation chain to stop. 
56    
57       If StopValidation is raised, no more validators in the validation chain are 
58       called. If raised with a message, the message will be added to the errors 
59       list. 
60       """
61   
62       def __init__(self, message="", *args, **kwargs):
63           Exception.__init__(self, message, *args, **kwargs)

66   class EqualTo:
67       """ 
68       Compares the values of two fields. 
69    
70       :param fieldname: 
71           The name of the other field to compare to. 
72       :param message: 
73           Error message to raise in case of a validation error. Can be 
74           interpolated with `%(other_label)s` and `%(other_name)s` to provide a 
75           more helpful error. 
76       """
77   
78       def __init__(self, fieldname, message=None):
79           self.fieldname = fieldname
80           self.message = message
81   
82       def __call__(self, form, field):
83           try:
84               other = form[self.fieldname]
85           except KeyError as exc:
86               raise ValidationError(
87                   field.gettext("Invalid field name '%s'.") % self.fieldname
88               ) from exc
89           if field.data == other.data:
90               return
91   
92           d = {
93               "other_label": hasattr(other, "label")
94               and other.label.text
95               or self.fieldname,
96               "other_name": self.fieldname,
97           }
98           message = self.message
99           if message is None:
100              message = field.gettext("Field must be equal to %(other_name)s.")
101  
102          raise ValidationError(message % d)

105  class Length:
106      """ 
107      Validates the length of a string. 
108   
109      :param min: 
110          The minimum required length of the string. If not provided, minimum 
111          length will not be checked. 
112      :param max: 
113          The maximum length of the string. If not provided, maximum length 
114          will not be checked. 
115      :param message: 
116          Error message to raise in case of a validation error. Can be 
117          interpolated using `%(min)d` and `%(max)d` if desired. Useful defaults 
118          are provided depending on the existence of min and max. 
119   
120      When supported, sets the `minlength` and `maxlength` attributes on widgets. 
121      """
122  
123      def __init__(self, min=-1, max=-1, message=None):
124          assert (
125              min != -1 or max != -1
126          ), "At least one of `min` or `max` must be specified."
127          assert max == -1 or min <= max, "`min` cannot be more than `max`."
128          self.min = min
129          self.max = max
130          self.message = message
131          self.field_flags = {}
132          if self.min != -1:
133              self.field_flags["minlength"] = self.min
134          if self.max != -1:
135              self.field_flags["maxlength"] = self.max
136  
137      def __call__(self, form, field):
138          length = field.data and len(field.data) or 0
139          if length >= self.min and (self.max == -1 or length <= self.max):
140              return
141  
142          if self.message is not None:
143              message = self.message
144  
145          elif self.max == -1:
146              message = field.ngettext(
147                  "Field must be at least %(min)d character long.",
148                  "Field must be at least %(min)d characters long.",
149                  self.min,
150              )
151          elif self.min == -1:
152              message = field.ngettext(
153                  "Field cannot be longer than %(max)d character.",
154                  "Field cannot be longer than %(max)d characters.",
155                  self.max,
156              )
157          elif self.min == self.max:
158              message = field.ngettext(
159                  "Field must be exactly %(max)d character long.",
160                  "Field must be exactly %(max)d characters long.",
161                  self.max,
162              )
163          else:
164              message = field.gettext(
165                  "Field must be between %(min)d and %(max)d characters long."
166              )
167  
168          raise ValidationError(message % dict(min=self.min, max=self.max, length=length))

171  class NumberRange:
172      """ 
173      Validates that a number is of a minimum and/or maximum value, inclusive. 
174      This will work with any comparable number type, such as floats and 
175      decimals, not just integers. 
176   
177      :param min: 
178          The minimum required value of the number. If not provided, minimum 
179          value will not be checked. 
180      :param max: 
181          The maximum value of the number. If not provided, maximum value 
182          will not be checked. 
183      :param message: 
184          Error message to raise in case of a validation error. Can be 
185          interpolated using `%(min)s` and `%(max)s` if desired. Useful defaults 
186          are provided depending on the existence of min and max. 
187   
188      When supported, sets the `min` and `max` attributes on widgets. 
189      """
190  
191      def __init__(self, min=None, max=None, message=None):
192          self.min = min
193          self.max = max
194          self.message = message
195          self.field_flags = {}
196          if self.min is not None:
197              self.field_flags["min"] = self.min
198          if self.max is not None:
199              self.field_flags["max"] = self.max
200  
201      def __call__(self, form, field):
202          data = field.data
203          if (
204              data is not None
205              and not math.isnan(data)
206              and (self.min is None or data >= self.min)
207              and (self.max is None or data <= self.max)
208          ):
209              return
210  
211          if self.message is not None:
212              message = self.message
213  
214          # we use %(min)s interpolation to support floats, None, and
215          # Decimals without throwing a formatting exception.
216          elif self.max is None:
217              message = field.gettext("Number must be at least %(min)s.")
218  
219          elif self.min is None:
220              message = field.gettext("Number must be at most %(max)s.")
221  
222          else:
223              message = field.gettext("Number must be between %(min)s and %(max)s.")
224  
225          raise ValidationError(message % dict(min=self.min, max=self.max))

228  class Optional:
229      """ 
230      Allows empty input and stops the validation chain from continuing. 
231   
232      If input is empty, also removes prior errors (such as processing errors) 
233      from the field. 
234   
235      :param strip_whitespace: 
236          If True (the default) also stop the validation chain on input which 
237          consists of only whitespace. 
238   
239      Sets the `optional` attribute on widgets. 
240      """
241  
242      def __init__(self, strip_whitespace=True):
243          if strip_whitespace:
244              self.string_check = lambda s: s.strip()
245          else:
246              self.string_check = lambda s: s
247  
248          self.field_flags = {"optional": True}
249  
250      def __call__(self, form, field):
251          if (
252              not field.raw_data
253              or isinstance(field.raw_data[0], str)
254              and not self.string_check(field.raw_data[0])
255          ):
256              field.errors[:] = []
257              raise StopValidation()

260  class DataRequired:
261      """ 
262      Checks the field's data is 'truthy' otherwise stops the validation chain. 
263   
264      This validator checks that the ``data`` attribute on the field is a 'true' 
265      value (effectively, it does ``if field.data``.) Furthermore, if the data 
266      is a string type, a string containing only whitespace characters is 
267      considered false. 
268   
269      If the data is empty, also removes prior errors (such as processing errors) 
270      from the field. 
271   
272      **NOTE** this validator used to be called `Required` but the way it behaved 
273      (requiring coerced data, not input data) meant it functioned in a way 
274      which was not symmetric to the `Optional` validator and furthermore caused 
275      confusion with certain fields which coerced data to 'falsey' values like 
276      ``0``, ``Decimal(0)``, ``time(0)`` etc. Unless a very specific reason 
277      exists, we recommend using the :class:`InputRequired` instead. 
278   
279      :param message: 
280          Error message to raise in case of a validation error. 
281   
282      Sets the `required` attribute on widgets. 
283      """
284  
285      def __init__(self, message=None):
286          self.message = message
287          self.field_flags = {"required": True}
288  
289      def __call__(self, form, field):
290          if field.data and (not isinstance(field.data, str) or field.data.strip()):
291              return
292  
293          if self.message is None:
294              message = field.gettext("This field is required.")
295          else:
296              message = self.message
297  
298          field.errors[:] = []
299          raise StopValidation(message)

302  class InputRequired:
303      """ 
304      Validates that input was provided for this field. 
305   
306      Note there is a distinction between this and DataRequired in that 
307      InputRequired looks that form-input data was provided, and DataRequired 
308      looks at the post-coercion data. 
309   
310      Sets the `required` attribute on widgets. 
311      """
312  
313      def __init__(self, message=None):
314          self.message = message
315          self.field_flags = {"required": True}
316  
317      def __call__(self, form, field):
318          if field.raw_data and field.raw_data[0]:
319              return
320  
321          if self.message is None:
322              message = field.gettext("This field is required.")
323          else:
324              message = self.message
325  
326          field.errors[:] = []
327          raise StopValidation(message)

330  class Regexp:
331      """ 
332      Validates the field against a user provided regexp. 
333   
334      :param regex: 
335          The regular expression string to use. Can also be a compiled regular 
336          expression pattern. 
337      :param flags: 
338          The regexp flags to use, for example re.IGNORECASE. Ignored if 
339          `regex` is not a string. 
340      :param message: 
341          Error message to raise in case of a validation error. 
342      """
343  
344      def __init__(self, regex, flags=0, message=None):
345          if isinstance(regex, str):
346              regex = re.compile(regex, flags)
347          self.regex = regex
348          self.message = message
349  
350      def __call__(self, form, field, message=None):
351          match = self.regex.match(field.data or "")
352          if match:
353              return match
354  
355          if message is None:
356              if self.message is None:
357                  message = field.gettext("Invalid input.")
358              else:
359                  message = self.message
360  
361          raise ValidationError(message)

364  class Email:
365      """ 
366      Validates an email address. Requires email_validator package to be 
367      installed. For ex: pip install wtforms[email]. 
368   
369      :param message: 
370          Error message to raise in case of a validation error. 
371      :param granular_message: 
372          Use validation failed message from email_validator library 
373          (Default False). 
374      :param check_deliverability: 
375          Perform domain name resolution check (Default False). 
376      :param allow_smtputf8: 
377          Fail validation for addresses that would require SMTPUTF8 
378          (Default True). 
379      :param allow_empty_local: 
380          Allow an empty local part (i.e. @example.com), e.g. for validating 
381          Postfix aliases (Default False). 
382      """
383  
384      def __init__(
385          self,
386          message=None,
387          granular_message=False,
388          check_deliverability=False,
389          allow_smtputf8=True,
390          allow_empty_local=False,
391      ):
392          if email_validator is None:  # pragma: no cover
393              raise Exception("Install 'email_validator' for email validation support.")
394          self.message = message
395          self.granular_message = granular_message
396          self.check_deliverability = check_deliverability
397          self.allow_smtputf8 = allow_smtputf8
398          self.allow_empty_local = allow_empty_local
399  
400      def __call__(self, form, field):
401          try:
402              if field.data is None:
403                  raise email_validator.EmailNotValidError()
404              email_validator.validate_email(
405                  field.data,
406                  check_deliverability=self.check_deliverability,
407                  allow_smtputf8=self.allow_smtputf8,
408                  allow_empty_local=self.allow_empty_local,
409              )
410          except email_validator.EmailNotValidError as e:
411              message = self.message
412              if message is None:
413                  if self.granular_message:
414                      message = field.gettext(e)
415                  else:
416                      message = field.gettext("Invalid email address.")
417              raise ValidationError(message) from e

420  class IPAddress:
421      """ 
422      Validates an IP address. 
423   
424      :param ipv4: 
425          If True, accept IPv4 addresses as valid (default True) 
426      :param ipv6: 
427          If True, accept IPv6 addresses as valid (default False) 
428      :param message: 
429          Error message to raise in case of a validation error. 
430      """
431  
432      def __init__(self, ipv4=True, ipv6=False, message=None):
433          if not ipv4 and not ipv6:
434              raise ValueError(
435                  "IP Address Validator must have at least one of ipv4 or ipv6 enabled."
436              )
437          self.ipv4 = ipv4
438          self.ipv6 = ipv6
439          self.message = message
440  
441      def __call__(self, form, field):
442          value = field.data
443          valid = False
444          if value:
445              valid = (self.ipv4 and self.check_ipv4(value)) or (
446                  self.ipv6 and self.check_ipv6(value)
447              )
448  
449          if valid:
450              return
451  
452          message = self.message
453          if message is None:
454              message = field.gettext("Invalid IP address.")
455          raise ValidationError(message)
456  
457      @classmethod
458      def check_ipv4(cls, value):
459          try:
460              address = ipaddress.ip_address(value)
461          except ValueError:
462              return False
463  
464          if not isinstance(address, ipaddress.IPv4Address):
465              return False
466  
467          return True
468  
469      @classmethod
470      def check_ipv6(cls, value):
471          try:
472              address = ipaddress.ip_address(value)
473          except ValueError:
474              return False
475  
476          if not isinstance(address, ipaddress.IPv6Address):
477              return False
478  
479          return True

502  class URL(Regexp):
503      """ 
504      Simple regexp based url validation. Much like the email validator, you 
505      probably want to validate the url later by other means if the url must 
506      resolve. 
507   
508      :param require_tld: 
509          If true, then the domain-name portion of the URL must contain a .tld 
510          suffix.  Set this to false if you want to allow domains like 
511          `localhost`. 
512      :param message: 
513          Error message to raise in case of a validation error. 
514      """
515  
516      def __init__(self, require_tld=True, message=None):
517          regex = (
518              r"^[a-z]+://"
519              r"(?P<host>[^\/\?:]+)"
520              r"(?P<port>:[0-9]+)?"
521              r"(?P<path>\/.*?)?"
522              r"(?P<query>\?.*)?$"
523          )
524          super().__init__(regex, re.IGNORECASE, message)
525          self.validate_hostname = HostnameValidation(
526              require_tld=require_tld, allow_ip=True
527          )
528  
529      def __call__(self, form, field):
530          message = self.message
531          if message is None:
532              message = field.gettext("Invalid URL.")
533  
534          match = super().__call__(form, field, message)
535          if not self.validate_hostname(match.group("host")):
536              raise ValidationError(message)

560  class AnyOf:
561      """ 
562      Compares the incoming data to a sequence of valid inputs. 
563   
564      :param values: 
565          A sequence of valid inputs. 
566      :param message: 
567          Error message to raise in case of a validation error. `%(values)s` 
568          contains the list of values. 
569      :param values_formatter: 
570          Function used to format the list of values in the error message. 
571      """
572  
573      def __init__(self, values, message=None, values_formatter=None):
574          self.values = values
575          self.message = message
576          if values_formatter is None:
577              values_formatter = self.default_values_formatter
578          self.values_formatter = values_formatter
579  
580      def __call__(self, form, field):
581          if field.data in self.values:
582              return
583  
584          message = self.message
585          if message is None:
586              message = field.gettext("Invalid value, must be one of: %(values)s.")
587  
588          raise ValidationError(message % dict(values=self.values_formatter(self.values)))
589  
590      @staticmethod
591      def default_values_formatter(values):
592          return ", ".join(str(x) for x in values)

595  class NoneOf:
596      """ 
597      Compares the incoming data to a sequence of invalid inputs. 
598   
599      :param values: 
600          A sequence of invalid inputs. 
601      :param message: 
602          Error message to raise in case of a validation error. `%(values)s` 
603          contains the list of values. 
604      :param values_formatter: 
605          Function used to format the list of values in the error message. 
606      """
607  
608      def __init__(self, values, message=None, values_formatter=None):
609          self.values = values
610          self.message = message
611          if values_formatter is None:
612              values_formatter = self.default_values_formatter
613          self.values_formatter = values_formatter
614  
615      def __call__(self, form, field):
616          if field.data not in self.values:
617              return
618  
619          message = self.message
620          if message is None:
621              message = field.gettext("Invalid value, can't be any of: %(values)s.")
622  
623          raise ValidationError(message % dict(values=self.values_formatter(self.values)))
624  
625      @staticmethod
626      def default_values_formatter(v):
627          return ", ".join(str(x) for x in v)

630  class HostnameValidation:
631      """ 
632      Helper class for checking hostnames for validation. 
633   
634      This is not a validator in and of itself, and as such is not exported. 
635      """
636  
637      hostname_part = re.compile(r"^(xn-|[a-z0-9_]+)(-[a-z0-9_-]+)*$", re.IGNORECASE)
638      tld_part = re.compile(r"^([a-z]{2,20}|xn--([a-z0-9]+-)*[a-z0-9]+)$", re.IGNORECASE)
639  
640      def __init__(self, require_tld=True, allow_ip=False):
641          self.require_tld = require_tld
642          self.allow_ip = allow_ip
643  
644      def __call__(self, hostname):
645          if self.allow_ip and (
646              IPAddress.check_ipv4(hostname) or IPAddress.check_ipv6(hostname)
647          ):
648              return True
649  
650          # Encode out IDNA hostnames. This makes further validation easier.
651          try:
652              hostname = hostname.encode("idna")
653          except UnicodeError:
654              pass
655  
656          # Turn back into a string in Python 3x
657          if not isinstance(hostname, str):
658              hostname = hostname.decode("ascii")
659  
660          if len(hostname) > 253:
661              return False
662  
663          # Check that all labels in the hostname are valid
664          parts = hostname.split(".")
665          for part in parts:
666              if not part or len(part) > 63:
667                  return False
668              if not self.hostname_part.match(part):
669                  return False
670  
671          if self.require_tld and (len(parts) < 2 or not self.tld_part.match(parts[-1])):
672              return False
673  
674          return True
