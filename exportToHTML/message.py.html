185      def to_text(self, origin=None, relativize=True, **kw):
186          """Convert the message to text. 
187   
188          The *origin*, *relativize*, and any other keyword 
189          arguments are passed to the RRset ``to_wire()`` method. 
190   
191          Returns a ``str``. 
192          """
193  
194          s = io.StringIO()
195          s.write('id %d\n' % self.id)
196          s.write('opcode %s\n' % dns.opcode.to_text(self.opcode()))
197          s.write('rcode %s\n' % dns.rcode.to_text(self.rcode()))
198          s.write('flags %s\n' % dns.flags.to_text(self.flags))
199          if self.edns >= 0:
200              s.write('edns %s\n' % self.edns)
201              if self.ednsflags != 0:
202                  s.write('eflags %s\n' %
203                          dns.flags.edns_to_text(self.ednsflags))
204              s.write('payload %d\n' % self.payload)
205          for opt in self.options:
206              s.write('option %s\n' % opt.to_text())
207          for (name, which) in self._section_enum.__members__.items():
208              s.write(f';{name}\n')
209              for rrset in self.section_from_number(which):
210                  s.write(rrset.to_text(origin, relativize, **kw))
211                  s.write('\n')
212          #
213          # We strip off the final \n so the caller can print the result without
214          # doing weird things to get around eccentricities in Python print
215          # formatting
216          #
217          return s.getvalue()[:-1]

219      def __eq__(self, other):
220          """Two messages are equal if they have the same content in the 
221          header, question, answer, and authority sections. 
222   
223          Returns a ``bool``. 
224          """
225  
226          if not isinstance(other, Message):
227              return False
228          if self.id != other.id:
229              return False
230          if self.flags != other.flags:
231              return False
232          for i, section in enumerate(self.sections):
233              other_section = other.sections[i]
234              for n in section:
235                  if n not in other_section:
236                      return False
237              for n in other_section:
238                  if n not in section:
239                      return False
240          return True
241  
242      def __ne__(self, other):
243          return not self.__eq__(other)

245      def is_response(self, other):
246          """Is *other*, also a ``dns.message.Message``, a response to this 
247          message? 
248   
249          Returns a ``bool``. 
250          """
251  
252          if other.flags & dns.flags.QR == 0 or \
253             self.id != other.id or \
254             dns.opcode.from_flags(self.flags) != \
255             dns.opcode.from_flags(other.flags):
256              return False
257          if other.rcode() in {dns.rcode.FORMERR, dns.rcode.SERVFAIL,
258                               dns.rcode.NOTIMP, dns.rcode.REFUSED}:
259              # We don't check the question section in these cases if
260              # the other question section is empty, even though they
261              # still really ought to have a question section.
262              if len(other.question) == 0:
263                  return True
264          if dns.opcode.is_update(self.flags):
265              # This is assuming the "sender doesn't include anything
266              # from the update", but we don't care to check the other
267              # case, which is that all the sections are returned and
268              # identical.
269              return True
270          for n in self.question:
271              if n not in other.question:
272                  return False
273          for n in other.question:
274              if n not in self.question:
275                  return False
276          return True

278      def section_number(self, section):
279          """Return the "section number" of the specified section for use 
280          in indexing. 
281   
282          *section* is one of the section attributes of this message. 
283   
284          Raises ``ValueError`` if the section isn't known. 
285   
286          Returns an ``int``. 
287          """
288  
289          for i, our_section in enumerate(self.sections):
290              if section is our_section:
291                  return self._section_enum(i)
292          raise ValueError('unknown section')

294      def section_from_number(self, number):
295          """Return the section list associated with the specified section 
296          number. 
297   
298          *number* is a section number `int` or the text form of a section 
299          name. 
300   
301          Raises ``ValueError`` if the section isn't known. 
302   
303          Returns a ``list``. 
304          """
305  
306          section = self._section_enum.make(number)
307          return self.sections[section]

309      def find_rrset(self, section, name, rdclass, rdtype,
310                     covers=dns.rdatatype.NONE, deleting=None, create=False,
311                     force_unique=False):
312          """Find the RRset with the given attributes in the specified section. 
313   
314          *section*, an ``int`` section number, or one of the section 
315          attributes of this message.  This specifies the 
316          the section of the message to search.  For example:: 
317   
318              my_message.find_rrset(my_message.answer, name, rdclass, rdtype) 
319              my_message.find_rrset(dns.message.ANSWER, name, rdclass, rdtype) 
320   
321          *name*, a ``dns.name.Name``, the name of the RRset. 
322   
323          *rdclass*, an ``int``, the class of the RRset. 
324   
325          *rdtype*, an ``int``, the type of the RRset. 
326   
327          *covers*, an ``int`` or ``None``, the covers value of the RRset. 
328          The default is ``None``. 
329   
330          *deleting*, an ``int`` or ``None``, the deleting value of the RRset. 
331          The default is ``None``. 
332   
333          *create*, a ``bool``.  If ``True``, create the RRset if it is not found. 
334          The created RRset is appended to *section*. 
335   
336          *force_unique*, a ``bool``.  If ``True`` and *create* is also ``True``, 
337          create a new RRset regardless of whether a matching RRset exists 
338          already.  The default is ``False``.  This is useful when creating 
339          DDNS Update messages, as order matters for them. 
340   
341          Raises ``KeyError`` if the RRset was not found and create was 
342          ``False``. 
343   
344          Returns a ``dns.rrset.RRset object``. 
345          """
346  
347          if isinstance(section, int):
348              section_number = section
349              section = self.section_from_number(section_number)
350          else:
351              section_number = self.section_number(section)
352          key = (section_number, name, rdclass, rdtype, covers, deleting)
353          if not force_unique:
354              if self.index is not None:
355                  rrset = self.index.get(key)
356                  if rrset is not None:
357                      return rrset
358              else:
359                  for rrset in section:
360                      if rrset.full_match(name, rdclass, rdtype, covers,
361                                          deleting):
362                          return rrset
363          if not create:
364              raise KeyError
365          rrset = dns.rrset.RRset(name, rdclass, rdtype, covers, deleting)
366          section.append(rrset)
367          if self.index is not None:
368              self.index[key] = rrset
369          return rrset

371      def get_rrset(self, section, name, rdclass, rdtype,
372                    covers=dns.rdatatype.NONE, deleting=None, create=False,
373                    force_unique=False):
374          """Get the RRset with the given attributes in the specified section. 
375   
376          If the RRset is not found, None is returned. 
377   
378          *section*, an ``int`` section number, or one of the section 
379          attributes of this message.  This specifies the 
380          the section of the message to search.  For example:: 
381   
382              my_message.get_rrset(my_message.answer, name, rdclass, rdtype) 
383              my_message.get_rrset(dns.message.ANSWER, name, rdclass, rdtype) 
384   
385          *name*, a ``dns.name.Name``, the name of the RRset. 
386   
387          *rdclass*, an ``int``, the class of the RRset. 
388   
389          *rdtype*, an ``int``, the type of the RRset. 
390   
391          *covers*, an ``int`` or ``None``, the covers value of the RRset. 
392          The default is ``None``. 
393   
394          *deleting*, an ``int`` or ``None``, the deleting value of the RRset. 
395          The default is ``None``. 
396   
397          *create*, a ``bool``.  If ``True``, create the RRset if it is not found. 
398          The created RRset is appended to *section*. 
399   
400          *force_unique*, a ``bool``.  If ``True`` and *create* is also ``True``, 
401          create a new RRset regardless of whether a matching RRset exists 
402          already.  The default is ``False``.  This is useful when creating 
403          DDNS Update messages, as order matters for them. 
404   
405          Returns a ``dns.rrset.RRset object`` or ``None``. 
406          """
407  
408          try:
409              rrset = self.find_rrset(section, name, rdclass, rdtype, covers,
410                                      deleting, create, force_unique)
411          except KeyError:
412              rrset = None
413          return rrset

415      def to_wire(self, origin=None, max_size=0, multi=False, tsig_ctx=None,
416                  **kw):
417          """Return a string containing the message in DNS compressed wire 
418          format. 
419   
420          Additional keyword arguments are passed to the RRset ``to_wire()`` 
421          method. 
422   
423          *origin*, a ``dns.name.Name`` or ``None``, the origin to be appended 
424          to any relative names.  If ``None``, and the message has an origin 
425          attribute that is not ``None``, then it will be used. 
426   
427          *max_size*, an ``int``, the maximum size of the wire format 
428          output; default is 0, which means "the message's request 
429          payload, if nonzero, or 65535". 
430   
431          *multi*, a ``bool``, should be set to ``True`` if this message is 
432          part of a multiple message sequence. 
433   
434          *tsig_ctx*, a ``dns.tsig.HMACTSig`` or ``dns.tsig.GSSTSig`` object, the 
435          ongoing TSIG context, used when signing zone transfers. 
436   
437          Raises ``dns.exception.TooBig`` if *max_size* was exceeded. 
438   
439          Returns a ``bytes``. 
440          """
441  
442          if origin is None and self.origin is not None:
443              origin = self.origin
444          if max_size == 0:
445              if self.request_payload != 0:
446                  max_size = self.request_payload
447              else:
448                  max_size = 65535
449          if max_size < 512:
450              max_size = 512
451          elif max_size > 65535:
452              max_size = 65535
453          r = dns.renderer.Renderer(self.id, self.flags, max_size, origin)
454          for rrset in self.question:
455              r.add_question(rrset.name, rrset.rdtype, rrset.rdclass)
456          for rrset in self.answer:
457              r.add_rrset(dns.renderer.ANSWER, rrset, **kw)
458          for rrset in self.authority:
459              r.add_rrset(dns.renderer.AUTHORITY, rrset, **kw)
460          if self.opt is not None:
461              r.add_rrset(dns.renderer.ADDITIONAL, self.opt)
462          for rrset in self.additional:
463              r.add_rrset(dns.renderer.ADDITIONAL, rrset, **kw)
464          r.write_header()
465          if self.tsig is not None:
466              (new_tsig, ctx) = dns.tsig.sign(r.get_wire(),
467                                              self.keyring,
468                                              self.tsig[0],
469                                              int(time.time()),
470                                              self.request_mac,
471                                              tsig_ctx,
472                                              multi)
473              self.tsig.clear()
474              self.tsig.add(new_tsig)
475              r.add_rrset(dns.renderer.ADDITIONAL, self.tsig)
476              r.write_header()
477              if multi:
478                  self.tsig_ctx = ctx
479          return r.get_wire()

489      def use_tsig(self, keyring, keyname=None, fudge=300,
490                   original_id=None, tsig_error=0, other_data=b'',
491                   algorithm=dns.tsig.default_algorithm):
492          """When sending, a TSIG signature using the specified key 
493          should be added. 
494   
495          *key*, a ``dns.tsig.Key`` is the key to use.  If a key is specified, 
496          the *keyring* and *algorithm* fields are not used. 
497   
498          *keyring*, a ``dict``, ``callable`` or ``dns.tsig.Key``, is either 
499          the TSIG keyring or key to use. 
500   
501          The format of a keyring dict is a mapping from TSIG key name, as 
502          ``dns.name.Name`` to ``dns.tsig.Key`` or a TSIG secret, a ``bytes``. 
503          If a ``dict`` *keyring* is specified but a *keyname* is not, the key 
504          used will be the first key in the *keyring*.  Note that the order of 
505          keys in a dictionary is not defined, so applications should supply a 
506          keyname when a ``dict`` keyring is used, unless they know the keyring 
507          contains only one key.  If a ``callable`` keyring is specified, the 
508          callable will be called with the message and the keyname, and is 
509          expected to return a key. 
510   
511          *keyname*, a ``dns.name.Name``, ``str`` or ``None``, the name of 
512          this TSIG key to use; defaults to ``None``.  If *keyring* is a 
513          ``dict``, the key must be defined in it.  If *keyring* is a 
514          ``dns.tsig.Key``, this is ignored. 
515   
516          *fudge*, an ``int``, the TSIG time fudge. 
517   
518          *original_id*, an ``int``, the TSIG original id.  If ``None``, 
519          the message's id is used. 
520   
521          *tsig_error*, an ``int``, the TSIG error code. 
522   
523          *other_data*, a ``bytes``, the TSIG other data. 
524   
525          *algorithm*, a ``dns.name.Name``, the TSIG algorithm to use.  This is 
526          only used if *keyring* is a ``dict``, and the key entry is a ``bytes``. 
527          """
528  
529          if isinstance(keyring, dns.tsig.Key):
530              key = keyring
531              keyname = key.name
532          elif callable(keyring):
533              key = keyring(self, keyname)
534          else:
535              if isinstance(keyname, str):
536                  keyname = dns.name.from_text(keyname)
537              if keyname is None:
538                  keyname = next(iter(keyring))
539              key = keyring[keyname]
540              if isinstance(key, bytes):
541                  key = dns.tsig.Key(keyname, key, algorithm)
542          self.keyring = key
543          if original_id is None:
544              original_id = self.id
545          self.tsig = self._make_tsig(keyname, self.keyring.algorithm, 0, fudge,
546                                      b'', original_id, tsig_error, other_data)

586      def use_edns(self, edns=0, ednsflags=0, payload=DEFAULT_EDNS_PAYLOAD,
587                   request_payload=None, options=None):
588          """Configure EDNS behavior. 
589   
590          *edns*, an ``int``, is the EDNS level to use.  Specifying 
591          ``None``, ``False``, or ``-1`` means "do not use EDNS", and in this case 
592          the other parameters are ignored.  Specifying ``True`` is 
593          equivalent to specifying 0, i.e. "use EDNS0". 
594   
595          *ednsflags*, an ``int``, the EDNS flag values. 
596   
597          *payload*, an ``int``, is the EDNS sender's payload field, which is the 
598          maximum size of UDP datagram the sender can handle.  I.e. how big 
599          a response to this message can be. 
600   
601          *request_payload*, an ``int``, is the EDNS payload size to use when 
602          sending this message.  If not specified, defaults to the value of 
603          *payload*. 
604   
605          *options*, a list of ``dns.edns.Option`` objects or ``None``, the EDNS 
606          options. 
607          """
608  
609          if edns is None or edns is False:
610              edns = -1
611          elif edns is True:
612              edns = 0
613          if edns < 0:
614              self.opt = None
615              self.request_payload = 0
616          else:
617              # make sure the EDNS version in ednsflags agrees with edns
618              ednsflags &= 0xFF00FFFF
619              ednsflags |= (edns << 16)
620              if options is None:
621                  options = []
622              self.opt = self._make_opt(ednsflags, payload, options)
623              if request_payload is None:
624                  request_payload = payload
625              self.request_payload = request_payload

662      def want_dnssec(self, wanted=True):
663          """Enable or disable 'DNSSEC desired' flag in requests. 
664   
665          *wanted*, a ``bool``.  If ``True``, then DNSSEC data is 
666          desired in the response, EDNS is enabled if required, and then 
667          the DO bit is set.  If ``False``, the DO bit is cleared if 
668          EDNS is enabled. 
669          """
670  
671          if wanted:
672              self.ednsflags |= dns.flags.DO
673          elif self.opt:
674              self.ednsflags &= ~dns.flags.DO

684          """Set the rcode. 
685   
686          *rcode*, an ``int``, is the rcode to set. 
687          """
688          (value, evalue) = dns.rcode.to_flags(rcode)
689          self.flags &= 0xFFF0
690          self.flags |= value
691          self.ednsflags &= 0x00FFFFFF
692          self.ednsflags |= evalue
693  
694      def opcode(self):
695          """Return the opcode. 
696   
697          Returns an ``int``. 
698          """
699          return dns.opcode.from_flags(int(self.flags))
700  
701      def set_opcode(self, opcode):
702          """Set the opcode. 
703   
704          *opcode*, an ``int``, is the opcode to set. 
705          """
706          self.flags &= 0x87FF
707          self.flags |= dns.opcode.to_flags(opcode)
708  
709      def _get_one_rr_per_rrset(self, value):
710          # What the caller picked is fine.
711          return value
712  
713      # pylint: disable=unused-argument
714  
715      def _parse_rr_header(self, section, name, rdclass, rdtype):
716          return (rdclass, rdtype, None, False)
717  
718      # pylint: enable=unused-argument
719  
720      def _parse_special_rr_header(self, section, count, position,
721                                   name, rdclass, rdtype):
722          if rdtype == dns.rdatatype.OPT:
723              if section != MessageSection.ADDITIONAL or self.opt or \
724                 name != dns.name.root:
725                  raise BadEDNS
726          elif rdtype == dns.rdatatype.TSIG:
727              if section != MessageSection.ADDITIONAL or \
728                 rdclass != dns.rdatatype.ANY or \
729                 position != count - 1:
730                  raise BadTSIG
731          return (rdclass, rdtype, None, False)

734  class ChainingResult:
735      """The result of a call to dns.message.QueryMessage.resolve_chaining(). 
736   
737      The ``answer`` attribute is the answer RRSet, or ``None`` if it doesn't 
738      exist. 
739   
740      The ``canonical_name`` attribute is the canonical name after all 
741      chaining has been applied (this is the name as ``rrset.name`` in cases 
742      where rrset is not ``None``). 
743   
744      The ``minimum_ttl`` attribute is the minimum TTL, i.e. the TTL to 
745      use if caching the data.  It is the smallest of all the CNAME TTLs 
746      and either the answer TTL if it exists or the SOA TTL and SOA 
747      minimum values for negative answers. 
748   
749      The ``cnames`` attribute is a list of all the CNAME RRSets followed to 
750      get to the canonical name. 
751      """
752      def __init__(self, canonical_name, answer, minimum_ttl, cnames):
753          self.canonical_name = canonical_name
754          self.answer = answer
755          self.minimum_ttl = minimum_ttl
756          self.cnames = cnames

759  class QueryMessage(Message):
760      def resolve_chaining(self):
761          """Follow the CNAME chain in the response to determine the answer 
762          RRset. 
763   
764          Raises ``dns.message.NotQueryResponse`` if the message is not 
765          a response. 
766   
767          Raises ``dns.message.ChainTooLong`` if the CNAME chain is too long. 
768   
769          Raises ``dns.message.AnswerForNXDOMAIN`` if the rcode is NXDOMAIN 
770          but an answer was found. 
771   
772          Raises ``dns.exception.FormError`` if the question count is not 1. 
773   
774          Returns a ChainingResult object. 
775          """
776          if self.flags & dns.flags.QR == 0:
777              raise NotQueryResponse
778          if len(self.question) != 1:
779              raise dns.exception.FormError
780          question = self.question[0]
781          qname = question.name
782          min_ttl = dns.ttl.MAX_TTL
783          answer = None
784          count = 0
785          cnames = []
786          while count < MAX_CHAIN:
787              try:
788                  answer = self.find_rrset(self.answer, qname, question.rdclass,
789                                           question.rdtype)
790                  min_ttl = min(min_ttl, answer.ttl)
791                  break
792              except KeyError:
793                  if question.rdtype != dns.rdatatype.CNAME:
794                      try:
795                          crrset = self.find_rrset(self.answer, qname,
796                                                   question.rdclass,
797                                                   dns.rdatatype.CNAME)
798                          cnames.append(crrset)
799                          min_ttl = min(min_ttl, crrset.ttl)
800                          for rd in crrset:
801                              qname = rd.target
802                              break
803                          count += 1
804                          continue
805                      except KeyError:
806                          # Exit the chaining loop
807                          break
808                  else:
809                      # Exit the chaining loop
810                      break
811          if count >= MAX_CHAIN:
812              raise ChainTooLong
813          if self.rcode() == dns.rcode.NXDOMAIN and answer is not None:
814              raise AnswerForNXDOMAIN
815          if answer is None:
816              # Further minimize the TTL with NCACHE.
817              auname = qname
818              while True:
819                  # Look for an SOA RR whose owner name is a superdomain
820                  # of qname.
821                  try:
822                      srrset = self.find_rrset(self.authority, auname,
823                                               question.rdclass,
824                                               dns.rdatatype.SOA)
825                      min_ttl = min(min_ttl, srrset.ttl, srrset[0].minimum)
826                      break
827                  except KeyError:
828                      try:
829                          auname = auname.parent()
830                      except dns.name.NoParent:
831                          break
832          return ChainingResult(qname, answer, min_ttl, cnames)
833  
834      def canonical_name(self):
835          """Return the canonical name of the first name in the question 
836          section. 
837   
838          Raises ``dns.message.NotQueryResponse`` if the message is not 
839          a response. 
840   
841          Raises ``dns.message.ChainTooLong`` if the CNAME chain is too long. 
842   
843          Raises ``dns.message.AnswerForNXDOMAIN`` if the rcode is NXDOMAIN 
844          but an answer was found. 
845   
846          Raises ``dns.exception.FormError`` if the question count is not 1. 
847          """
848          return self.resolve_chaining().canonical_name

870  class _WireReader:
871  
872      """Wire format reader. 
873   
874      parser: the binary parser 
875      message: The message object being built 
876      initialize_message: Callback to set message parsing options 
877      question_only: Are we only reading the question? 
878      one_rr_per_rrset: Put each RR into its own RRset? 
879      keyring: TSIG keyring 
880      ignore_trailing: Ignore trailing junk at end of request? 
881      multi: Is this message part of a multi-message sequence? 
882      DNS dynamic updates. 
883      continue_on_error: try to extract as much information as possible from 
884      the message, accumulating MessageErrors in the *errors* attribute instead of 
885      raising them. 
886      """
887  
888      def __init__(self, wire, initialize_message, question_only=False,
889                   one_rr_per_rrset=False, ignore_trailing=False,
890                   keyring=None, multi=False, continue_on_error=False):
891          self.parser = dns.wire.Parser(wire)
892          self.message = None
893          self.initialize_message = initialize_message
894          self.question_only = question_only
895          self.one_rr_per_rrset = one_rr_per_rrset
896          self.ignore_trailing = ignore_trailing
897          self.keyring = keyring
898          self.multi = multi
899          self.continue_on_error = continue_on_error
900          self.errors = []
901  
902      def _get_question(self, section_number, qcount):
903          """Read the next *qcount* records from the wire data and add them to 
904          the question section. 
905          """
906  
907          section = self.message.sections[section_number]
908          for _ in range(qcount):
909              qname = self.parser.get_name(self.message.origin)
910              (rdtype, rdclass) = self.parser.get_struct('!HH')
911              (rdclass, rdtype, _, _) = \
912                  self.message._parse_rr_header(section_number, qname, rdclass,
913                                                rdtype)
914              self.message.find_rrset(section, qname, rdclass, rdtype,
915                                      create=True, force_unique=True)
916  
917      def _add_error(self, e):
918          self.errors.append(MessageError(e, self.parser.current))
919  
920      def _get_section(self, section_number, count):
921          """Read the next I{count} records from the wire data and add them to 
922          the specified section. 
923   
924          section_number: the section of the message to which to add records 
925          count: the number of records to read 
926          """
927  
928          section = self.message.sections[section_number]
929          force_unique = self.one_rr_per_rrset
930          for i in range(count):
931              rr_start = self.parser.current
932              absolute_name = self.parser.get_name()
933              if self.message.origin is not None:
934                  name = absolute_name.relativize(self.message.origin)
935              else:
936                  name = absolute_name
937              (rdtype, rdclass, ttl, rdlen) = self.parser.get_struct('!HHIH')
938              if rdtype in (dns.rdatatype.OPT, dns.rdatatype.TSIG):
939                  (rdclass, rdtype, deleting, empty) = \
940                      self.message._parse_special_rr_header(section_number,
941                                                            count, i, name,
942                                                            rdclass, rdtype)
943              else:
944                  (rdclass, rdtype, deleting, empty) = \
945                      self.message._parse_rr_header(section_number,
946                                                    name, rdclass, rdtype)
947              try:
948                  rdata_start = self.parser.current
949                  if empty:
950                      if rdlen > 0:
951                          raise dns.exception.FormError
952                      rd = None
953                      covers = dns.rdatatype.NONE
954                  else:
955                      with self.parser.restrict_to(rdlen):
956                          rd = dns.rdata.from_wire_parser(rdclass, rdtype,
957                                                          self.parser,
958                                                          self.message.origin)
959                      covers = rd.covers()
960                  if self.message.xfr and rdtype == dns.rdatatype.SOA:
961                      force_unique = True
962                  if rdtype == dns.rdatatype.OPT:
963                      self.message.opt = dns.rrset.from_rdata(name, ttl, rd)
964                  elif rdtype == dns.rdatatype.TSIG:
965                      if self.keyring is None:
966                          raise UnknownTSIGKey('got signed message without '
967                                               'keyring')
968                      if isinstance(self.keyring, dict):
969                          key = self.keyring.get(absolute_name)
970                          if isinstance(key, bytes):
971                              key = dns.tsig.Key(absolute_name, key, rd.algorithm)
972                      elif callable(self.keyring):
973                          key = self.keyring(self.message, absolute_name)
974                      else:
975                          key = self.keyring
976                      if key is None:
977                          raise UnknownTSIGKey("key '%s' unknown" % name)
978                      self.message.keyring = key
979                      self.message.tsig_ctx = \
980                          dns.tsig.validate(self.parser.wire,
981                                          key,
982                                          absolute_name,
983                                          rd,
984                                          int(time.time()),
985                                          self.message.request_mac,
986                                          rr_start,
987                                          self.message.tsig_ctx,
988                                          self.multi)
989                      self.message.tsig = dns.rrset.from_rdata(absolute_name, 0,
990                                                               rd)
991                  else:
992                      rrset = self.message.find_rrset(section, name,
993                                                      rdclass, rdtype, covers,
994                                                      deleting, True,
995                                                      force_unique)
996                      if rd is not None:
997                          if ttl > 0x7fffffff:
998                              ttl = 0
999                          rrset.add(rd, ttl)
1000             except Exception as e:
1001                 if self.continue_on_error:
1002                     self._add_error(e)
1003                     self.parser.seek(rdata_start + rdlen)
1004                 else:
1005                     raise
1006 
1007     def read(self):
1008         """Read a wire format DNS message and build a dns.message.Message 
1009         object."""
1010 
1011         if self.parser.remaining() < 12:
1012             raise ShortHeader
1013         (id, flags, qcount, ancount, aucount, adcount) = \
1014             self.parser.get_struct('!HHHHHH')
1015         factory = _message_factory_from_opcode(dns.opcode.from_flags(flags))
1016         self.message = factory(id=id)
1017         self.message.flags = dns.flags.Flag(flags)
1018         self.initialize_message(self.message)
1019         self.one_rr_per_rrset = \
1020             self.message._get_one_rr_per_rrset(self.one_rr_per_rrset)
1021         try:
1022             self._get_question(MessageSection.QUESTION, qcount)
1023             if self.question_only:
1024                 return self.message
1025             self._get_section(MessageSection.ANSWER, ancount)
1026             self._get_section(MessageSection.AUTHORITY, aucount)
1027             self._get_section(MessageSection.ADDITIONAL, adcount)
1028             if not self.ignore_trailing and self.parser.remaining() != 0:
1029                 raise TrailingJunk
1030             if self.multi and self.message.tsig_ctx and \
1031                 not self.message.had_tsig:
1032                 self.message.tsig_ctx.update(self.parser.wire)
1033         except Exception as e:
1034             if self.continue_on_error:
1035                 self._add_error(e)
1036             else:
1037                 raise
1038         return self.message

1041 def from_wire(wire, keyring=None, request_mac=b'', xfr=False, origin=None,
1042               tsig_ctx=None, multi=False,
1043               question_only=False, one_rr_per_rrset=False,
1044               ignore_trailing=False, raise_on_truncation=False,
1045               continue_on_error=False):
1046     """Convert a DNS wire format message into a message object. 
1047  
1048     *keyring*, a ``dns.tsig.Key`` or ``dict``, the key or keyring to use if the 
1049     message is signed. 
1050  
1051     *request_mac*, a ``bytes``.  If the message is a response to a TSIG-signed 
1052     request, *request_mac* should be set to the MAC of that request. 
1053  
1054     *xfr*, a ``bool``, should be set to ``True`` if this message is part of a 
1055     zone transfer. 
1056  
1057     *origin*, a ``dns.name.Name`` or ``None``.  If the message is part of a zone 
1058     transfer, *origin* should be the origin name of the zone.  If not ``None``, 
1059     names will be relativized to the origin. 
1060  
1061     *tsig_ctx*, a ``dns.tsig.HMACTSig`` or ``dns.tsig.GSSTSig`` object, the 
1062     ongoing TSIG context, used when validating zone transfers. 
1063  
1064     *multi*, a ``bool``, should be set to ``True`` if this message is part of a 
1065     multiple message sequence. 
1066  
1067     *question_only*, a ``bool``.  If ``True``, read only up to the end of the 
1068     question section. 
1069  
1070     *one_rr_per_rrset*, a ``bool``.  If ``True``, put each RR into its own 
1071     RRset. 
1072  
1073     *ignore_trailing*, a ``bool``.  If ``True``, ignore trailing junk at end of 
1074     the message. 
1075  
1076     *raise_on_truncation*, a ``bool``.  If ``True``, raise an exception if the 
1077     TC bit is set. 
1078  
1079     *continue_on_error*, a ``bool``.  If ``True``, try to continue parsing even 
1080     if errors occur.  Erroneous rdata will be ignored.  Errors will be 
1081     accumulated as a list of MessageError objects in the message's ``errors`` 
1082     attribute.  This option is recommended only for DNS analysis tools, or for 
1083     use in a server as part of an error handling path.  The default is 
1084     ``False``. 
1085  
1086     Raises ``dns.message.ShortHeader`` if the message is less than 12 octets 
1087     long. 
1088  
1089     Raises ``dns.message.TrailingJunk`` if there were octets in the message past 
1090     the end of the proper DNS message, and *ignore_trailing* is ``False``. 
1091  
1092     Raises ``dns.message.BadEDNS`` if an OPT record was in the wrong section, or 
1093     occurred more than once. 
1094  
1095     Raises ``dns.message.BadTSIG`` if a TSIG record was not the last record of 
1096     the additional data section. 
1097  
1098     Raises ``dns.message.Truncated`` if the TC flag is set and 
1099     *raise_on_truncation* is ``True``. 
1100  
1101     Returns a ``dns.message.Message``. 
1102     """
1103 
1104     def initialize_message(message):
1105         message.request_mac = request_mac
1106         message.xfr = xfr
1107         message.origin = origin
1108         message.tsig_ctx = tsig_ctx
1109 
1110     reader = _WireReader(wire, initialize_message, question_only,
1111                          one_rr_per_rrset, ignore_trailing, keyring, multi,
1112                          continue_on_error)
1113     try:
1114         m = reader.read()
1115     except dns.exception.FormError:
1116         if reader.message and (reader.message.flags & dns.flags.TC) and \
1117            raise_on_truncation:
1118             raise Truncated(message=reader.message)
1119         else:
1120             raise
1121     # Reading a truncated message might not have any errors, so we
1122     # have to do this check here too.
1123     if m.flags & dns.flags.TC and raise_on_truncation:
1124         raise Truncated(message=m)
1125     if continue_on_error:
1126         m.errors = reader.errors
1127 
1128     return m

1131 class _TextReader:
1132 
1133     """Text format reader. 
1134  
1135     tok: the tokenizer. 
1136     message: The message object being built. 
1137     DNS dynamic updates. 
1138     last_name: The most recently read name when building a message object. 
1139     one_rr_per_rrset: Put each RR into its own RRset? 
1140     origin: The origin for relative names 
1141     relativize: relativize names? 
1142     relativize_to: the origin to relativize to. 
1143     """
1144 
1145     def __init__(self, text, idna_codec, one_rr_per_rrset=False,
1146                  origin=None, relativize=True, relativize_to=None):
1147         self.message = None
1148         self.tok = dns.tokenizer.Tokenizer(text, idna_codec=idna_codec)
1149         self.last_name = None
1150         self.one_rr_per_rrset = one_rr_per_rrset
1151         self.origin = origin
1152         self.relativize = relativize
1153         self.relativize_to = relativize_to
1154         self.id = None
1155         self.edns = -1
1156         self.ednsflags = 0
1157         self.payload = DEFAULT_EDNS_PAYLOAD
1158         self.rcode = None
1159         self.opcode = dns.opcode.QUERY
1160         self.flags = 0
1161 
1162     def _header_line(self, _):
1163         """Process one line from the text format header section."""
1164 
1165         token = self.tok.get()
1166         what = token.value
1167         if what == 'id':
1168             self.id = self.tok.get_int()
1169         elif what == 'flags':
1170             while True:
1171                 token = self.tok.get()
1172                 if not token.is_identifier():
1173                     self.tok.unget(token)
1174                     break
1175                 self.flags = self.flags | dns.flags.from_text(token.value)
1176         elif what == 'edns':
1177             self.edns = self.tok.get_int()
1178             self.ednsflags = self.ednsflags | (self.edns << 16)
1179         elif what == 'eflags':
1180             if self.edns < 0:
1181                 self.edns = 0
1182             while True:
1183                 token = self.tok.get()
1184                 if not token.is_identifier():
1185                     self.tok.unget(token)
1186                     break
1187                 self.ednsflags = self.ednsflags | \
1188                     dns.flags.edns_from_text(token.value)
1189         elif what == 'payload':
1190             self.payload = self.tok.get_int()
1191             if self.edns < 0:
1192                 self.edns = 0
1193         elif what == 'opcode':
1194             text = self.tok.get_string()
1195             self.opcode = dns.opcode.from_text(text)
1196             self.flags = self.flags | dns.opcode.to_flags(self.opcode)
1197         elif what == 'rcode':
1198             text = self.tok.get_string()
1199             self.rcode = dns.rcode.from_text(text)
1200         else:
1201             raise UnknownHeaderField
1202         self.tok.get_eol()
1203 
1204     def _question_line(self, section_number):
1205         """Process one line from the text format question section."""
1206 
1207         section = self.message.sections[section_number]
1208         token = self.tok.get(want_leading=True)
1209         if not token.is_whitespace():
1210             self.last_name = self.tok.as_name(token, self.message.origin,
1211                                               self.relativize,
1212                                               self.relativize_to)
1213         name = self.last_name
1214         if name is None:
1215             raise NoPreviousName
1216         token = self.tok.get()
1217         if not token.is_identifier():
1218             raise dns.exception.SyntaxError
1219         # Class
1220         try:
1221             rdclass = dns.rdataclass.from_text(token.value)
1222             token = self.tok.get()
1223             if not token.is_identifier():
1224                 raise dns.exception.SyntaxError
1225         except dns.exception.SyntaxError:
1226             raise dns.exception.SyntaxError
1227         except Exception:
1228             rdclass = dns.rdataclass.IN
1229         # Type
1230         rdtype = dns.rdatatype.from_text(token.value)
1231         (rdclass, rdtype, _, _) = \
1232             self.message._parse_rr_header(section_number, name, rdclass, rdtype)
1233         self.message.find_rrset(section, name, rdclass, rdtype, create=True,
1234                                 force_unique=True)
1235         self.tok.get_eol()
1236 
1237     def _rr_line(self, section_number):
1238         """Process one line from the text format answer, authority, or 
1239         additional data sections. 
1240         """
1241 
1242         section = self.message.sections[section_number]
1243         # Name
1244         token = self.tok.get(want_leading=True)
1245         if not token.is_whitespace():
1246             self.last_name = self.tok.as_name(token, self.message.origin,
1247                                               self.relativize,
1248                                               self.relativize_to)
1249         name = self.last_name
1250         if name is None:
1251             raise NoPreviousName
1252         token = self.tok.get()
1253         if not token.is_identifier():
1254             raise dns.exception.SyntaxError
1255         # TTL
1256         try:
1257             ttl = int(token.value, 0)
1258             token = self.tok.get()
1259             if not token.is_identifier():
1260                 raise dns.exception.SyntaxError
1261         except dns.exception.SyntaxError:
1262             raise dns.exception.SyntaxError
1263         except Exception:
1264             ttl = 0
1265         # Class
1266         try:
1267             rdclass = dns.rdataclass.from_text(token.value)
1268             token = self.tok.get()
1269             if not token.is_identifier():
1270                 raise dns.exception.SyntaxError
1271         except dns.exception.SyntaxError:
1272             raise dns.exception.SyntaxError
1273         except Exception:
1274             rdclass = dns.rdataclass.IN
1275         # Type
1276         rdtype = dns.rdatatype.from_text(token.value)
1277         (rdclass, rdtype, deleting, empty) = \
1278             self.message._parse_rr_header(section_number, name, rdclass, rdtype)
1279         token = self.tok.get()
1280         if empty and not token.is_eol_or_eof():
1281             raise dns.exception.SyntaxError
1282         if not empty and token.is_eol_or_eof():
1283             raise dns.exception.UnexpectedEnd
1284         if not token.is_eol_or_eof():
1285             self.tok.unget(token)
1286             rd = dns.rdata.from_text(rdclass, rdtype, self.tok,
1287                                      self.message.origin, self.relativize,
1288                                      self.relativize_to)
1289             covers = rd.covers()
1290         else:
1291             rd = None
1292             covers = dns.rdatatype.NONE
1293         rrset = self.message.find_rrset(section, name,
1294                                         rdclass, rdtype, covers,
1295                                         deleting, True, self.one_rr_per_rrset)
1296         if rd is not None:
1297             rrset.add(rd, ttl)
1298 
1299     def _make_message(self):
1300         factory = _message_factory_from_opcode(self.opcode)
1301         message = factory(id=self.id)
1302         message.flags = self.flags
1303         if self.edns >= 0:
1304             message.use_edns(self.edns, self.ednsflags, self.payload)
1305         if self.rcode:
1306             message.set_rcode(self.rcode)
1307         if self.origin:
1308             message.origin = self.origin
1309         return message

1311     def read(self):
1312         """Read a text format DNS message and build a dns.message.Message 
1313         object."""
1314 
1315         line_method = self._header_line
1316         section_number = None
1317         while 1:
1318             token = self.tok.get(True, True)
1319             if token.is_eol_or_eof():
1320                 break
1321             if token.is_comment():
1322                 u = token.value.upper()
1323                 if u == 'HEADER':
1324                     line_method = self._header_line
1325 
1326                 if self.message:
1327                     message = self.message
1328                 else:
1329                     # If we don't have a message, create one with the current
1330                     # opcode, so that we know which section names to parse.
1331                     message = self._make_message()
1332                 try:
1333                     section_number = message._section_enum.from_text(u)
1334                     # We found a section name.  If we don't have a message,
1335                     # use the one we just created.
1336                     if not self.message:
1337                         self.message = message
1338                         self.one_rr_per_rrset = \
1339                             message._get_one_rr_per_rrset(self.one_rr_per_rrset)
1340                     if section_number == MessageSection.QUESTION:
1341                         line_method = self._question_line
1342                     else:
1343                         line_method = self._rr_line
1344                 except Exception:
1345                     # It's just a comment.
1346                     pass
1347                 self.tok.get_eol()
1348                 continue
1349             self.tok.unget(token)
1350             line_method(section_number)
1351         if not self.message:
1352             self.message = self._make_message()
1353         return self.message

1356 def from_text(text, idna_codec=None, one_rr_per_rrset=False,
1357               origin=None, relativize=True, relativize_to=None):
1358     """Convert the text format message into a message object. 
1359  
1360     The reader stops after reading the first blank line in the input to 
1361     facilitate reading multiple messages from a single file with 
1362     ``dns.message.from_file()``. 
1363  
1364     *text*, a ``str``, the text format message. 
1365  
1366     *idna_codec*, a ``dns.name.IDNACodec``, specifies the IDNA 
1367     encoder/decoder.  If ``None``, the default IDNA 2003 encoder/decoder 
1368     is used. 
1369  
1370     *one_rr_per_rrset*, a ``bool``.  If ``True``, then each RR is put 
1371     into its own rrset.  The default is ``False``. 
1372  
1373     *origin*, a ``dns.name.Name`` (or ``None``), the 
1374     origin to use for relative names. 
1375  
1376     *relativize*, a ``bool``.  If true, name will be relativized. 
1377  
1378     *relativize_to*, a ``dns.name.Name`` (or ``None``), the origin to use 
1379     when relativizing names.  If not set, the *origin* value will be used. 
1380  
1381     Raises ``dns.message.UnknownHeaderField`` if a header is unknown. 
1382  
1383     Raises ``dns.exception.SyntaxError`` if the text is badly formed. 
1384  
1385     Returns a ``dns.message.Message object`` 
1386     """
1387 
1388     # 'text' can also be a file, but we don't publish that fact
1389     # since it's an implementation detail.  The official file
1390     # interface is from_file().
1391 
1392     reader = _TextReader(text, idna_codec, one_rr_per_rrset, origin,
1393                          relativize, relativize_to)
1394     return reader.read()

1397 def from_file(f, idna_codec=None, one_rr_per_rrset=False):
1398     """Read the next text format message from the specified file. 
1399  
1400     Message blocks are separated by a single blank line. 
1401  
1402     *f*, a ``file`` or ``str``.  If *f* is text, it is treated as the 
1403     pathname of a file to open. 
1404  
1405     *idna_codec*, a ``dns.name.IDNACodec``, specifies the IDNA 
1406     encoder/decoder.  If ``None``, the default IDNA 2003 encoder/decoder 
1407     is used. 
1408  
1409     *one_rr_per_rrset*, a ``bool``.  If ``True``, then each RR is put 
1410     into its own rrset.  The default is ``False``. 
1411  
1412     Raises ``dns.message.UnknownHeaderField`` if a header is unknown. 
1413  
1414     Raises ``dns.exception.SyntaxError`` if the text is badly formed. 
1415  
1416     Returns a ``dns.message.Message object`` 
1417     """
1418 
1419     with contextlib.ExitStack() as stack:
1420         if isinstance(f, str):
1421             f = stack.enter_context(open(f))
1422         return from_text(f, idna_codec, one_rr_per_rrset)

1425 def make_query(qname, rdtype, rdclass=dns.rdataclass.IN, use_edns=None,
1426                want_dnssec=False, ednsflags=None, payload=None,
1427                request_payload=None, options=None, idna_codec=None,
1428                id=None, flags=dns.flags.RD):
1429     """Make a query message. 
1430  
1431     The query name, type, and class may all be specified either 
1432     as objects of the appropriate type, or as strings. 
1433  
1434     The query will have a randomly chosen query id, and its DNS flags 
1435     will be set to dns.flags.RD. 
1436  
1437     qname, a ``dns.name.Name`` or ``str``, the query name. 
1438  
1439     *rdtype*, an ``int`` or ``str``, the desired rdata type. 
1440  
1441     *rdclass*, an ``int`` or ``str``,  the desired rdata class; the default 
1442     is class IN. 
1443  
1444     *use_edns*, an ``int``, ``bool`` or ``None``.  The EDNS level to use; the 
1445     default is ``None``.  If ``None``, EDNS will be enabled only if other 
1446     parameters (*ednsflags*, *payload*, *request_payload*, or *options*) are 
1447     set. 
1448     See the description of dns.message.Message.use_edns() for the possible 
1449     values for use_edns and their meanings. 
1450  
1451     *want_dnssec*, a ``bool``.  If ``True``, DNSSEC data is desired. 
1452  
1453     *ednsflags*, an ``int``, the EDNS flag values. 
1454  
1455     *payload*, an ``int``, is the EDNS sender's payload field, which is the 
1456     maximum size of UDP datagram the sender can handle.  I.e. how big 
1457     a response to this message can be. 
1458  
1459     *request_payload*, an ``int``, is the EDNS payload size to use when 
1460     sending this message.  If not specified, defaults to the value of 
1461     *payload*. 
1462  
1463     *options*, a list of ``dns.edns.Option`` objects or ``None``, the EDNS 
1464     options. 
1465  
1466     *idna_codec*, a ``dns.name.IDNACodec``, specifies the IDNA 
1467     encoder/decoder.  If ``None``, the default IDNA 2003 encoder/decoder 
1468     is used. 
1469  
1470     *id*, an ``int`` or ``None``, the desired query id.  The default is 
1471     ``None``, which generates a random query id. 
1472  
1473     *flags*, an ``int``, the desired query flags.  The default is 
1474     ``dns.flags.RD``. 
1475  
1476     Returns a ``dns.message.QueryMessage`` 
1477     """
1478 
1479     if isinstance(qname, str):
1480         qname = dns.name.from_text(qname, idna_codec=idna_codec)
1481     rdtype = dns.rdatatype.RdataType.make(rdtype)
1482     rdclass = dns.rdataclass.RdataClass.make(rdclass)
1483     m = QueryMessage(id=id)
1484     m.flags = dns.flags.Flag(flags)
1485     m.find_rrset(m.question, qname, rdclass, rdtype, create=True,
1486                  force_unique=True)
1487     # only pass keywords on to use_edns if they have been set to a
1488     # non-None value.  Setting a field will turn EDNS on if it hasn't
1489     # been configured.
1490     kwargs = {}
1491     if ednsflags is not None:
1492         kwargs['ednsflags'] = ednsflags
1493     if payload is not None:
1494         kwargs['payload'] = payload
1495     if request_payload is not None:
1496         kwargs['request_payload'] = request_payload
1497     if options is not None:
1498         kwargs['options'] = options
1499     if kwargs and use_edns is None:
1500         use_edns = 0
1501     kwargs['edns'] = use_edns
1502     m.use_edns(**kwargs)
1503     m.want_dnssec(want_dnssec)
1504     return m

1507 def make_response(query, recursion_available=False, our_payload=8192,
1508                   fudge=300, tsig_error=0):
1509     """Make a message which is a response for the specified query. 
1510     The message returned is really a response skeleton; it has all 
1511     of the infrastructure required of a response, but none of the 
1512     content. 
1513  
1514     The response's question section is a shallow copy of the query's 
1515     question section, so the query's question RRsets should not be 
1516     changed. 
1517  
1518     *query*, a ``dns.message.Message``, the query to respond to. 
1519  
1520     *recursion_available*, a ``bool``, should RA be set in the response? 
1521  
1522     *our_payload*, an ``int``, the payload size to advertise in EDNS 
1523     responses. 
1524  
1525     *fudge*, an ``int``, the TSIG time fudge. 
1526  
1527     *tsig_error*, an ``int``, the TSIG error. 
1528  
1529     Returns a ``dns.message.Message`` object whose specific class is 
1530     appropriate for the query.  For example, if query is a 
1531     ``dns.update.UpdateMessage``, response will be too. 
1532     """
1533 
1534     if query.flags & dns.flags.QR:
1535         raise dns.exception.FormError('specified query message is not a query')
1536     factory = _message_factory_from_opcode(query.opcode())
1537     response = factory(id=query.id)
1538     response.flags = dns.flags.QR | (query.flags & dns.flags.RD)
1539     if recursion_available:
1540         response.flags |= dns.flags.RA
1541     response.set_opcode(query.opcode())
1542     response.question = list(query.question)
1543     if query.edns >= 0:
1544         response.use_edns(0, 0, our_payload, query.payload)
1545     if query.had_tsig:
1546         response.use_tsig(query.keyring, query.keyname, fudge, None,
1547                           tsig_error, b'', query.keyalgorithm)
1548         response.request_mac = query.mac
1549     return response
