17   @immutable
18   class Dict(collections.abc.Mapping):
19       def __init__(self, dictionary, no_copy=False):
20           """Make an immutable dictionary from the specified dictionary. 
21    
22           If *no_copy* is `True`, then *dictionary* will be wrapped instead 
23           of copied.  Only set this if you are sure there will be no external 
24           references to the dictionary. 
25           """
26           if no_copy and isinstance(dictionary, odict):
27               self._odict = dictionary
28           else:
29               self._odict = odict(dictionary)
30           self._hash = None
31   
32       def __getitem__(self, key):
33           return self._odict.__getitem__(key)
34   
35       def __hash__(self):  # pylint: disable=invalid-hash-returned
36           if self._hash is None:
37               h = 0
38               for key in sorted(self._odict.keys()):
39                   h ^= hash(key)
40               object.__setattr__(self, '_hash', h)
41           # this does return an int, but pylint doesn't figure that out
42           return self._hash
43   
44       def __len__(self):
45           return len(self._odict)
46   
47       def __iter__(self):
48           return iter(self._odict)
49

51   def constify(o):
52       """ 
53       Convert mutable types to immutable types. 
54       """
55       if isinstance(o, bytearray):
56           return bytes(o)
57       if isinstance(o, tuple):
58           try:
59               hash(o)
60               return o
61           except Exception:
62               return tuple(constify(elt) for elt in o)
63       if isinstance(o, list):
64           return tuple(constify(elt) for elt in o)
65       if isinstance(o, dict):
66           cdict = odict()
67           for k, v in o.items():
68               cdict[k] = constify(v)
69           return Dict(cdict, True)
70       return o
