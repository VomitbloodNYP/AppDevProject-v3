37   class Resolver(dns.resolver.BaseResolver):
38       """Asynchronous DNS stub resolver."""

40       async def resolve(self, qname, rdtype=dns.rdatatype.A,
41                         rdclass=dns.rdataclass.IN,
42                         tcp=False, source=None, raise_on_no_answer=True,
43                         source_port=0, lifetime=None, search=None,
44                         backend=None):
45           """Query nameservers asynchronously to find the answer to the question. 
46    
47           *backend*, a ``dns.asyncbackend.Backend``, or ``None``.  If ``None``, 
48           the default, then dnspython will use the default backend. 
49    
50           See :py:func:`dns.resolver.Resolver.resolve()` for the 
51           documentation of the other parameters, exceptions, and return 
52           type of this method. 
53           """
54   
55           resolution = dns.resolver._Resolution(self, qname, rdtype, rdclass, tcp,
56                                                 raise_on_no_answer, search)
57           if not backend:
58               backend = dns.asyncbackend.get_default_backend()
59           start = time.time()
60           while True:
61               (request, answer) = resolution.next_request()
62               # Note we need to say "if answer is not None" and not just
63               # "if answer" because answer implements __len__, and python
64               # will call that.  We want to return if we have an answer
65               # object, including in cases where its length is 0.
66               if answer is not None:
67                   # cache hit!
68                   return answer
69               done = False
70               while not done:
71                   (nameserver, port, tcp, backoff) = resolution.next_nameserver()
72                   if backoff:
73                       await backend.sleep(backoff)
74                   timeout = self._compute_timeout(start, lifetime,
75                                                   resolution.errors)
76                   try:
77                       if dns.inet.is_address(nameserver):
78                           if tcp:
79                               response = await _tcp(request, nameserver,
80                                                     timeout, port,
81                                                     source, source_port,
82                                                     backend=backend)
83                           else:
84                               response = await _udp(request, nameserver,
85                                                     timeout, port,
86                                                     source, source_port,
87                                                     raise_on_truncation=True,
88                                                     backend=backend)
89                       else:
90                           response = await dns.asyncquery.https(request,
91                                                                 nameserver,
92                                                                 timeout=timeout)
93                   except Exception as ex:
94                       (_, done) = resolution.query_result(None, ex)
95                       continue
96                   (answer, done) = resolution.query_result(response, None)
97                   # Note we need to say "if answer is not None" and not just
98                   # "if answer" because answer implements __len__, and python
99                   # will call that.  We want to return if we have an answer
100                  # object, including in cases where its length is 0.
101                  if answer is not None:
102                      return answer

104      async def resolve_address(self, ipaddr, *args, **kwargs):
105          """Use an asynchronous resolver to run a reverse query for PTR 
106          records. 
107   
108          This utilizes the resolve() method to perform a PTR lookup on the 
109          specified IP address. 
110   
111          *ipaddr*, a ``str``, the IPv4 or IPv6 address you want to get 
112          the PTR record for. 
113   
114          All other arguments that can be passed to the resolve() function 
115          except for rdtype and rdclass are also supported by this 
116          function. 
117   
118          """
119  
120          return await self.resolve(dns.reversename.from_address(ipaddr),
121                                    rdtype=dns.rdatatype.PTR,
122                                    rdclass=dns.rdataclass.IN,
123                                    *args, **kwargs)

127      async def canonical_name(self, name):
128          """Determine the canonical name of *name*. 
129   
130          The canonical name is the name the resolver uses for queries 
131          after all CNAME and DNAME renamings have been applied. 
132   
133          *name*, a ``dns.name.Name`` or ``str``, the query name. 
134   
135          This method can raise any exception that ``resolve()`` can 
136          raise, other than ``dns.resolver.NoAnswer`` and 
137          ``dns.resolver.NXDOMAIN``. 
138   
139          Returns a ``dns.name.Name``. 
140          """
141          try:
142              answer = await self.resolve(name, raise_on_no_answer=False)
143              canonical_name = answer.canonical_name
144          except dns.resolver.NXDOMAIN as e:
145              canonical_name = e.canonical_name
146          return canonical_name

149  default_resolver = None

152  def get_default_resolver():
153      """Get the default asynchronous resolver, initializing it if necessary."""
154      if default_resolver is None:
155          reset_default_resolver()
156      return default_resolver
157  
158  
159  def reset_default_resolver():
160      """Re-initialize default asynchronous resolver. 
161   
162      Note that the resolver configuration (i.e. /etc/resolv.conf on UNIX 
163      systems) will be re-read immediately. 
164      """
165  
166      global default_resolver
167      default_resolver = Resolver()

170  async def resolve(qname, rdtype=dns.rdatatype.A, rdclass=dns.rdataclass.IN,
171                    tcp=False, source=None, raise_on_no_answer=True,
172                    source_port=0, lifetime=None, search=None, backend=None):
173      """Query nameservers asynchronously to find the answer to the question. 
174   
175      This is a convenience function that uses the default resolver 
176      object to make the query. 
177   
178      See :py:func:`dns.asyncresolver.Resolver.resolve` for more 
179      information on the parameters. 
180      """
181  
182      return await get_default_resolver().resolve(qname, rdtype, rdclass, tcp,
183                                                  source, raise_on_no_answer,
184                                                  source_port, lifetime, search,
185                                                  backend)

206  async def zone_for_name(name, rdclass=dns.rdataclass.IN, tcp=False,
207                          resolver=None, backend=None):
208      """Find the name of the zone which contains the specified name. 
209   
210      See :py:func:`dns.resolver.Resolver.zone_for_name` for more 
211      information on the parameters and possible exceptions. 
212      """
213  
214      if isinstance(name, str):
215          name = dns.name.from_text(name, dns.name.root)
216      if resolver is None:
217          resolver = get_default_resolver()
218      if not name.is_absolute():
219          raise NotAbsolute(name)
220      while True:
221          try:
222              answer = await resolver.resolve(name, dns.rdatatype.SOA, rdclass,
223                                              tcp, backend=backend)
224              if answer.rrset.name == name:
225                  return name
226              # otherwise we were CNAMEd or DNAMEd and need to look higher
227          except (NXDOMAIN, NoAnswer):
228              pass
229          try:
230              name = name.parent()
231          except dns.name.NoParent:  # pragma: no cover
232              raise NoRootSOA
