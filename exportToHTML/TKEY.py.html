18   import base64
19   import struct
20   
21   import dns.dnssec
22   import dns.immutable
23   import dns.exception
24   import dns.rdata
25   
26   
27   @dns.immutable.immutable
28   class TKEY(dns.rdata.Rdata):
29   
30       """TKEY Record"""
31   
32       __slots__ = ['algorithm', 'inception', 'expiration', 'mode', 'error',
33                    'key', 'other']
34   
35       def __init__(self, rdclass, rdtype, algorithm, inception, expiration,
36                    mode, error, key, other=b''):
37           super().__init__(rdclass, rdtype)
38           self.algorithm = self._as_name(algorithm)
39           self.inception = self._as_uint32(inception)
40           self.expiration = self._as_uint32(expiration)
41           self.mode = self._as_uint16(mode)
42           self.error = self._as_uint16(error)
43           self.key = self._as_bytes(key)
44           self.other = self._as_bytes(other)
45   
46       def to_text(self, origin=None, relativize=True, **kw):
47           _algorithm = self.algorithm.choose_relativity(origin, relativize)
48           text = '%s %u %u %u %u %s' % (str(_algorithm), self.inception,
49                                         self.expiration, self.mode, self.error,
50                                         dns.rdata._base64ify(self.key, 0))
51           if len(self.other) > 0:
52               text += ' %s' % (dns.rdata._base64ify(self.other, 0))
53   
54           return text
55   
56       @classmethod
57       def from_text(cls, rdclass, rdtype, tok, origin=None, relativize=True,
58                     relativize_to=None):
59           algorithm = tok.get_name(relativize=False)
60           inception = tok.get_uint32()
61           expiration = tok.get_uint32()
62           mode = tok.get_uint16()
63           error = tok.get_uint16()
64           key_b64 = tok.get_string().encode()
65           key = base64.b64decode(key_b64)
66           other_b64 = tok.concatenate_remaining_identifiers(True).encode()
67           other = base64.b64decode(other_b64)
68   
69           return cls(rdclass, rdtype, algorithm, inception, expiration, mode,
70                      error, key, other)
71   
72       def _to_wire(self, file, compress=None, origin=None, canonicalize=False):
73           self.algorithm.to_wire(file, compress, origin)
74           file.write(struct.pack("!IIHH", self.inception, self.expiration,
75                                  self.mode, self.error))
76           file.write(struct.pack("!H", len(self.key)))
77           file.write(self.key)
78           file.write(struct.pack("!H", len(self.other)))
79           if len(self.other) > 0:
80               file.write(self.other)
81   
82       @classmethod
83       def from_wire_parser(cls, rdclass, rdtype, parser, origin=None):
84           algorithm = parser.get_name(origin)
85           inception, expiration, mode, error = parser.get_struct("!IIHH")
86           key = parser.get_counted_bytes(2)
87           other = parser.get_counted_bytes(2)
88   
89           return cls(rdclass, rdtype, algorithm, inception, expiration, mode,
90                      error, key, other)
91   
92       # Constants for the mode field - from RFC 2930:
93       # 2.5 The Mode Field
94       #
95       #    The mode field specifies the general scheme for key agreement or
96       #    the purpose of the TKEY DNS message.  Servers and resolvers
97       #    supporting this specification MUST implement the Diffie-Hellman key
98       #    agreement mode and the key deletion mode for queries.  All other
99       #    modes are OPTIONAL.  A server supporting TKEY that receives a TKEY
100      #    request with a mode it does not support returns the BADMODE error.
101      #    The following values of the Mode octet are defined, available, or
102      #    reserved:
103      #
104      #          Value    Description
105      #          -----    -----------
106      #           0        - reserved, see section 7
107      #           1       server assignment
108      #           2       Diffie-Hellman exchange
109      #           3       GSS-API negotiation
110      #           4       resolver assignment
111      #           5       key deletion
112      #          6-65534   - available, see section 7
113      #          65535     - reserved, see section 7
114      SERVER_ASSIGNMENT = 1
115      DIFFIE_HELLMAN_EXCHANGE = 2
116      GSSAPI_NEGOTIATION = 3
117      RESOLVER_ASSIGNMENT = 4
118      KEY_DELETION = 5
