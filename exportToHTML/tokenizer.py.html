39   class UngetBufferFull(dns.exception.DNSException):
40       """An attempt was made to unget a token when the unget buffer was full."""

43   class Token:
44       """A DNS zone file format token. 
45    
46       ttype: The token type 
47       value: The token value 
48       has_escape: Does the token value contain escapes? 
49       """
50   
51       def __init__(self, ttype, value='', has_escape=False, comment=None):
52           """Initialize a token instance."""
53   
54           self.ttype = ttype
55           self.value = value
56           self.has_escape = has_escape
57           self.comment = comment
58   
59       def is_eof(self):
60           return self.ttype == EOF
61   
62       def is_eol(self):
63           return self.ttype == EOL
64   
65       def is_whitespace(self):
66           return self.ttype == WHITESPACE
67   
68       def is_identifier(self):
69           return self.ttype == IDENTIFIER
70   
71       def is_quoted_string(self):
72           return self.ttype == QUOTED_STRING
73   
74       def is_comment(self):
75           return self.ttype == COMMENT
76   
77       def is_delimiter(self):  # pragma: no cover (we don't return delimiters yet)
78           return self.ttype == DELIMITER
79   
80       def is_eol_or_eof(self):
81           return self.ttype == EOL or self.ttype == EOF
82   
83       def __eq__(self, other):
84           if not isinstance(other, Token):
85               return False
86           return (self.ttype == other.ttype and
87                   self.value == other.value)
88   
89       def __ne__(self, other):
90           if not isinstance(other, Token):
91               return True
92           return (self.ttype != other.ttype or
93                   self.value != other.value)
94   
95       def __str__(self):
96           return '%d "%s"' % (self.ttype, self.value)
97   
98       def unescape(self):
99           if not self.has_escape:
100              return self
101          unescaped = ''
102          l = len(self.value)
103          i = 0
104          while i < l:
105              c = self.value[i]
106              i += 1
107              if c == '\\':
108                  if i >= l:  # pragma: no cover   (can't happen via get())
109                      raise dns.exception.UnexpectedEnd
110                  c = self.value[i]
111                  i += 1
112                  if c.isdigit():
113                      if i >= l:
114                          raise dns.exception.UnexpectedEnd
115                      c2 = self.value[i]
116                      i += 1
117                      if i >= l:
118                          raise dns.exception.UnexpectedEnd
119                      c3 = self.value[i]
120                      i += 1
121                      if not (c2.isdigit() and c3.isdigit()):
122                          raise dns.exception.SyntaxError
123                      codepoint = int(c) * 100 + int(c2) * 10 + int(c3)
124                      if codepoint > 255:
125                          raise dns.exception.SyntaxError
126                      c = chr(codepoint)
127              unescaped += c
128          return Token(self.ttype, unescaped)
129  
130      def unescape_to_bytes(self):
131          # We used to use unescape() for TXT-like records, but this
132          # caused problems as we'd process DNS escapes into Unicode code
133          # points instead of byte values, and then a to_text() of the
134          # processed data would not equal the original input.  For
135          # example, \226 in the TXT record would have a to_text() of
136          # \195\162 because we applied UTF-8 encoding to Unicode code
137          # point 226.
138          #
139          # We now apply escapes while converting directly to bytes,
140          # avoiding this double encoding.
141          #
142          # This code also handles cases where the unicode input has
143          # non-ASCII code-points in it by converting it to UTF-8.  TXT
144          # records aren't defined for Unicode, but this is the best we
145          # can do to preserve meaning.  For example,
146          #
147          #     foo\u200bbar
148          #
149          # (where \u200b is Unicode code point 0x200b) will be treated
150          # as if the input had been the UTF-8 encoding of that string,
151          # namely:
152          #
153          #     foo\226\128\139bar
154          #
155          unescaped = b''
156          l = len(self.value)
157          i = 0
158          while i < l:
159              c = self.value[i]
160              i += 1
161              if c == '\\':
162                  if i >= l:  # pragma: no cover   (can't happen via get())
163                      raise dns.exception.UnexpectedEnd
164                  c = self.value[i]
165                  i += 1
166                  if c.isdigit():
167                      if i >= l:
168                          raise dns.exception.UnexpectedEnd
169                      c2 = self.value[i]
170                      i += 1
171                      if i >= l:
172                          raise dns.exception.UnexpectedEnd
173                      c3 = self.value[i]
174                      i += 1
175                      if not (c2.isdigit() and c3.isdigit()):
176                          raise dns.exception.SyntaxError
177                      codepoint = int(c) * 100 + int(c2) * 10 + int(c3)
178                      if codepoint > 255:
179                          raise dns.exception.SyntaxError
180                      unescaped += b'%c' % (codepoint)
181                  else:
182                      # Note that as mentioned above, if c is a Unicode
183                      # code point outside of the ASCII range, then this
184                      # += is converting that code point to its UTF-8
185                      # encoding and appending multiple bytes to
186                      # unescaped.
187                      unescaped += c.encode()
188              else:
189                  unescaped += c.encode()
190          return Token(self.ttype, bytes(unescaped))

193  class Tokenizer:
194      """A DNS zone file format tokenizer. 
195   
196      A token object is basically a (type, value) tuple.  The valid 
197      types are EOF, EOL, WHITESPACE, IDENTIFIER, QUOTED_STRING, 
198      COMMENT, and DELIMITER. 
199   
200      file: The file to tokenize 
201   
202      ungotten_char: The most recently ungotten character, or None. 
203   
204      ungotten_token: The most recently ungotten token, or None. 
205   
206      multiline: The current multiline level.  This value is increased 
207      by one every time a '(' delimiter is read, and decreased by one every time 
208      a ')' delimiter is read. 
209   
210      quoting: This variable is true if the tokenizer is currently 
211      reading a quoted string. 
212   
213      eof: This variable is true if the tokenizer has encountered EOF. 
214   
215      delimiters: The current delimiter dictionary. 
216   
217      line_number: The current line number 
218   
219      filename: A filename that will be returned by the where() method. 
220   
221      idna_codec: A dns.name.IDNACodec, specifies the IDNA 
222      encoder/decoder.  If None, the default IDNA 2003 
223      encoder/decoder is used. 
224      """
225  
226      def __init__(self, f=sys.stdin, filename=None, idna_codec=None):
227          """Initialize a tokenizer instance. 
228   
229          f: The file to tokenize.  The default is sys.stdin. 
230          This parameter may also be a string, in which case the tokenizer 
231          will take its input from the contents of the string. 
232   
233          filename: the name of the filename that the where() method 
234          will return. 
235   
236          idna_codec: A dns.name.IDNACodec, specifies the IDNA 
237          encoder/decoder.  If None, the default IDNA 2003 
238          encoder/decoder is used. 
239          """
240  
241          if isinstance(f, str):
242              f = io.StringIO(f)
243              if filename is None:
244                  filename = '<string>'
245          elif isinstance(f, bytes):
246              f = io.StringIO(f.decode())
247              if filename is None:
248                  filename = '<string>'
249          else:
250              if filename is None:
251                  if f is sys.stdin:
252                      filename = '<stdin>'
253                  else:
254                      filename = '<file>'
255          self.file = f
256          self.ungotten_char = None
257          self.ungotten_token = None
258          self.multiline = 0
259          self.quoting = False
260          self.eof = False
261          self.delimiters = _DELIMITERS
262          self.line_number = 1
263          self.filename = filename
264          if idna_codec is None:
265              idna_codec = dns.name.IDNA_2003
266          self.idna_codec = idna_codec
267  
268      def _get_char(self):
269          """Read a character from input. 
270          """
271  
272          if self.ungotten_char is None:
273              if self.eof:
274                  c = ''
275              else:
276                  c = self.file.read(1)
277                  if c == '':
278                      self.eof = True
279                  elif c == '\n':
280                      self.line_number += 1
281          else:
282              c = self.ungotten_char
283              self.ungotten_char = None
284          return c
285  
286      def where(self):
287          """Return the current location in the input. 
288   
289          Returns a (string, int) tuple.  The first item is the filename of 
290          the input, the second is the current line number. 
291          """
292  
293          return (self.filename, self.line_number)
294  
295      def _unget_char(self, c):
296          """Unget a character. 
297   
298          The unget buffer for characters is only one character large; it is 
299          an error to try to unget a character when the unget buffer is not 
300          empty. 
301   
302          c: the character to unget 
303          raises UngetBufferFull: there is already an ungotten char 
304          """
305  
306          if self.ungotten_char is not None:
307              # this should never happen!
308              raise UngetBufferFull  # pragma: no cover
309          self.ungotten_char = c
310  
311      def skip_whitespace(self):
312          """Consume input until a non-whitespace character is encountered. 
313   
314          The non-whitespace character is then ungotten, and the number of 
315          whitespace characters consumed is returned. 
316   
317          If the tokenizer is in multiline mode, then newlines are whitespace. 
318   
319          Returns the number of characters skipped. 
320          """
321  
322          skipped = 0
323          while True:
324              c = self._get_char()
325              if c != ' ' and c != '\t':
326                  if (c != '\n') or not self.multiline:
327                      self._unget_char(c)
328                      return skipped
329              skipped += 1
330  
331      def get(self, want_leading=False, want_comment=False):
332          """Get the next token. 
333   
334          want_leading: If True, return a WHITESPACE token if the 
335          first character read is whitespace.  The default is False. 
336   
337          want_comment: If True, return a COMMENT token if the 
338          first token read is a comment.  The default is False. 
339   
340          Raises dns.exception.UnexpectedEnd: input ended prematurely 
341   
342          Raises dns.exception.SyntaxError: input was badly formed 
343   
344          Returns a Token. 
345          """
346  
347          if self.ungotten_token is not None:
348              token = self.ungotten_token
349              self.ungotten_token = None
350              if token.is_whitespace():
351                  if want_leading:
352                      return token
353              elif token.is_comment():
354                  if want_comment:
355                      return token
356              else:
357                  return token
358          skipped = self.skip_whitespace()
359          if want_leading and skipped > 0:
360              return Token(WHITESPACE, ' ')
361          token = ''
362          ttype = IDENTIFIER
363          has_escape = False
364          while True:
365              c = self._get_char()
366              if c == '' or c in self.delimiters:
367                  if c == '' and self.quoting:
368                      raise dns.exception.UnexpectedEnd
369                  if token == '' and ttype != QUOTED_STRING:
370                      if c == '(':
371                          self.multiline += 1
372                          self.skip_whitespace()
373                          continue
374                      elif c == ')':
375                          if self.multiline <= 0:
376                              raise dns.exception.SyntaxError
377                          self.multiline -= 1
378                          self.skip_whitespace()
379                          continue
380                      elif c == '"':
381                          if not self.quoting:
382                              self.quoting = True
383                              self.delimiters = _QUOTING_DELIMITERS
384                              ttype = QUOTED_STRING
385                              continue
386                          else:
387                              self.quoting = False
388                              self.delimiters = _DELIMITERS
389                              self.skip_whitespace()
390                              continue
391                      elif c == '\n':
392                          return Token(EOL, '\n')
393                      elif c == ';':
394                          while 1:
395                              c = self._get_char()
396                              if c == '\n' or c == '':
397                                  break
398                              token += c
399                          if want_comment:
400                              self._unget_char(c)
401                              return Token(COMMENT, token)
402                          elif c == '':
403                              if self.multiline:
404                                  raise dns.exception.SyntaxError(
405                                      'unbalanced parentheses')
406                              return Token(EOF, comment=token)
407                          elif self.multiline:
408                              self.skip_whitespace()
409                              token = ''
410                              continue
411                          else:
412                              return Token(EOL, '\n', comment=token)
413                      else:
414                          # This code exists in case we ever want a
415                          # delimiter to be returned.  It never produces
416                          # a token currently.
417                          token = c
418                          ttype = DELIMITER
419                  else:
420                      self._unget_char(c)
421                  break
422              elif self.quoting and c == '\n':
423                  raise dns.exception.SyntaxError('newline in quoted string')
424              elif c == '\\':
425                  #
426                  # It's an escape.  Put it and the next character into
427                  # the token; it will be checked later for goodness.
428                  #
429                  token += c
430                  has_escape = True
431                  c = self._get_char()
432                  if c == '' or (c == '\n' and not self.quoting):
433                      raise dns.exception.UnexpectedEnd
434              token += c
435          if token == '' and ttype != QUOTED_STRING:
436              if self.multiline:
437                  raise dns.exception.SyntaxError('unbalanced parentheses')
438              ttype = EOF
439          return Token(ttype, token, has_escape)
440  
441      def unget(self, token):
442          """Unget a token. 
443   
444          The unget buffer for tokens is only one token large; it is 
445          an error to try to unget a token when the unget buffer is not 
446          empty. 
447   
448          token: the token to unget 
449   
450          Raises UngetBufferFull: there is already an ungotten token 
451          """
452  
453          if self.ungotten_token is not None:
454              raise UngetBufferFull
455          self.ungotten_token = token
456  
457      def next(self):
458          """Return the next item in an iteration. 
459   
460          Returns a Token. 
461          """
462  
463          token = self.get()
464          if token.is_eof():
465              raise StopIteration
466          return token
467  
468      __next__ = next
469  
470      def __iter__(self):
471          return self

475      def get_int(self, base=10):
476          """Read the next token and interpret it as an unsigned integer. 
477   
478          Raises dns.exception.SyntaxError if not an unsigned integer. 
479   
480          Returns an int. 
481          """
482  
483          token = self.get().unescape()
484          if not token.is_identifier():
485              raise dns.exception.SyntaxError('expecting an identifier')
486          if not token.value.isdigit():
487              raise dns.exception.SyntaxError('expecting an integer')
488          return int(token.value, base)
489  
490      def get_uint8(self):
491          """Read the next token and interpret it as an 8-bit unsigned 
492          integer. 
493   
494          Raises dns.exception.SyntaxError if not an 8-bit unsigned integer. 
495   
496          Returns an int. 
497          """
498  
499          value = self.get_int()
500          if value < 0 or value > 255:
501              raise dns.exception.SyntaxError(
502                  '%d is not an unsigned 8-bit integer' % value)
503          return value
504  
505      def get_uint16(self, base=10):
506          """Read the next token and interpret it as a 16-bit unsigned 
507          integer. 
508   
509          Raises dns.exception.SyntaxError if not a 16-bit unsigned integer. 
510   
511          Returns an int. 
512          """
513  
514          value = self.get_int(base=base)
515          if value < 0 or value > 65535:
516              if base == 8:
517                  raise dns.exception.SyntaxError(
518                      '%o is not an octal unsigned 16-bit integer' % value)
519              else:
520                  raise dns.exception.SyntaxError(
521                      '%d is not an unsigned 16-bit integer' % value)
522          return value
523  
524      def get_uint32(self, base=10):
525          """Read the next token and interpret it as a 32-bit unsigned 
526          integer. 
527   
528          Raises dns.exception.SyntaxError if not a 32-bit unsigned integer. 
529   
530          Returns an int. 
531          """
532  
533          value = self.get_int(base=base)
534          if value < 0 or value > 4294967295:
535              raise dns.exception.SyntaxError(
536                  '%d is not an unsigned 32-bit integer' % value)
537          return value
538  
539      def get_uint48(self, base=10):
540          """Read the next token and interpret it as a 48-bit unsigned 
541          integer. 
542   
543          Raises dns.exception.SyntaxError if not a 48-bit unsigned integer. 
544   
545          Returns an int. 
546          """
547  
548          value = self.get_int(base=base)
549          if value < 0 or value > 281474976710655:
550              raise dns.exception.SyntaxError(
551                  '%d is not an unsigned 48-bit integer' % value)
552          return value

554      def get_string(self, max_length=None):
555          """Read the next token and interpret it as a string. 
556   
557          Raises dns.exception.SyntaxError if not a string. 
558          Raises dns.exception.SyntaxError if token value length 
559          exceeds max_length (if specified). 
560   
561          Returns a string. 
562          """
563  
564          token = self.get().unescape()
565          if not (token.is_identifier() or token.is_quoted_string()):
566              raise dns.exception.SyntaxError('expecting a string')
567          if max_length and len(token.value) > max_length:
568              raise dns.exception.SyntaxError("string too long")
569          return token.value
570  
571      def get_identifier(self):
572          """Read the next token, which should be an identifier. 
573   
574          Raises dns.exception.SyntaxError if not an identifier. 
575   
576          Returns a string. 
577          """
578  
579          token = self.get().unescape()
580          if not token.is_identifier():
581              raise dns.exception.SyntaxError('expecting an identifier')
582          return token.value
583  
584      def get_remaining(self, max_tokens=None):
585          """Return the remaining tokens on the line, until an EOL or EOF is seen. 
586   
587          max_tokens: If not None, stop after this number of tokens. 
588   
589          Returns a list of tokens. 
590          """
591  
592          tokens = []
593          while True:
594              token = self.get()
595              if token.is_eol_or_eof():
596                  self.unget(token)
597                  break
598              tokens.append(token)
599              if len(tokens) == max_tokens:
600                  break
601          return tokens
602  
603      def concatenate_remaining_identifiers(self, allow_empty=False):
604          """Read the remaining tokens on the line, which should be identifiers. 
605   
606          Raises dns.exception.SyntaxError if there are no remaining tokens, 
607          unless `allow_empty=True` is given. 
608   
609          Raises dns.exception.SyntaxError if a token is seen that is not an 
610          identifier. 
611   
612          Returns a string containing a concatenation of the remaining 
613          identifiers. 
614          """
615          s = ""
616          while True:
617              token = self.get().unescape()
618              if token.is_eol_or_eof():
619                  self.unget(token)
620                  break
621              if not token.is_identifier():
622                  raise dns.exception.SyntaxError
623              s += token.value
624          if not (allow_empty or s):
625              raise dns.exception.SyntaxError('expecting another identifier')
626          return s

628      def as_name(self, token, origin=None, relativize=False, relativize_to=None):
629          """Try to interpret the token as a DNS name. 
630   
631          Raises dns.exception.SyntaxError if not a name. 
632   
633          Returns a dns.name.Name. 
634          """
635          if not token.is_identifier():
636              raise dns.exception.SyntaxError('expecting an identifier')
637          name = dns.name.from_text(token.value, origin, self.idna_codec)
638          return name.choose_relativity(relativize_to or origin, relativize)
639  
640      def get_name(self, origin=None, relativize=False, relativize_to=None):
641          """Read the next token and interpret it as a DNS name. 
642   
643          Raises dns.exception.SyntaxError if not a name. 
644   
645          Returns a dns.name.Name. 
646          """
647  
648          token = self.get()
649          return self.as_name(token, origin, relativize, relativize_to)

651      def get_eol_as_token(self):
652          """Read the next token and raise an exception if it isn't EOL or 
653          EOF. 
654   
655          Returns a string. 
656          """
657  
658          token = self.get()
659          if not token.is_eol_or_eof():
660              raise dns.exception.SyntaxError(
661                  'expected EOL or EOF, got %d "%s"' % (token.ttype,
662                                                        token.value))
663          return token
664  
665      def get_eol(self):
666          return self.get_eol_as_token().value

668      def get_ttl(self):
669          """Read the next token and interpret it as a DNS TTL. 
670   
671          Raises dns.exception.SyntaxError or dns.ttl.BadTTL if not an 
672          identifier or badly formed. 
673   
674          Returns an int. 
675          """
676  
677          token = self.get().unescape()
678          if not token.is_identifier():
679              raise dns.exception.SyntaxError('expecting an identifier')
680          return dns.ttl.from_text(token.value)
