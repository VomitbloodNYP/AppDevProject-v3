18   """DNS E.164 helpers."""
19   
20   import dns.exception
21   import dns.name
22   import dns.resolver
23   
24   #: The public E.164 domain.
25   public_enum_domain = dns.name.from_text('e164.arpa.')
26   
27   
28   def from_e164(text, origin=public_enum_domain):
29       """Convert an E.164 number in textual form into a Name object whose 
30       value is the ENUM domain name for that number. 
31    
32       Non-digits in the text are ignored, i.e. "16505551212", 
33       "+1.650.555.1212" and "1 (650) 555-1212" are all the same. 
34    
35       *text*, a ``str``, is an E.164 number in textual form. 
36    
37       *origin*, a ``dns.name.Name``, the domain in which the number 
38       should be constructed.  The default is ``e164.arpa.``. 
39    
40       Returns a ``dns.name.Name``. 
41       """
42   
43       parts = [d for d in text if d.isdigit()]
44       parts.reverse()
45       return dns.name.from_text('.'.join(parts), origin=origin)
46

48   def to_e164(name, origin=public_enum_domain, want_plus_prefix=True):
49       """Convert an ENUM domain name into an E.164 number. 
50    
51       Note that dnspython does not have any information about preferred 
52       number formats within national numbering plans, so all numbers are 
53       emitted as a simple string of digits, prefixed by a '+' (unless 
54       *want_plus_prefix* is ``False``). 
55    
56       *name* is a ``dns.name.Name``, the ENUM domain name. 
57    
58       *origin* is a ``dns.name.Name``, a domain containing the ENUM 
59       domain name.  The name is relativized to this domain before being 
60       converted to text.  If ``None``, no relativization is done. 
61    
62       *want_plus_prefix* is a ``bool``.  If True, add a '+' to the beginning of 
63       the returned number. 
64    
65       Returns a ``str``. 
66    
67       """
68       if origin is not None:
69           name = name.relativize(origin)
70       dlabels = [d for d in name.labels if d.isdigit() and len(d) == 1]
71       if len(dlabels) != len(name.labels):
72           raise dns.exception.SyntaxError('non-digit labels in ENUM domain name')
73       dlabels.reverse()
74       text = b''.join(dlabels)
75       if want_plus_prefix:
76           text = b'+' + text
77       return text.decode()

80   def query(number, domains, resolver=None):
81       """Look for NAPTR RRs for the specified number in the specified domains. 
82    
83       e.g. lookup('16505551212', ['e164.dnspython.org.', 'e164.arpa.']) 
84    
85       *number*, a ``str`` is the number to look for. 
86    
87       *domains* is an iterable containing ``dns.name.Name`` values. 
88    
89       *resolver*, a ``dns.resolver.Resolver``, is the resolver to use.  If 
90       ``None``, the default resolver is used. 
91       """
92   
93       if resolver is None:
94           resolver = dns.resolver.get_default_resolver()
95       e_nx = dns.resolver.NXDOMAIN()
96       for domain in domains:
97           if isinstance(domain, str):
98               domain = dns.name.from_text(domain)
99           qname = dns.e164.from_e164(number, domain)
100          try:
101              return resolver.resolve(qname, 'NAPTR')
102          except dns.resolver.NXDOMAIN as e:
103              e_nx += e
104      raise e_nx
