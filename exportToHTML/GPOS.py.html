26   def _validate_float_string(what):
27       if len(what) == 0:
28           raise dns.exception.FormError
29       if what[0] == b'-'[0] or what[0] == b'+'[0]:
30           what = what[1:]
31       if what.isdigit():
32           return
33       try:
34           (left, right) = what.split(b'.')
35       except ValueError:
36           raise dns.exception.FormError
37       if left == b'' and right == b'':
38           raise dns.exception.FormError
39       if not left == b'' and not left.decode().isdigit():
40           raise dns.exception.FormError
41       if not right == b'' and not right.decode().isdigit():
42           raise dns.exception.FormError

45   @dns.immutable.immutable
46   class GPOS(dns.rdata.Rdata):
47   
48       """GPOS record"""
49   
50       # see: RFC 1712
51   
52       __slots__ = ['latitude', 'longitude', 'altitude']
53   
54       def __init__(self, rdclass, rdtype, latitude, longitude, altitude):
55           super().__init__(rdclass, rdtype)
56           if isinstance(latitude, float) or \
57              isinstance(latitude, int):
58               latitude = str(latitude)
59           if isinstance(longitude, float) or \
60              isinstance(longitude, int):
61               longitude = str(longitude)
62           if isinstance(altitude, float) or \
63              isinstance(altitude, int):
64               altitude = str(altitude)
65           latitude = self._as_bytes(latitude, True, 255)
66           longitude = self._as_bytes(longitude, True, 255)
67           altitude = self._as_bytes(altitude, True, 255)
68           _validate_float_string(latitude)
69           _validate_float_string(longitude)
70           _validate_float_string(altitude)
71           self.latitude = latitude
72           self.longitude = longitude
73           self.altitude = altitude
74           flat = self.float_latitude
75           if flat < -90.0 or flat > 90.0:
76               raise dns.exception.FormError('bad latitude')
77           flong = self.float_longitude
78           if flong < -180.0 or flong > 180.0:
79               raise dns.exception.FormError('bad longitude')
80   
81       def to_text(self, origin=None, relativize=True, **kw):
82           return '{} {} {}'.format(self.latitude.decode(),
83                                    self.longitude.decode(),
84                                    self.altitude.decode())
85   
86       @classmethod
87       def from_text(cls, rdclass, rdtype, tok, origin=None, relativize=True,
88                     relativize_to=None):
89           latitude = tok.get_string()
90           longitude = tok.get_string()
91           altitude = tok.get_string()
92           return cls(rdclass, rdtype, latitude, longitude, altitude)
93   
94       def _to_wire(self, file, compress=None, origin=None, canonicalize=False):
95           l = len(self.latitude)
96           assert l < 256
97           file.write(struct.pack('!B', l))
98           file.write(self.latitude)
99           l = len(self.longitude)
100          assert l < 256
101          file.write(struct.pack('!B', l))
102          file.write(self.longitude)
103          l = len(self.altitude)
104          assert l < 256
105          file.write(struct.pack('!B', l))
106          file.write(self.altitude)
107  
108      @classmethod
109      def from_wire_parser(cls, rdclass, rdtype, parser, origin=None):
110          latitude = parser.get_counted_bytes()
111          longitude = parser.get_counted_bytes()
112          altitude = parser.get_counted_bytes()
113          return cls(rdclass, rdtype, latitude, longitude, altitude)
114  
115      @property
116      def float_latitude(self):
117          "latitude as a floating point value"
118          return float(self.latitude)
119  
120      @property
121      def float_longitude(self):
122          "longitude as a floating point value"
123          return float(self.longitude)
124  
125      @property
126      def float_altitude(self):
127          "altitude as a floating point value"
128          return float(self.altitude)
