30   class BadSigTime(dns.exception.DNSException):
31   
32       """Time in DNS SIG or RRSIG resource record cannot be parsed."""
33   
34   
35   def sigtime_to_posixtime(what):
36       if len(what) <= 10 and what.isdigit():
37           return int(what)
38       if len(what) != 14:
39           raise BadSigTime
40       year = int(what[0:4])
41       month = int(what[4:6])
42       day = int(what[6:8])
43       hour = int(what[8:10])
44       minute = int(what[10:12])
45       second = int(what[12:14])
46       return calendar.timegm((year, month, day, hour, minute, second,
47                               0, 0, 0))
48   
49   
50   def posixtime_to_sigtime(what):
51       return time.strftime('%Y%m%d%H%M%S', time.gmtime(what))

54   @dns.immutable.immutable
55   class RRSIG(dns.rdata.Rdata):
56   
57       """RRSIG record"""
58   
59       __slots__ = ['type_covered', 'algorithm', 'labels', 'original_ttl',
60                    'expiration', 'inception', 'key_tag', 'signer',
61                    'signature']
62   
63       def __init__(self, rdclass, rdtype, type_covered, algorithm, labels,
64                    original_ttl, expiration, inception, key_tag, signer,
65                    signature):
66           super().__init__(rdclass, rdtype)
67           self.type_covered = self._as_rdatatype(type_covered)
68           self.algorithm = dns.dnssec.Algorithm.make(algorithm)
69           self.labels = self._as_uint8(labels)
70           self.original_ttl = self._as_ttl(original_ttl)
71           self.expiration = self._as_uint32(expiration)
72           self.inception = self._as_uint32(inception)
73           self.key_tag = self._as_uint16(key_tag)
74           self.signer = self._as_name(signer)
75           self.signature = self._as_bytes(signature)
76   
77       def covers(self):
78           return self.type_covered
79   
80       def to_text(self, origin=None, relativize=True, **kw):
81           return '%s %d %d %d %s %s %d %s %s' % (
82               dns.rdatatype.to_text(self.type_covered),
83               self.algorithm,
84               self.labels,
85               self.original_ttl,
86               posixtime_to_sigtime(self.expiration),
87               posixtime_to_sigtime(self.inception),
88               self.key_tag,
89               self.signer.choose_relativity(origin, relativize),
90               dns.rdata._base64ify(self.signature, **kw)
91           )
92   
93       @classmethod
94       def from_text(cls, rdclass, rdtype, tok, origin=None, relativize=True,
95                     relativize_to=None):
96           type_covered = dns.rdatatype.from_text(tok.get_string())
97           algorithm = dns.dnssec.algorithm_from_text(tok.get_string())
98           labels = tok.get_int()
99           original_ttl = tok.get_ttl()
100          expiration = sigtime_to_posixtime(tok.get_string())
101          inception = sigtime_to_posixtime(tok.get_string())
102          key_tag = tok.get_int()
103          signer = tok.get_name(origin, relativize, relativize_to)
104          b64 = tok.concatenate_remaining_identifiers().encode()
105          signature = base64.b64decode(b64)
106          return cls(rdclass, rdtype, type_covered, algorithm, labels,
107                     original_ttl, expiration, inception, key_tag, signer,
108                     signature)
109  
110      def _to_wire(self, file, compress=None, origin=None, canonicalize=False):
111          header = struct.pack('!HBBIIIH', self.type_covered,
112                               self.algorithm, self.labels,
113                               self.original_ttl, self.expiration,
114                               self.inception, self.key_tag)
115          file.write(header)
116          self.signer.to_wire(file, None, origin, canonicalize)
117          file.write(self.signature)
118  
119      @classmethod
120      def from_wire_parser(cls, rdclass, rdtype, parser, origin=None):
121          header = parser.get_struct('!HBBIIIH')
122          signer = parser.get_name(origin)
123          signature = parser.get_remaining()
124          return cls(rdclass, rdtype, *header, signer, signature)
