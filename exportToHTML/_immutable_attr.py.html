3    # This implementation of the immutable decorator is for python 3.6,
4    # which doesn't have Context Variables.  This implementation is somewhat
5    # costly for classes with slots, as it adds a __dict__ to them.
6    
7    
8    import inspect
9    
10   
11   class _Immutable:
12       """Immutable mixin class"""
13   
14       # Note we MUST NOT have __slots__ as that causes
15       #
16       #    TypeError: multiple bases have instance lay-out conflict
17       #
18       # when we get mixed in with another class with slots.  When we
19       # get mixed into something with slots, it effectively adds __dict__ to
20       # the slots of the other class, which allows attribute setting to work,
21       # albeit at the cost of the dictionary.
22   
23       def __setattr__(self, name, value):
24           if not hasattr(self, '_immutable_init') or \
25              self._immutable_init is not self:
26               raise TypeError("object doesn't support attribute assignment")
27           else:
28               super().__setattr__(name, value)
29   
30       def __delattr__(self, name):
31           if not hasattr(self, '_immutable_init') or \
32              self._immutable_init is not self:
33               raise TypeError("object doesn't support attribute assignment")
34           else:
35               super().__delattr__(name)
36   
37   
38   def _immutable_init(f):
39       def nf(*args, **kwargs):
40           try:
41               # Are we already initializing an immutable class?
42               previous = args[0]._immutable_init
43           except AttributeError:
44               # We are the first!
45               previous = None
46               object.__setattr__(args[0], '_immutable_init', args[0])
47           try:
48               # call the actual __init__
49               f(*args, **kwargs)
50           finally:
51               if not previous:
52                   # If we started the initialization, establish immutability
53                   # by removing the attribute that allows mutation
54                   object.__delattr__(args[0], '_immutable_init')
55       nf.__signature__ = inspect.signature(f)
56       return nf
57   
58   
59   def immutable(cls):
60       if _Immutable in cls.__mro__:
61           # Some ancestor already has the mixin, so just make sure we keep
62           # following the __init__ protocol.
63           cls.__init__ = _immutable_init(cls.__init__)
64           if hasattr(cls, '__setstate__'):
65               cls.__setstate__ = _immutable_init(cls.__setstate__)
66           ncls = cls
67       else:
68           # Mixin the Immutable class and follow the __init__ protocol.
69           class ncls(_Immutable, cls):
70   
71               @_immutable_init
72               def __init__(self, *args, **kwargs):
73                   super().__init__(*args, **kwargs)
74   
75               if hasattr(cls, '__setstate__'):
76                   @_immutable_init
77                   def __setstate__(self, *args, **kwargs):
78                       super().__setstate__(*args, **kwargs)
79   
80           # make ncls have the same name and module as cls
81           ncls.__name__ = cls.__name__
82           ncls.__qualname__ = cls.__qualname__
83           ncls.__module__ = cls.__module__
84       return ncls
