16   class LocaleAwareNumberField(Field):
17       """ 
18       Base class for implementing locale-aware number parsing. 
19    
20       Locale-aware numbers require the 'babel' package to be present. 
21       """
22   
23       def __init__(
24           self,
25           label=None,
26           validators=None,
27           use_locale=False,
28           number_format=None,
29           **kwargs,
30       ):
31           super().__init__(label, validators, **kwargs)
32           self.use_locale = use_locale
33           if use_locale:
34               self.number_format = number_format
35               self.locale = kwargs["_form"].meta.locales[0]
36               self._init_babel()
37   
38       def _init_babel(self):
39           try:
40               from babel import numbers
41   
42               self.babel_numbers = numbers
43           except ImportError as exc:
44               raise ImportError(
45                   "Using locale-aware decimals requires the babel library."
46               ) from exc
47   
48       def _parse_decimal(self, value):
49           return self.babel_numbers.parse_decimal(value, self.locale)
50   
51       def _format_decimal(self, value):
52           return self.babel_numbers.format_decimal(value, self.number_format, self.locale)

55   class IntegerField(Field):
56       """ 
57       A text field, except all input is coerced to an integer.  Erroneous input 
58       is ignored and will not be accepted as a value. 
59       """
60   
61       widget = widgets.NumberInput()
62   
63       def __init__(self, label=None, validators=None, **kwargs):
64           super().__init__(label, validators, **kwargs)
65   
66       def _value(self):
67           if self.raw_data:
68               return self.raw_data[0]
69           if self.data is not None:
70               return str(self.data)
71           return ""
72   
73       def process_data(self, value):
74           if value is None or value is unset_value:
75               self.data = None
76               return
77   
78           try:
79               self.data = int(value)
80           except (ValueError, TypeError) as exc:
81               self.data = None
82               raise ValueError(self.gettext("Not a valid integer value.")) from exc
83   
84       def process_formdata(self, valuelist):
85           if not valuelist:
86               return
87   
88           try:
89               self.data = int(valuelist[0])
90           except ValueError as exc:
91               self.data = None
92               raise ValueError(self.gettext("Not a valid integer value.")) from exc

95   class DecimalField(LocaleAwareNumberField):
96       """ 
97       A text field which displays and coerces data of the `decimal.Decimal` type. 
98    
99       :param places: 
100          How many decimal places to quantize the value to for display on form. 
101          If None, does not quantize value. 
102      :param rounding: 
103          How to round the value during quantize, for example 
104          `decimal.ROUND_UP`. If unset, uses the rounding value from the 
105          current thread's context. 
106      :param use_locale: 
107          If True, use locale-based number formatting. Locale-based number 
108          formatting requires the 'babel' package. 
109      :param number_format: 
110          Optional number format for locale. If omitted, use the default decimal 
111          format for the locale. 
112      """
113  
114      widget = widgets.NumberInput(step="any")
115  
116      def __init__(
117          self, label=None, validators=None, places=unset_value, rounding=None, **kwargs
118      ):
119          super().__init__(label, validators, **kwargs)
120          if self.use_locale and (places is not unset_value or rounding is not None):
121              raise TypeError(
122                  "When using locale-aware numbers, 'places' and 'rounding' are ignored."
123              )
124  
125          if places is unset_value:
126              places = 2
127          self.places = places
128          self.rounding = rounding
129  
130      def _value(self):
131          if self.raw_data:
132              return self.raw_data[0]
133  
134          if self.data is None:
135              return ""
136  
137          if self.use_locale:
138              return str(self._format_decimal(self.data))
139  
140          if self.places is None:
141              return str(self.data)
142  
143          if not hasattr(self.data, "quantize"):
144              # If for some reason, data is a float or int, then format
145              # as we would for floats using string formatting.
146              format = "%%0.%df" % self.places
147              return format % self.data
148  
149          exp = decimal.Decimal(".1") ** self.places
150          if self.rounding is None:
151              quantized = self.data.quantize(exp)
152          else:
153              quantized = self.data.quantize(exp, rounding=self.rounding)
154          return str(quantized)
155  
156      def process_formdata(self, valuelist):
157          if not valuelist:
158              return
159  
160          try:
161              if self.use_locale:
162                  self.data = self._parse_decimal(valuelist[0])
163              else:
164                  self.data = decimal.Decimal(valuelist[0])
165          except (decimal.InvalidOperation, ValueError) as exc:
166              self.data = None
167              raise ValueError(self.gettext("Not a valid decimal value.")) from exc
